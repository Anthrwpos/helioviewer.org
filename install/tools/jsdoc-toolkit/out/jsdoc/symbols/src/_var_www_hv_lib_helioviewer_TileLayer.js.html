 24 	/**
 25 	 * @constructs
 26 	 * @description Creates a new TileLayer
 27 	 * @param {Object} viewport Viewport to place the tiles in
 28 	 * <br>
 29 	 * <br><div style='font-size:16px'>Options:</div><br>
 30 	 * <div style='margin-left:15px'>
 31 	 * 		<b>type</b>	       - The type of the layer (used by layer manager to differentiate event vs. tile layers)<br>
 32 	 *      <b>tileSize</b>	   - Tilesize to use<br>
 33 	 *		<b>source</b>      - Tile source ["database" | "filesystem"]<br>
 34 	 * 		<b>rootDir</b>	   - The root directory where the tiles are stored (when using filesystem as the tile source)<br>
 35 	 *      <b>opacity</b>	   - Default opacity (adjusted automatically when layer is added)<br>
 36 	 *		<b>autoOpaicty</b> - Whether or not the opacity should be automatically determined when the image properties are loaded<br>
 37 	 *		<b>startOpened</b> - Whether or not the layer menu entry should initially be open or closed<br>
 38 	 * </div>
 39 	 */
 40 	initialize: function (viewport, options) {
 41 		Object.extend(this, this.defaultOptions);
 42 		Object.extend(this, options);
 43 		this.viewport = viewport;
 44 		
 45 		this.tileSize = viewport.tileSize; 
 46 		
 47 		this.layerManager = viewport.controller.layerManager;
 48 		this.id = 'tilelayer' + new Date().getTime();
 49 
 50 		// Create an HTML container to hold the layer tiles
 51 		this.domNode = new Element('div', {className: 'tile-layer-container', style: 'position: absolute;'});
 52 		viewport.movingContainer.appendChild(this.domNode);
 53 
 54 		this.viewport.addObserver('move', this.viewportMove.bind(this));
 55 
 56 		this.tiles = [];
 57 		this.loadClosestImage();
 58 	},
 59 
 60 	/**
 61 	 * @description Refreshes the TileLayer
 62 	 */

 65 	},

 69 	 */

 74 	/**
 75 	 * @description Reload the tile layer
 76 	 * @param {Object} A 2-d binary array indicating which tiles have been (should be) loaded 
 77 	 */
 78 	reset: function (visible) {
 79 		var i, j;
 80 		
 81 		// Start loading indicator
 82 		this.viewport.controller.loadingIndicator.loadingStarted();
 83 		
 84 		// Update relevant dimensions
 85 		var zoomOffset = this.lowestRegularZoom - this.viewport.zoomLevel;
 86 		this.relWidth  = this.width  * Math.pow(2, zoomOffset);
 87 		this.relHeight = this.height * Math.pow(2, zoomOffset);
 88 
 89 		// Let user know if the requested zoom-level is lower than the lowest level natively supported
 90 		if ((this.viewport.zoomLevel < this.minZoom) && (this.viewport.controller.userSettings.get('warn-zoom-level') == "false")) {
 91 			this.viewport.controller.messageConsole.log("Note: " + this.name + " is not available at this resolution. Images will be artificially enlarged.");
 92 			this.viewport.controller.userSettings.set('warn-zoom-level', true);
 93 		}
 94 
 95 		// Remove tiles in cache
 96 		this.removeTiles();
 97 
 98 		this.refreshUTCDate();
 99 
100 		// Reference old tile nodes to remove after new ones are done loading
101 		var old = [];
102 		this.domNode.childElements().each(function (tile) {
103 			old.push(tile);
104 		});
105 
106 		//TODO: Determine range to check
107 		var numTiles = 0;
108 		var numTilesLoaded = 0;
109 
110 		var indices = this.viewport.visibleRange;
111 		
112 		for (i = indices.xStart; i <= indices.xEnd; i++) {
113 			for (j = indices.yStart; j <= indices.yEnd; j++) {
114 				if (visible[i][j]) {
115 					var tile = $(this.domNode.appendChild(this.getTile(i, j, this.viewport.zoomLevel)));
116 
117 					if (!this.tiles[i]) {
118 						this.tiles[i] = [];
119 					}
120 
121 					this.tiles[i][j] = {};
122 					this.tiles[i][j].img = tile;
123 
124 					numTiles++;
125 
126 				   // Makes sure all of the images have finished downloading before swapping them in
127 				   var self = this;
128 					Event.observe(this.tiles[i][j].img, 'load', function(e) {
129 						numTilesLoaded++;
130 						if (numTilesLoaded == numTiles) {
131 							//console.log("Finished loading ALL images! (" + numTiles + ") total.");
132 							old.each(function (tile) {
133 								tile.parentNode && tile.remove();
134 							});
135 							self.viewport.controller.loadingIndicator.loadingFinished();
136 						}
137 					});
138 				}
139 			}
140 		}
141 	},

148 		date.toUTCDate();

152 	/**
153 	 * @description Store retrieved image properties
154 	 * @param {Object} imageProperties Properties of the image associated with the TileLayer  
155 	 */
156 	setImageProperties: function (imageProperties) {
157 		//Only load image if it is different form what is currently displayed
158 		if (imageProperties.imageId === this.imageId) {
159 			this.fire('obs_time_change', this);
160 			return;
161 		}
162 		
163 		Object.extend(this, imageProperties);
164 
165 		this.fire('obs_time_change', this);
166 
167 		//IE7: Want z-indices < 1 to ensure event icon visibility
168 		this.setZIndex(parseInt(this.opacityGroupId) - 10);
169 
170 		//handle opacities for any overlapping images
171 		if (this.autoOpacity) {
172 			this.setInitialOpacity();
173 			this.autoOpacity = false;
174 		}
175 
176 		// Let others know layer has been added
177 		this.fire('change', this);
178 
179 		this.viewport.checkTiles(true);
180 
181 		this.reset(this.viewport.visible);
182 	 },

184 	/**
185 	 * @description Associates an image with the TileLayer and fetches some meta information relating to that image
186 	 * @param {String} imageId The identifier of the image to be tiled 
187 	 */
188 	setImage: function (imageId) {
189 		if (imageId === this.imageId) {
190 			return;
191 		}
192 		this.imageId = imageId;
193 		this.loadImageProperties();
194 		this.reset(this.viewport.visible);
195 	},

199 	 */
200 	setInitialOpacity: function () {
201 		var self = this;
202 		var opacity = 1;
203 		var counter = 0;
204 
205 		//Note: No longer adjust other layer's opacities... only the new layer's (don't want to overide user settings).
206 		this.layerManager.layers.each (function (layer) {
207 			if (parseInt(layer.opacityGroupId) == parseInt(self.opacityGroupId)) {
208 			   counter++;
209 			}

213 		if (counter > 1) {

215 			this.domNode.setOpacity(opacity);
216 			this.opacity = opacity * 100;
217 		}

246 	/**
247 	 * @description Loads the closest image in time to that requested
248 	 */
249 	loadClosestImage: function () {
250 		var date = this.viewport.controller.date;
251 
252 		// Ajax responder
253 		var processResponse = function (transport) {
254 			this.setImageProperties(transport.responseJSON);
255 			
256 			var hv = this.viewport.controller;
257 			
258 			// update viewport sandbox if necessary
259 			this.viewport.updateSandbox();
260 
261 			// Add to tileLayer Accordion if it's not already there
262 			if (!hv.tileLayerAccordion.hasId(this.id)) {
263 				hv.tileLayerAccordion.addLayer(this);
264 			}
265 			// Otherwise update the accordion entry information
266 			else {
267 				hv.tileLayerAccordion.updateTimeStamp(this);
268 				hv.tileLayerAccordion.updateLayerDesc(this.id, this.name);
269 				hv.tileLayerAccordion.updateOpacitySlider(this.id, this.opacity);
270 			}
271 		};
272 		
273 		// Ajax request
274 		var xhr = new Ajax.Request(this.viewport.controller.imageAPI, {
275 			method: 'POST',
276 			parameters: {
277 				action: 'getClosestImage',
278 				observatory: this.observatory,
279 				instrument:  this.instrument,
280 				detector:    this.detector,
281 				measurement: this.measurement,
282 				timestamp:   date.getTime() / 1000,
283 				debug: false				
284 			},
285 			onSuccess: processResponse.bind(this)
286 		});
287 	},

289 	/**
290 	 * @description Check to see if all visible tiles have been loaded
291 	 * @param {Object} position Position
292 	 */
293 	viewportMove: function (position) {
294 		var visible = this.viewport.visible;
295 		var indices = this.viewport.visibleRange;
296 
297 		//console.log("Checking tiles from " + indices.xStart + " to " + indices.xEnd);
298 
299 		for (var i = indices.xStart; i <= indices.xEnd; i++) {
300 			for (var j = indices.yStart; j <= indices.yEnd; j++) {
301 				if (!this.tiles[i]) {
302 					this.tiles[i] = [];
303 				}
304 				if(visible[i][j] && (!this.tiles[i][j])) {
305 					//console.log("Loading new tile");
306 					this.tiles[i][j] = $(this.domNode.appendChild(this.getTile(i, j, this.viewport.zoomLevel)));
307 				}
308 			}
309 		}
310 	},

312 	/**
313 	 * @description Generates URL to retrieve a single Tile and displays the transparent tile if request fails
314 	 * @param {Int} x Tile X-coordinate
315 	 * @param {Int} y Tile Y-coordinate
316 	 * @returns {String} URL to retrieve the requested tile
317 	 */
318 	getTile: function (x, y) {
319 		var left = x * this.tileSize;
320 		var top  = y * this.tileSize;
321 		var zoom  = this.viewport.zoomLevel;
322 		var ts = this.tileSize;
323 
324 		var img = $(new Image());
325 		img.addClassName('tile');
326 		img.setStyle({
327 			left: left + 'px',
328 			top: top + 'px'
329 		});
330 		img.unselectable = 'on';
331 		var rf = function() {
332 			return false;
333 		};
334 		img.onmousedown   = rf;
335 		img.ondrag        = rf;
336 		img.onmouseover   = rf;
337 		img.oncontextmenu = rf;
338 		img.galleryimg    = 'no';
339 		img.alt           = "";
340 
341 		
342 		// If tile doesn't exist, load the transparent tile in it's place
343 
344 		Event.observe(img, 'error', function () {
345 			this.src = 'images/transparent_' + ts + '.gif'; 
346 		})
347 
348 		// Load tile
349 		img.src = this.tileAPI + '?action=getTile&x=' + x + '&y=' + y + '&zoom=' + zoom + '&imageId=' + this.imageId + '&ts=' + ts;
350 		
351 		return img;
352 	}
