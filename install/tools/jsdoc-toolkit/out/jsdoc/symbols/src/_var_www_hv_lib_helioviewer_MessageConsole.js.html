1 /**
  2  * @fileOverview Contains the "MessageConsole" class definition.
  3  * @author <a href="mailto:keith.hughitt@gmail.com">Keith Hughitt</a>
  4  */
  5  /*global document, UIElement, Effect, $, Class */
  6 var MessageConsole = Class.create(UIElement ,
  7 	/** @lends MessageConsole.prototype */
  8 	{
  9     /**
 10      * @constructs
 11      * @description Creates a new MessageConsole.<br><br>
 12      * MessageConsole Provides a mechanism for displayed useful information to the user.
 13 	 * For ease of use, the class provides methods comparable to Firebug's for outputting
 14 	 * messages of different natures: "log" for generic unstyled messages, or for debbuging
 15 	 * use, "info" to inform the user of some interesting change or event, and "warning" and
 16 	 * "error" for getting the user's attention.
 17      * @param {Object} controller A reference to the Helioviewer (controller)
 18      * @param {String} container The id of the container for messages to be displayed in
 19      * @param {String} viewport  The id of the viewport container
 20      */
 21     initialize: function (controller, container, viewport) {
 22         this.controller = controller;
 23         this.console =  $(container);
 24         this.viewportId = viewport;
 25     },
 26     
 27 	/**
 28 	 * @description Logs a message to the message-console
 29 	 * @param {String} msg Message to display
 30 	 */
 31     log: function (msg) {
 32         this.console.update(new Element('p', {style: 'color: #6495ED; font-weight: bold;'}).insert(msg));
 33         var trash = new Effect.Appear(this.console, { duration: 3.0 });
 34     
 35         //Hide the message after several seconds have passed
 36         var self = this;
 37         window.setTimeout(function () {
 38             var trash = new Effect.Fade(self.console, { duration: 3.0 });
 39         }, 6500);
 40     },
 41     
 42     //info: function (msg) {
 43     //},
 44     
 45 	/**
 46 	 * @description Displays a warning message in the message console
 47 	 * @param {String} msg Message to display
 48 	 */
 49     warn: function (msg) {
 50         this.console.update(new Element('p', {style: 'color: yellow; font-weight: bolder;'}).insert(msg));
 51         var trash = new Effect.Appear(this.console, { duration: 3.0 });
 52     
 53         //Hide the message after several seconds have passed
 54         var self = this;
 55         window.setTimeout(function () {
 56             var trash = new Effect.Fade(self.console, { duration: 3.0 });
 57         }, 6500);        
 58     },
 59     
 60 	/**
 61 	 * @description Displays an error message in the message console
 62 	 * @param {String} msg Message to display
 63 	 */
 64     error: function (msg) {
 65         this.console.update(new Element('p', {style: 'color: red'}).insert(msg));
 66         var trash = new Effect.Shake(this.viewportId, {distance: 15, duration: 0.1});
 67         trash = new Effect.Appear(this.console, { duration: 3.0 });
 68     
 69         //Hide the message after several seconds have passed
 70         var self = this;
 71         window.setTimeout(function () {
 72            var trash = new Effect.Fade(self.console, { duration: 3.0 });
 73         }, 6500);
 74     },
 75     
 76 	/**
 77 	 * @description Displays message along with a hyperlink in the message console
 78 	 * @param {String} msg Message to display
 79 	 * @param {String} Hyperlink text (Note: Event-handler should be used to handle hyperlink clicks. The link address thus is set to "#")
 80 	 */
 81     link: function (msg, linkText) {
 82     	// Generate a temporary id
 83     	var linkId = 'link-' + this.controller.date.getTime()/1000;
 84     	
 85     	// Html
 86     	var wrapper = new Element('span');
 87     	var link = new Element('a', {href: '#', id: linkId, 'class': 'message-console-link'}).update(linkText);
 88     	wrapper.insert(msg);
 89     	wrapper.insert(link)
 90     	
 91     	this.console.update(new Element('p', {style: 'color: #6495ED;'}).insert(wrapper));
 92         var trash = new Effect.Appear(this.console, { duration: 2.0 });
 93     
 94         //For downloads, leave the message up until the user clicks on the link provided.
 95         //Note: another possibility is to add a "close" option.
 96         var self = this;
 97         Event.observe(linkId, 'click', function() {
 98 			self.console.hide();        	
 99         });
100         
101         return linkId;
102     }
103 });
