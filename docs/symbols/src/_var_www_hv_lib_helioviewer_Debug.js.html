1 /**
  2  * @author <a href="mailto:patrick.schmiedel@gmx.net">Patrick Schmiedel</a>
  3  * @fileOverview Contains the definition for a utility debugging class.
  4  * @description Contains the definition for a utility debugging class.
  5  * @classDescription This class provides basic debugging functionality to Helioviewer.
  6  */
  7 var Debug = Class.create();
  8 
  9 Debug.outputBuffer = '';
 10 Debug.outputTimeout = null;
 11 
 12 Debug.test = function (xIndex, yIndex, level) {
 13 	xIndex = 9;
 14 	level = 3;
 15 	for (var l = 0; l <= level; l++) {
 16 		var x = (xIndex >> (level - l + 1)) / (1 << l);
 17 		var x2 = 1 / (1 << l + 1);
 18 		Debug.output(x, (1 << l), (xIndex >> (level - l)), x2);
 19 	}
 20 };
 21 
 22 Debug.output = function () {
 23 	var txt = '';
 24 	
 25 	for (var i = 0; i < arguments.length; i++) {
 26 		txt += arguments[i];
 27 		if (i < arguments.length - 1) {
 28 		    txt += ', ';
 29 		}
 30 	}
 31 	
 32 	if (window.console) {
 33 		window.console.log(txt);
 34 	} else {
 35 		if (Debug.outputTimeout) {
 36 		    clearTimeout(Debug.outputTimeout);
 37 		}
 38 		Debug.outputBuffer = txt + '\n' + Debug.outputBuffer;
 39 		Debug.outputTimeout = setTimeout(Debug.flush, 100);
 40 	}
 41 };
 42 
 43 Debug.plotArray = function (a) {
 44 	Debug.output('----\n' + Debug.strArray(a, '', 0));
 45 };
 46 
 47 Debug.strArray = function (a, indent, index) {
 48 	//Debug.output(typeof(a), a instanceof Array);
 49 	if (a instanceof Array) {
 50 		if (a.length === 0) {
 51 		    return indent + index + ': []\n';
 52 		}
 53 		var txt = indent + index + ': [\n';
 54 		var c = 0;
 55 		a.each(function (m) {
 56 			txt += Debug.strArray(m, indent + '  ', c);
 57 			++c;
 58 		});
 59 		return txt + indent + ']\n';
 60 	} else {
 61 		return indent + index + ': ' + a + '\n';
 62 	}
 63 };
 64 
 65 Debug.flush = function () {
 66     if ($('debugOutput')) {
 67         $('debugOutput').innerHTML = '----\n' + Debug.outputBuffer + $('debugOutput').innerHTML;
 68         Debug.outputBuffer = '';
 69     }
 70 };
 71 
 72 Debug.ajaxFailure = function (transport, url) {
 73 	Debug.output('Error getting file "' + url + '": ' + transport.status + ' ' + transport.statusText);
 74 };
