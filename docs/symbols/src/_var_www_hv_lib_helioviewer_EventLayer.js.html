1 /**
  2  * @author <a href="mailto:keith.hughitt@gmail.com">Keith Hughitt</a>
  3  * @fileOverview Contains the class definition for an EventLayer class.
  4  * @see Layer, TileLayer
  5  *
  6  * Syntax: jQuery, Prototype
  7  */
  8 var EventLayer = Class.create(Layer,
  9 	/** @lends EventLayer.prototype */
 10 	{
 11 	/**
 12 	 * @description Default EventLayer options<br><br>
 13 	 * <strong>Notes:</strong><br><br>
 14 	 * "sunRadius0" refers to the Sun's radius at zoom-level 0, in pixels = 94 * 2^12.<br>
 15 	 * "windowSize" refers to the size of the window (in seconds) from which events should be displayed.<br>
 16 	 */
 17 	defaultOptions: {
 18 		type: 'EventLayer',
 19 		sunRadius0: 94 * (1 << 12),
 20 		opacityGroupId: -1,
 21 		displayLabels: false,
 22 		windowSize: 43200,
 23 		eventMarkerOffset: {
 24 			top: 10,
 25 			left: 0
 26 		}
 27 	},
 28 
 29 	/**
 30 	 * @constructs
 31 	 * @description Creates a new EventLayer.
 32 	 * @param {Object} viewport A reference to Helioviewer's viewport.
 33 	 * @param {Object} options The event layer settings.
 34 	 */
 35 	initialize: function (viewport, options) {
 36 		Object.extend(this, this.defaultOptions);
 37 		Object.extend(this, options);
 38 		this.viewport = viewport;
 39 
 40 		this.id = 'eventlayer' + new Date().getTime();
 41 
 42 		this.events = $A([]);
 43 
 44 		this.domNode = new Element('div', {className: 'event-layer'});
 45 		this.viewport.movingContainer.appendChild(this.domNode);
 46 	
 47 		// Add to eventLayer Accordion (if is not already there).
 48 		this.eventAccordion.addLayer(this);
 49 		
 50 		this.queryEvents();
 51 	},
 52 
 53 	/**
 54 	 * @description Sends an AJAX request to get a list of events.
 55 	 */
 56 	queryEvents: function () {
 57 		// Ajax responder
 58 		var processResponse = function (transport) {
 59 			this.clear();
 60 			if (transport.responseJSON) {
 61 				this.displayEvents(transport.responseJSON);
 62 			}			
 63 		};
 64 
 65 		// Work-around: In Firefox 3.1, Date.toISOString() Returns single-quoted strings
 66 		// http://code.google.com/p/datejs/issues/detail?id=54
 67 		var queryDate = null;
 68 		if (navigator.userAgent.search(/3\.1/) !== -1) {
 69 			queryDate = this.viewport.controller.date.toISOString();
 70 		}
 71 		else {
 72 			queryDate = this.viewport.controller.date.toISOString().slice(1, -1)
 73 		}
 74 
 75 		// Ajax request
 76 		var xhr = new Ajax.Request(this.viewport.controller.eventAPI, {
 77 			method: 'POST',
 78 			onSuccess: processResponse.bind(this),
 79 			parameters: {
 80 				action: 'getEvents',
 81 				date: queryDate,
 82 				windowSize: this.windowSize,
 83 				catalogs: this.catalog
 84 			}
 85 		});
 86 	},
 87 
 88 	/**
 89 	 * @description Place-holder for an event-handler which will handle viewportMove events.
 90 	 */
 91 	viewportMove: function () {
 92 
 93 	},
 94 	
 95 	/**
 96 	 * @description Removes all events from the screen and clears event container
 97 	 */
 98 	clear: function () {
 99 		this.events.each(function (e) {
100 			e.remove();
101 		});
102 		this.events.clear();
103 	},
104 
105 	/**
106 	 * @description Draws events to screen
107 	 * @param {JSON} jsonEvents An JSON array of event meta-information
108 	 */
109 	displayEvents: function (jsonEvents) {
110 		var self = this;
111 		var date = this.viewport.controller.date;
112 		var UTCOffset = - parseInt(date.getTimezoneOffset()) * 60;
113 		var sunRadius = this.sunRadius0 >> this.viewport.zoomLevel;
114 		
115 		// Stylize each event marker based on it's the event type
116 		$A(jsonEvents).each(function (event) {
117 			self.events.push(new EventMarker(self, event, date, UTCOffset, sunRadius, {offset: self.eventMarkerOffset}));
118 		});
119 	},
120 	
121 	/**
122 	 * @description Updates the icon associated with the EventLayer.
123 	 * @param {String} newIcon New icon to use.
124 	 */
125 	updateIcon: function (newIcon) {
126 		this.icon = "small-" + newIcon;
127 		var type = this.eventAccordion.eventCatalogs.get(this.catalog).eventType.gsub(' ', "_");
128 		var url = 'url(images/events/small-' + newIcon + '-' + type + '.png)';
129 		
130 		// Update event markers
131 		this.events.each(function(event) {
132 			event.marker.setStyle({
133 				'background': url
134 			})
135 		});
136 		
137 		// Update event accordion icon
138 		jQuery('#event-icon-' + this.id).css('background', url);
139 		
140 		// Update user's stored settings
141 		this.eventAccordion.eventIcons[this.catalog] = "small-" + newIcon;
142 		this.viewport.controller.userSettings.set('event-icons', this.eventAccordion.eventIcons);
143 	},
144 
145 	/**
146 	 * @description Toggle event label visibility
147 	 */
148 	toggleLabelVisibility: function () {
149 		this.displayLabels = !this.displayLabels;
150 		this.events.each(function(e) {
151         	e.toggleLabel();
152 		});
153 	},
154 
155 	/**
156 	 * @description Reload event-layer
157 	 */
158 	reload: function () {
159 		this.queryEvents();
160 	},
161 
162 	/**
163 	 * @description Reset event-layer
164 	 */
165 	reset: function () {
166 		var sunRadius = this.sunRadius0 >> this.viewport.zoomLevel;
167 
168 		this.events.each(function (e) {
169 			e.refresh(sunRadius);
170 		});
171 	}
172 });
