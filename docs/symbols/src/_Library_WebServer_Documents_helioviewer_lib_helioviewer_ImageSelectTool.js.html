1 /**
  2  * @author Jaclyn Beck
  3  * @fileoverview A class that deals with using the imgAreaSelect plugin, which allows the user to click and drag 
  4  * 				to select a subregion of the image in the viewport. It also currently handles taking screenshots.
  5  * syntax: prototype, jQuery ()
  6  */
  7 /*global Class, jQuery, $, $A, Ajax, Builder, Shadowbox, setTimeout, window */
  8 var ImageSelectTool = Class.create(
  9 	/** @lends ImageSelectTool.prototype */
 10     {
 11 		
 12 	/**
 13 	 * @constructs
 14 	 * @description Sets up an event handler for the select region button and finds the divs where
 15 	 * 				the fake transparent image will be inserted
 16 	 * @param {Object} controller -- the helioviewer class
 17 	 */
 18 	initialize: function (controller) {
 19 		Object.extend(this);
 20 		this.button 	= jQuery("#select-region-button");
 21 		this.bodyNode 	= $("outsideBox").parentNode;
 22 		this.controller = controller;
 23 		this.active 	= false;
 24 		this.mediaSettings 	= this.controller.mediaSettings;
 25 			
 26 		var self = this;
 27 	
 28 		this.button.click(function () {
 29 			var fullscreen = jQuery("#fullscreen-btn"), width, height, imgContainer, transImg, helioviewer;
 30 			
 31 			// If the user has already pushed the button but not done anything, this will turn the feature off.
 32 			if (self.active) {
 33 				self.cleanup();
 34 			}
 35 			// Otherwise, turn it on.
 36 			else {
 37 				// Disable keyboard shortcuts for fullscreen mode
 38 				fullscreen.addClass('requests-disabled');
 39 				self.active = true;
 40 				helioviewer = self.controller;
 41 				
 42 				// Get viewport dimensions to make the transparent image with. 
 43 				width  = helioviewer.viewport.dimensions.width;
 44 				height = helioviewer.viewport.dimensions.height;
 45 				
 46 				/* 
 47 				 * Create a temporary transparent image that spans the height and width of the viewport. Necessary because the viewport image
 48 				 * is done in tiles and imgAreaSelect cannot cross over tile boundaries.
 49 				 * Add the transparent image to the viewport, on top of the other tiles.
 50 				 * 
 51 				 * viewport.domNode corresponds to the div "#helioviewer-viewport", so add the tile directly inside this div.
 52 				 * It is necessary to specify a z-index because otherwise it gets added underneath the rest of the tiles and
 53 				 * the plugin will not work.
 54 				 */
 55 				transImg = helioviewer.viewport.domNode.appendChild(Builder.node('img', {
 56 					id	: 'transparent-image',
 57 					src	: 'images/transparent_512.png',
 58 					alt	: "",
 59 					width	: width,
 60 					height	: height,
 61 					style	: "position: relative; cursor: crosshair; z-index: 5;"
 62 				}));
 63 				
 64 				/* Make a temporary container for imgAreaSelect to put all of its divs into.
 65 				 * Note that this container must be OUTSIDE of the viewport because otherwise the plugin's boundaries
 66 				 * do not span the whole image for some reason. All of the divs are put in "#outside-box"
 67 				 */
 68 				imgContainer = self.bodyNode.appendChild(Builder.node('div', {
 69 					id: 'imgContainer'
 70 				}));
 71 				
 72 				self.selectArea(self);
 73 			}
 74 		}); 
 75 	},
 76 
 77 	/**
 78 	 * @description Loads the imgAreaSelect plugin and uses it on the transparent image that covers the viewport.
 79 	 * 				The function imgAreaSelect() returns two variables, "img", which is the original transparent image, and "selection", which
 80 	 * 				is an array describing the selected area. Available data for "selection" is x1, y1, x2, y2, width, and height.
 81 	 * 				See http://odyniec.net/projects/imgareaselect/  for usage examples and documentation. 
 82 	 * @param {Object} self
 83 	 */
 84 	selectArea: function (self) {
 85 		var helioviewer = self.controller, coords = $A(), visibleCoords = $A();
 86 		
 87 		// Use imgAreaSelect on the transparent region to get the top, left, bottom, and right coordinates of the selected region. 
 88 		jQuery("#transparent-image").imgAreaSelect({ 
 89 			handles: 	true,
 90 			parent:		"#imgContainer",
 91 /*			onSelectChange:	function (img, selection) {
 92 				helioviewer.messageConsole.info("Image dimensions: " + (selection.x2 - selection.x1) + "x" + (selection.y2 - selection.y1) + " px");
 93 			}, */
 94 			
 95 			onSelectEnd:	function (img, selection) {	
 96 				// Get the coordinates of the selected image, and adjust them to be heliocentric like the viewport coords. 
 97 				visibleCoords = helioviewer.viewport.getHCViewportPixelCoords();
 98 				
 99 				coords = {
100 					top		: visibleCoords.top  + selection.y1,
101 					left	: visibleCoords.left + selection.x1,
102 					bottom	: visibleCoords.top  + selection.y2,
103 					right	: visibleCoords.left + selection.x2
104 				};
105 				
106 				//Pop up a dialog box in shadowbox asking the user what they want to do with it.				 
107 				self.createDialog(self, coords);
108 			}
109 		});
110 	},
111 
112 	/**
113 	 * @description Loads a dialog pop-up that asks the user what they would like to do with the region they selected.
114 	 * 				Current options are: "Take Screenshot", "Build Movie", and "Cancel".
115 	 * @param {Array} coords -- an array of the heliocentric top, left, right, and bottom coordinates of the selected region, relative to the 
116 	 * 					viewport coordinates.
117 	 * @param {Object} self
118 	 */	
119 	createDialog: function (self, coords) {
120 		var helioviewer = self.controller;
121 
122 		Shadowbox.open({
123 			options:	{
124 				onFinish: function () {
125 					jQuery('#take-screenshot-button').click(function () {
126 						self.cleanup();
127 						self.takeScreenshot(helioviewer, coords);														
128 					});
129 					
130 					jQuery('#cancel-button').click(function () {
131 						self.cleanup();
132 					});
133 					
134 					jQuery('#build-movie-button').click(function () {
135 						self.cleanup();
136 						// Delay the action for one second so Shadowbox doesn't crash from opening and closing at the same time.
137 						setTimeout(function () {
138 							helioviewer.movieBuilder.checkMovieLayers(helioviewer.movieBuilder, coords);								
139 						}, 1000);				
140 					});
141 				}			
142 			},
143 			player:	'html',
144 			width: 	500,
145 			height: 160,
146 			title:	'Select Region Options',
147 			content:	'<div id="shadowbox-form" class="ui-widget-content ui-corner-all" style="margin: 10px; padding: 20px; height: 100px; font-size: 12pt;">' +
148 							'<h2 style="font-size: 14pt;">What would you like to do with the selected region?</h2>' +
149 							'<br />' +
150 							'<div id="buttons" style="text-align: left; float: right;">' +
151 								'<button id="cancel-button" class="ui-state-default ui-corner-all">Cancel</button>' +
152 								'<button id="build-movie-button" class="ui-state-default ui-corner-all">Build Movie</button>' +
153 								'<button id="take-screenshot-button" class="ui-state-default ui-corner-all">Take Screenshot</button>' +
154 							'</div>' + 
155 						'</div>'
156 		});	
157 	},
158 
159 	/**
160 	 * @description Gathers all necessary information to generate a screenshot, and then displays the image when it is ready.
161 	 * @param {Object} visibleCoords -- array containing the heliocentric top, left, bottom, and right coordinates of the visible region 
162 	 * @param {Object} helioviewer
163 	 */

165 		var xhr, imgWidth, imgHeight, url, mediaSettings, download, options, filename;
166

178 			method: 'POST',
179 			parameters: {
180 				action: 	"takeScreenshot",
181 				layers: 	(mediaSettings.layers).join("/"),
182 				obsDate: 	mediaSettings.startTime,
183 				zoomLevel: 	mediaSettings.zoomLevel,
184 				edges: 		mediaSettings.edgeEnhance,

187 				filename:	filename
188 			},
189

192 				url = transport.responseJSON;
193 
194 				// If the response is an error message instead of a url, show the message

209 						}
210 					};
211 
212 					// Create the jGrowl notification.
213 					helioviewer.messageConsole.info("<div id='screenshot-" + filename + "' style='cursor: pointer'>Click here to download '" + filename + "' </div>", options);

234 									'<img src=\"' + url + '\" width=' + imgWidth + ' height=' + imgHeight + ' ><br />' +
235 									'<div id="helioviewer-url-box" style="background:black">' +
236 									'Here is the url for your screenshot:' +
237 	            					'<form style="margin-top: 5px;"><input type="text" id="helioviewer-url-input-box" style="width:98%;" value="' + url + '"></form>' +
238 	            					'</div></div>',
239 						player:		"html",
240 			            title:      "Helioviewer Screenshot Viewer",
241 						height:     sbHeight + 65,
242 						width:      sbWidth + 20
243 			        });
244 */	
245 				}
246 			}
247 		});
248 	},
249 
250 	/**
251 	 * Removes all divs created by imgAreaSelect. Also closes Shadowbox.
252 	 * @param imgContainer -- has all imgAreaSelect divs inside
253 	 * @param transImg -- temporary transparent image that imgAreaSelect is used on.
254 	 * @TODO: add error checking if the divs are already gone for whatever reason.
255 	 */	
256 	cleanup: function () {
257 		Shadowbox.close();
258 		var fullscreen;
259 		
260 		jQuery('#imgContainer').remove();
261 		jQuery('#transparent-image').remove();
262 		
263 		this.active = false;
264 		fullscreen = jQuery('#fullscreen-btn');
265 		fullscreen.removeClass('requests-disabled');
266 	}
267 });
