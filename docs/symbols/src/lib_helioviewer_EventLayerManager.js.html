1 /**
  2  * @fileOverview Contains the class definition for an EventLayerManager class.
  3  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @see LayerManager, EventLayer
  5  * @requires LayerManager
  6  * 
  7  */
  8 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
  9 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 10 /*global LayerManager, Layer, $, Image */
 11 "use strict";
 12 var EventLayerManager = LayerManager.extend(
 13     /** @lends EventLayerManager.prototype */
 14     {
 15     
 16     /**
 17      * @constructs
 18      * @description Creates a new EventLayerManager instance
 19      */
 20     init: function (controller) {
 21         this._super(controller);
 22         this._layers = [];
 23         this._showLabels = false;
 24     },
 25     
 26     /**
 27      * @description Returns the current label visibility
 28      */
 29     getLabelVisibility: function () {
 30         return (this._showLabels === true) ? "inline" : "none";
 31     },
 32     
 33     /**
 34      * @description Toggle event label visibility
 35      */
 36     toggleLabels: function () {
 37         this._showLabels = !this._showLabels;
 38         
 39         var visible = this._showLabels;
 40 
 41         $('.event-label').each(function () {
 42             if (visible) {
 43                 $(this).show();
 44             } else {
 45                 $(this).hide();
 46             }
 47         });
 48     },
 49     
 50     /**
 51      * @description Checks for presence of a specific event catalog
 52      * @param {String} catalog Catalog ID
 53      */
 54     contains: function (catalog) {
 55         var catalogs = [];
 56         $.each(this._layers, function () {
 57             catalogs.push(this.catalog);
 58         });
 59         
 60         return ($.inArray(catalog, catalogs) !== -1);
 61     }    
 62 });
