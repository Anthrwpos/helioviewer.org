1 /**
  2  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  3  * @fileOverview Contains the class definition for an EventLayer class.
  4  * @see Layer, TileLayer
  5  */
  6 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
  7 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
  8 /*global Layer, $, EventMarker, navigator */
  9 "use strict";
 10 var EventLayer = Layer.extend(
 11     /** @lends EventLayer.prototype */
 12     {
 13     /**
 14      * @description Default EventLayer options<br><br>
 15      * <strong>Notes:</strong><br><br>
 16      * "windowSize" refers to the size of the window (in seconds) from which events should be displayed.<br>
 17      */
 18     defaultOptions: {
 19         type: 'EventLayer',
 20         layeringOrder: -1,
 21         displayLabels: false,
 22         windowSize: 43200,
 23         eventMarkerOffset: {
 24             top: 10,
 25             left: 0
 26         }
 27     },
 28 
 29     /**
 30      * @constructs
 31      * @description Creates a new EventLayer.
 32      * @param {Object} viewport A reference to Helioviewer's viewport.
 33      * @param {Object} options The event layer settings.
 34      */
 35     init: function (controller, options) {
 36         $.extend(this, this.defaultOptions);
 37         $.extend(this, options);
 38         
 39         this._super(controller);
 40 
 41         this.id = 'eventlayer' + new Date().getTime();
 42 
 43         this.events = [];
 44 
 45         this.domNode = $('<div class="event-layer"></div>').appendTo(this.viewport.movingContainer);
 46     
 47         // Add to eventLayer Accordion (if is not already there).
 48         this.eventAccordion.addLayer(this);
 49         
 50         this.queryEvents();
 51     },
 52 
 53     /**
 54      * @description Sends an AJAX request to get a list of events.
 55      */
 56     queryEvents: function () {
 57         var params, callback, self = this;
 58         
 59         params = {
 60             action     : 'getEvents',
 61             date       : this.viewport.controller.date.toISOString(),
 62             windowSize : this.windowSize,
 63             catalogs   : this.catalog
 64         };
 65         
 66         callback = function (data) {
 67             self.clear();
 68             if (data) {
 69                 self.displayEvents(data);
 70             }
 71         };
 72         
 73         $.post(this.viewport.controller.api, params, callback, "json");
 74     },
 75 
 76     /**
 77      * @description Place-holder for an event-handler which will handle viewportMove events.
 78      */
 79     viewportMove: function () {
 80 
 81     },
 82     
 83     /**
 84      * @description Removes all events from the screen and clears event container
 85      */
 86     clear: function () {
 87         $.each(this.events, function () {
 88             this.remove(); 
 89         });
 90         this.events = [];
 91     },
 92 
 93     /**
 94      * @description Draws events to screen
 95      * @param {JSON} jsonEvents An JSON array of event meta-information
 96      */
 97     displayEvents: function (jsonEvents) {
 98         var self = this,
 99             date = this.viewport.controller.getDate(),
100             rsun = this.viewport.getRSun();
101         
102         // Stylize each event marker based on it's the event type
103         $.each(jsonEvents, function () {
104             self.events.push(new EventMarker(self, this, date, rsun, {offset: self.eventMarkerOffset}));
105         });
106     },
107     
108     /**
109      * @description Updates the icon associated with the EventLayer.
110      * @param {String} newIcon New icon to use.
111      */
112     updateIcon: function (newIcon) {
113         var type, url;
114         
115         type = this.eventAccordion.eventCatalogs[this.catalog].eventType.replace(/ /g, "_");
116 
117         url = 'url(images/events/small-' + newIcon + '-' + type + '.png)';
118         
119         this.icon = "small-" + newIcon;
120         
121         // Update event markers
122         $.each(this.events, function () {
123             this.marker.css('background', url);
124         });
125         
126         // Update event accordion icon
127         $('#event-icon-' + this.id).css('background', url);
128         
129         // Update user's stored settings
130         this.eventAccordion.eventIcons[this.catalog] = "small-" + newIcon;
131         this.viewport.controller.userSettings.set('eventIcons', this.eventAccordion.eventIcons);
132     },
133 
134     /**
135      * @description Toggle event label visibility
136      */
137     toggleLabelVisibility: function () {
138         this.displayLabels = !this.displayLabels;
139         $.each(this.events, function () {
140             this.toggleLabel(); 
141         });
142     },
143 
144     /**
145      * @description Reload event-layer
146      */
147     reload: function () {
148         this.queryEvents();
149     },
150 
151     /**
152      * @description Reset event-layer
153      */
154     reset: function () {
155         var rsun = this.viewport.getRSun();
156 
157         $.each(this.events, function () {
158             this.refresh(rsun);
159         });
160     },
161     
162     // 2009/07/06 TODO
163     toString: function () {
164     },    
165     toJSON: function () {
166     }
167 });
