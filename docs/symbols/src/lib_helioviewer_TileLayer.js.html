 81     /**
 82      * @description Reload the tile layer
 83      * @param {Boolean} zoomLevelChanged Whether or not the zoom level has been changed
 84      */
 85     reset: function (zoomLevelChanged) {
 86         this.scaleFactor  = this.scale / this.viewport.getImageScale();  
 87         
 88         // Update relevant dimensions
 89         this.relWidth  = this.width  * this.scaleFactor;
 90         this.relHeight = this.height * this.scaleFactor;
 91         
 92         // Offset image
 93         this.offsetX = ((this.width  / 2) - this.x) * this.scaleFactor;
 94         this.offsetY = ((this.height / 2) - this.y) * this.scaleFactor;
 95         
 96         this.domNode.css({
 97             "left": this.offsetX,
 98             "top" : this.offsetY
 99         });
100     
101         // Update layer dimensions (only magnitude is important)
102         this.dimensions = {
103             "left"   : (this.relWidth  / 2) + this.offsetX,
104             "top"    : (this.relHeight / 2) + this.offsetY,
105             "bottom" : (this.relHeight / 2) - this.offsetY,
106             "right"  : (this.relWidth  / 2) - this.offsetX
107         };
108     
109         this.refreshUTCDate();
110         this.refreshTiles(zoomLevelChanged);
111     },

113     /**
114      * @description Loads the closest image in time to that requested
115      */
116     loadClosestImage: function () {
117         var callback, params, self = this;
118 
119         params = {
120             action:   'getClosestImage',
121             server:   this.server,
122             sourceId: this.sourceId,
123             date:     this.controller.date.toISOString()
124         };
125         
126         // Ajax responder
127         callback = function (image) {
128             var hv = self.controller;
129 
130             //Only load image if it is different form what is currently displayed
131             if (image.filename === this.filename) {
132                 return;
133             }
134             
135             $.extend(self, image);
136      
137             self.viewport.checkTiles(true);
138             self.reset(false);
139                        
140             // Update viewport sandbox if necessary
141             self.viewport.updateSandbox();
142 
143             // Add to tileLayer Accordion if it's not already there
144             if (!hv.tileLayerAccordion.hasId(self.id)) {
145                 hv.tileLayerAccordion.addLayer(self);
146             }
147                             
148             // Otherwise update the accordion entry information
149             else {
150                 hv.tileLayerAccordion.updateTimeStamp(self);
151                 hv.tileLayerAccordion.updateLayerDesc("#" + self.htmlId, self.name);
152                 hv.tileLayerAccordion.updateOpacitySlider(self.id, self.opacity);
153             }
154         };
155         
156         this._loadStaticProperties();
157         
158         // Ajax request
159         $.post(this.controller.api, params, callback, "json");
160     },

162     /**
163      * @description Refresh displayed tiles
164      * @param {Boolean} zoomLevelChanged Whether or not the zoom level has been changed
165      */
166     refreshTiles: function (zoomLevelChanged) {
167         var i, j, old, numTiles, numTilesLoaded, indices, tile, onLoadComplete, visible, self = this;
168         
169         visible = this.viewport.visible;
170 
171         this.computeValidTiles();
172         this.removeTiles();
173         old = this.getTileArray();
174 
175         // When zooming, remove old tiles right away to avoid visual glitches
176         if (zoomLevelChanged) {
177             this.removeTileDomNodes(old);
178         }
179         
180         numTiles = 0;
181         numTilesLoaded = 0;
182     
183         indices = this.viewport.visibleRange;
184         
185         // When stepping forward or back in time remove old times only after all new ones have been added
186         onLoadComplete = function () {
187             numTilesLoaded += 1;
188 
189             // After all tiles have loaded, stop indicator (and remove old-tiles if haven't already)
190             if (numTilesLoaded === numTiles) {
191                 if (!zoomLevelChanged) {
192                     self.removeTileDomNodes(old);
193                     //$('#loading').hide();
194                 }
195             }
196         };
197         
198         // Load tiles that lie within the current viewport
199         for (i = indices.xStart; i <= indices.xEnd; i += 1) {
200             for (j = indices.yStart; j <= indices.yEnd; j += 1) {
201                 if (!this.validTiles[i]) {
202                     this.validTiles[i] = [];
203                 }
204 
205                 if (visible[i][j] && this.validTiles[i][j]) {
206                     tile = this.getTile(i, j, this.viewport.zoomLevel);
207                     this.domNode.append(tile);
208     
209                     if (!this.tiles[i]) {
210                         this.tiles[i] = [];
211                     }
212     
213                     this.tiles[i][j] = {};
214                     this.tiles[i][j].img = tile;
215     
216                     numTiles += 1;
217     
218                     // Makes sure all of the images have finished downloading before swapping them in
219                     this.tiles[i][j].img.load(onLoadComplete);
220                 }
221             }
222         }        
223     },

266     /**
267      * @description Creates a 2d array representing the range of valid (potentially data-containing) tiles
268      */
269     computeValidTiles: function () {
270         var i, j, indices;
271         
272         indices = this.getValidTileRange();
273         
274         // Reset array
275         this.validTiles = [];
276         
277         // Update validTiles array
278         for (i = indices.xStart; i <= indices.xEnd; i += 1) {
279             for (j = indices.yStart; j <= indices.yEnd; j += 1) {
280                 if (!this.validTiles[i]) {
281                     this.validTiles[i] = [];
282                 }
283                 this.validTiles[i][j] = true;
284             }
285         }        
286     },

288     /**
289      * @description Determines the boundaries for the valid tile range
290      * @return {Array} An array containing the tile boundaries
291      */
292     getValidTileRange: function () {
293         var numTilesX, numTilesY, boundaries, ts = this.tileSize;
294         
295         // Number of tiles for the entire image
296         numTilesX = Math.max(2, Math.ceil(this.relWidth  / ts));
297         numTilesY = Math.max(2, Math.ceil(this.relHeight  / ts));
298         
299         // Tile placement architecture expects an even number of tiles along each dimension
300         if ((numTilesX % 2) !== 0) {
301             numTilesX += 1;
302         }
303 
304         if ((numTilesY % 2) !== 0) {
305             numTilesY += 1;
306         }
307 
308         // boundaries for tile range
309         boundaries = {
310             xStart: - (numTilesX / 2),
311             xEnd  :   (numTilesX / 2) - 1,
312             yStart: - (numTilesY / 2),
313             yEnd  :   (numTilesY / 2) - 1
314         };
315         
316         return boundaries;
317     },

319     /**
320      * @description Sets the opacity for the layer, taking into account layers which overlap one another.
321      */
322     setInitialOpacity: function () {
323         var self = this,
324             opacity = 1,
325             counter = 0;
326 
327         //Note: No longer adjust other layer's opacities... only the new layer's (don't want to overide user settings).
328         this.tileLayers.each(function () {
329             if (parseInt(this.layeringOrder, 10) === parseInt(self.layeringOrder, 10)) {
330                 counter += 1;
331             }
332         });
333         
334         //Do no need to adjust opacity if there is only one image
335         if (counter > 1) {
336             opacity = opacity / counter;
337             this.setOpacity(opacity * 100);
338         }
339         
340         this.autoOpacity = false;
341     },

398     /**

407         indices = this.viewport.visibleRange;

409         //console.log("Checking tiles from " + indices.xStart + " to " + indices.xEnd);

412                 if (!this.tiles[i]) {
413                     this.tiles[i] = [];
414                 }
415                 if (!this.validTiles[i]) {
416                     this.validTiles[i] = [];
417                 }
418                 if (visible[i][j] && (!this.tiles[i][j]) && this.validTiles[i][j]) {
419                     //console.log("Loading new tile");
420                     this.tiles[i][j] = this.getTile(i, j, this.viewport.zoomLevel).appendTo(this.domNode);
421                 }
422             }

426     /**
427      * @description Generates URL to retrieve a single Tile and displays the transparent tile if request fails
428      * @param {Int} x Tile X-coordinate
429      * @param {Int} y Tile Y-coordinate
430      * @returns {String} URL to retrieve the requested tile
431      * 
432      * IE: CSS opacities do not behave properly with absolutely positioned elements. Opacity is therefor 
433      * set at tile-level.
434      */
435     getTile: function (x, y) {
436         var top, left, zoom, ts, img, rf, emptyTile, uri, self  = this;
437 
438         left     = x * this.tileSize;
439         top      = y * this.tileSize;
440         zoom     = this.viewport.zoomLevel;
441         ts       = this.tileSize;
442         
443         rf = function () {
444             return false;
445         };
446         
447         emptyTile = 'images/transparent_' + ts + '.gif';
448             
449         //img = $('<img class="tile" style="left:' + left + 'px; top:' + top + 'px;"></img>');
450         img = new Image();
451 
452         img.unselectable = 'on';
453 
454         img.onmousedown   = rf;
455         img.ondrag        = rf;
456         img.onmouseover   = rf;
457         img.oncontextmenu = rf;
458         img.onselectstart = rf;
459         img.galleryimg    = 'no';
460 
461         
462         img = $(img).addClass("tile").css({"left": left, "top": top}).attr("alt", "");
463 
464         // IE
465         if (!$.support.opacity) {
466             img.css("opacity", this.opacity / 100);
467         }
468 
469         // If loading fails...
470         img.error(function (e) {
471             img.unbind("error");
472             
473             if (self.controller.backupEnabled) {
474             
475                 // If it still doesn't work, load the transparent tile
476                 $(this).error(function () {
477                     $(this).attr("src", emptyTile);
478                 });
479                 
480                 $(this).attr("src", self.getTileURL(self.controller.backupServer, x, y));
481                 
482             } else {
483                 $(this).attr("src", emptyTile);
484             }
485         });
486         
487         // Wait until image is done loading specify dimensions in order to prevent Firefox from displaying place-holders
488         img.load(function () {
489             $(this).width(512).height(512);
490         });
491 
492         uri = this.filepath + "/" + this.filename;
493 
494         // Load tile
495         img.attr("src", this.getTileURL(this.server, x, y));
496         
497         if (this.controller.debug) {
498             console.log(this.getTileURL(this.server, x, y));
499         }
500         
501         return img;
502     },

504     /**
505      * @description Returns a formatted string representing a query for a single tile
506      */
507     getTileURL: function (serverId, x, y) {
508         var uri, zoomLevel, format, src, offsetX, offsetY, baseURL;
509         
510         baseURL = this.controller.tileServers[serverId];
511         
512         uri    = this.filepath + "/" + this.filename;        
513         format = (this.layeringOrder === 1 ? "jpg" : "png");
514         zoomLevel = this.viewport.zoomLevel;
515         
516         src = baseURL + '?action=getTile&uri=' + uri + '&x=' + x + '&y=' + y + '&zoom=' + zoomLevel;
517         src += '&ts=' + this.tileSize + '&jp2Width=' + this.width + '&jp2Height=' + this.height + '&jp2Scale=';
518         src += this.scale + '&offsetX=' + this.offsetX + '&offsetY=' + this.offsetY + '&format=' + format;
519         src += '&obs=' + this.observatory + '&inst=' + this.instrument + '&det=' + this.detector;
520         src += '&meas=' + this.measurement;
521         
522         //console.log(src);
523         return src;
524     },

551     /**
552      * @description Tests all four corners of the visible image area to see if they are within the 
553      *              transparent circle region of LASCO C2 and LASCO C3 images. It uses the distance
554      *              formula: d = sqrt( (x2 - x1)^2 + (y2 - y1)^2 ) to find the distance from the center to 
555      *              each corner, and if that distance is less than the radius, it is inside the circle region. 
556      * @param {Object} radius -- The radius of the circle region in the jp2 image
557      * @param {Object} center -- The center coordinate of the jp2 image (jp2Width / 2). 
558      * @param {Object} left -- Left coordinate of the selected region
559      * @param {Object} top -- Top coordinate of the selected region
560      * @param {Object} width -- width of the selected region
561      * @param {Object} height -- height of the selected region
562      * @return false as soon as it finds a distance outside the radius, or true if it doesn't.
563      */
564     insideCircle: function (radius, center, left, top, width, height) {
565         var right = left + width, bottom = top + height, distance, distX, distY, corners, c;
566         corners = {
567             topLeft        : {x: left,  y: top},
568             topRight    : {x: right, y: top},
569             bottomLeft    : {x: left,  y: bottom},
570             bottomRight    : {x: right, y: bottom}
571         };
572 
573         for (c in corners) {
574             // Make JSLint happy...
575             if (true) {
576                 distX = Math.pow(center - corners[c].x, 2);
577                 distY = Math.pow(center - corners[c].y, 2);
578                 distance = Math.sqrt(distX + distY);
579                 if (distance > radius) {
580                     return false;
581                 }
582             }
583         }
584 
585         return true;
586     },
