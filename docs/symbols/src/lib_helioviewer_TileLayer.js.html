 77     /**
 78      * @description Reload the tile layer
 79      * @param {Boolean} zoomLevelChanged Whether or not the zoom level has been changed
 80      */
 81     reset: function (zoomLevelChanged) {
 82         var currentScale, scaleOffset, visible;
 83         
 84         // Start loading indicator
 85         this.viewport.controller.loadingIndicator.loadingStarted();
 86         
 87         // Update relevant dimensions
 88         //zoomOffset = this.lowestRegularZoom - this.viewport.zoomLevel;
 89         currentScale = this.viewport.controller.baseScale * Math.pow(2, this.viewport.zoomLevel - this.viewport.controller.baseZoom);
 90         scaleOffset  = this.naturalImageScale / currentScale;
 91 		//console.log("zoomLevel & baseZoom: " + this.viewport.zoomLevel + ", " + this.viewport.controller.baseZoom);
 92 		//console.log("currentScale & scaleOffset: " + currentScale + ", " + scaleOffset);
 93         this.relWidth  = this.width  * scaleOffset;
 94         this.relHeight = this.height * scaleOffset;
 95 		
 96     	//console.log("relative Width & Height: " + this.relWidth + " , " + this.relHeight + "\nWidth & Height: " + this.width + ", " + this.height);
 97     
 98     	// Let user know if the requested zoom-level is lower than the lowest level natively supported
 99     	if ((this.viewport.zoomLevel < this.minZoom) && (this.viewport.controller.userSettings.get('warnZoomLevel') === "false")) {
100     		this.viewport.controller.messageConsole.log("Note: " + this.name + " is not available at this resolution. Images will be artificially enlarged.");
101     		this.viewport.controller.userSettings.set('warnZoomLevel', true);
102     	}
103     
104     	this.refreshUTCDate();
105 
106         this.refreshTiles(zoomLevelChanged);
107     },

109     /**
110      * @description Refresh displayed tiles
111      * @param {Boolean} zoomLevelChanged Whether or not the zoom level has been changed
112      */
113     refreshTiles: function (zoomLevelChanged) {
114         var i, j, old, numTiles, numTilesLoaded, indices, tile, onLoadComplete, visible, self = this;
115         
116         visible = this.viewport.visible;
117 
118         // Determine valid tilespace
119         this.computeValidTiles();
120 
121     	// Remove tiles in cache
122     	this.removeTiles();
123     
124     	// Reference old tile nodes to remove after new ones are done loading
125     	old = this.getTileArray();
126 
127         // When zooming, remove old tiles right away to avoid visual glitches
128         if (zoomLevelChanged) {
129             this.removeTileDomNodes(old);
130         }
131         
132     	numTiles = 0;
133     	numTilesLoaded = 0;
134     
135     	indices = this.viewport.visibleRange;
136     	
137         // When stepping forward or back in time remove old times only after all new ones have been added
138     	onLoadComplete = function () {
139     		numTilesLoaded += 1;
140 
141             // After all tiles have loaded, stop indicator (and remove old-tiles if haven't already)
142     		if (numTilesLoaded === numTiles) {
143                 if (!zoomLevelChanged) {
144                     self.removeTileDomNodes(old);
145                 }
146     			self.viewport.controller.loadingIndicator.loadingFinished();
147     		}
148     	};
149     	
150         // Load tiles that lie within the current viewport
151     	for (i = indices.xStart; i <= indices.xEnd; i += 1) {
152     		for (j = indices.yStart; j <= indices.yEnd; j += 1) {
153                 if (!this.validTiles[i])
154                     this.validTiles[i] = [];
155     			if (visible[i][j] && this.validTiles[i][j]) {
156     				tile = $(this.domNode.appendChild(this.getTile(i, j, this.viewport.zoomLevel)));
157     
158     				if (!this.tiles[i]) {
159     					this.tiles[i] = [];
160     				}
161     
162     				this.tiles[i][j] = {};
163     				this.tiles[i][j].img = tile;
164     
165     				numTiles += 1;
166     
167     			   // Makes sure all of the images have finished downloading before swapping them in
168     				Event.observe(this.tiles[i][j].img, 'load', onLoadComplete);
169     			}
170     		}
171     	}        
172     },

208 	/**
209 	 * @description Store retrieved image properties
210 	 * @param {Object} imageProperties Properties of the image associated with the TileLayer  
211 	 */
212 	setImageProperties: function (imageProperties) {
213 		//Only load image if it is different form what is currently displayed
214 		if (imageProperties.uri === this.uri) {
215 			this.fire('obs_time_change', this);
216 			return;
217 		}
218 		
219 		Object.extend(this, imageProperties);
220 
221 		this.fire('obs_time_change', this);
222 
223 		//IE7: Want z-indices < 1 to ensure event icon visibility
224 		this.setZIndex(parseInt(this.opacityGroupId, 10) - 10);
225 
226         //TODO: move opacity & vis check to init avoid recomputing many times (need opacity grp)
227 		// opacity
228         if (this.opacity != 100)
229             this.setOpacity(this.opacity);
230 
231 		else if (this.autoOpacity) {
232 			this.setInitialOpacity();
233 			this.autoOpacity = false;
234 		}
235         
236         // visibility
237         if (!this.visible)
238             this.setVisibility(false);
239 
240 		// Let others know layer has been added
241 		this.fire('change', this);
242 
243 		this.viewport.checkTiles(true);
244 		this.reset(false);
245 	},

247     /**
248      * @description Creates a 2d array representing the range of valid (potentially data-containing) tiles
249      */
250     computeValidTiles: function () {
251    		var i, j, indices;
252         
253    		indices = this.getValidTileRange();
254         
255         // Reset array
256         this.validTiles = [];
257         
258         // Update validTiles array
259 		for (i = indices.xStart; i <= indices.xEnd; i += 1) {
260 			for (j = indices.yStart; j <= indices.yEnd; j += 1) {
261 				if (!this.validTiles[i]) {
262 					this.validTiles[i] = [];
263 				}
264 				this.validTiles[i][j] = true;
265 			}
266 		}        
267     },

269     /**
270      * @description Determines the boundaries for the valid tile range
271      * @return {Array} An array containing the tile boundaries
272      */
273 	getValidTileRange: function () {
274 		var numTilesX, numTilesY, boundaries, ts = this.tileSize;
275 		
276         // Number of tiles for the entire image
277 		numTilesX = Math.max(2, Math.ceil(this.relWidth  / ts));
278    		numTilesY = Math.max(2, Math.ceil(this.relHeight  / ts));
279 		
280 		// Tile placement architecture expects an even number of tiles along each dimension
281 		if ((numTilesX % 2) != 0)
282             numTilesX += 1;
283 
284 		if ((numTilesY % 2) != 0)
285 			numTilesY += 1;
286 
287 	    // boundaries for tile range
288         boundaries = {
289             xStart: - (numTilesX / 2),
290             xEnd  :   (numTilesX / 2) - 1,
291             yStart: - (numTilesY / 2),
292             yEnd  :   (numTilesY / 2) - 1
293         };
294         
295         return boundaries;
296     },

311 	/**
312 	 * @description Sets the opacity for the layer, taking into account layers which overlap one another.
313 	 */
314 	setInitialOpacity: function () {
315 		var self = this,
316 			opacity = 1,
317 			counter = 0;
318 
319 		//Note: No longer adjust other layer's opacities... only the new layer's (don't want to overide user settings).
320 		this.tileLayers.each(function (layer) {
321 			if (parseInt(layer.opacityGroupId, 10) === parseInt(self.opacityGroupId, 10)) {
322 				counter += 1;
323 			}
324 		});
325 		
326 		//Do no need to adjust opacity if there is only one image
327 		if (counter > 1) {
328 			opacity = opacity / counter;
329             this.setOpacity(opacity * 100);
330 		}
331 	},

342 	/**
343 	 * @description Loads the closest image in time to that requested
344 	 */
345 	loadClosestImage: function () {
346 		var date = this.viewport.controller.date,
347 			processResponse, xhr;
348 
349 		// Ajax responder
350 		processResponse = function (transport) {
351 			this.setImageProperties(transport.responseJSON);
352 			
353 			var hv = this.viewport.controller;
354 			           
355 			// update viewport sandbox if necessary
356 			this.viewport.updateSandbox();
357 
358 			// Add to tileLayer Accordion if it's not already there
359 			if (!hv.tileLayerAccordion.hasId(this.id)) {
360 				hv.tileLayerAccordion.addLayer(this);
361 			}
362 			// Otherwise update the accordion entry information
363 			else {
364 				hv.tileLayerAccordion.updateTimeStamp(this);
365 				hv.tileLayerAccordion.updateLayerDesc(this.id, this.name);
366 				hv.tileLayerAccordion.updateOpacitySlider(this.id, this.opacity);
367 			}
368 		};
369 		
370 		// Ajax request
371 		xhr = new Ajax.Request(this.viewport.controller.api, {
372 			method: 'POST',
373 			parameters: {
374 				action: 'getClosestImage',
375                 server:      this.server,
376 				observatory: this.observatory,
377 				instrument:  this.instrument,
378 				detector:    this.detector,
379 				measurement: this.measurement,
380 				timestamp:   date.getTime() / 1000
381 			},
382 			onSuccess: processResponse.bind(this)
383 		});
384 	},

399 	/**
400 	 * @description Check to see if all visible tiles have been loaded
401 	 */
402 	viewportMove: function () {
403 		var visible, indices, i, j;
404         
405         this.viewport.checkTiles();
406         
407       	visible = this.viewport.visible;
408 	    indices = this.viewport.visibleRange;	
409 
410 		//console.log("Checking tiles from " + indices.xStart + " to " + indices.xEnd);
411     	for (i = indices.xStart; i <= indices.xEnd; i += 1) {
412 			for (j = indices.yStart; j <= indices.yEnd; j += 1) {
413 				if (!this.tiles[i]) {
414 					this.tiles[i] = [];
415 				}
416                 if (!this.validTiles[i]) {
417                     this.validTiles[i] = [];
418                 }
419 				if (visible[i][j] && (!this.tiles[i][j]) && this.validTiles[i][j]) {
420 					//console.log("Loading new tile");
421 					this.tiles[i][j] = $(this.domNode.appendChild(this.getTile(i, j, this.viewport.zoomLevel)));
422 				}
423 			}
424 		}

427 	/**
428 	 * @description Generates URL to retrieve a single Tile and displays the transparent tile if request fails
429 	 * @param {Int} x Tile X-coordinate
430 	 * @param {Int} y Tile Y-coordinate
431 	 * @returns {String} URL to retrieve the requested tile
432 	 */
433 	getTile: function (x, y) {
434         var top, left, zoom, ts, img, rf, self, xhr, processResponse, emptyTile;
435 
436 	    left  = x * this.tileSize;
437 		top   = y * this.tileSize;
438 		zoom  = this.viewport.zoomLevel;
439 		ts    = this.tileSize;
440         self  = this;
441 		
442         rf = function () {
443 			return false;
444 		};
445         
446         emptyTile = 'images/transparent_' + ts + '.gif';
447 			
448 		img = $(new Image());
449 		img.addClassName('tile');
450 		img.setStyle({
451 			left: left + 'px',
452 			top : top + 'px'
453 		});
454 		img.unselectable = 'on';
455 
456 		img.onmousedown   = rf;
457 		img.ondrag        = rf;
458 		img.onmouseover   = rf;
459 		img.oncontextmenu = rf;
460 		img.galleryimg    = 'no';
461 		img.alt           = "";
462 
463 		
464 		// If loading fails...
465 		Event.observe(img, 'error', function(e) {
466             Event.stopObserving(img, 'error');
467             
468             // Use backup server if enabled
469             if (self.viewport.controller.backupEnabled) {
470 
471                 // If it still doesn't work, load the transparent tile
472                 Event.observe(this, 'error', function() {
473                     this.src = emptyTile;
474                 });
475                 
476                 this.src = self.viewport.controller.backupServer + '?action=getTile&x=' + x + '&y=' + y + '&zoom=' + zoom + '&uri=' + self.uri + '&ts=' + ts;
477                 
478             } else {
479                 this.src = emptyTile;
480             }
481 
482         });
483 
484 		// Load tile
485 		img.src = this.server + '?action=getTile&x=' + x + '&y=' + y + '&zoom=' + zoom + '&uri=' + this.uri + '&ts=' + ts;
486 		
487 		return img;
488 	},
