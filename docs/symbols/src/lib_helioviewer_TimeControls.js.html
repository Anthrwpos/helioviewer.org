1 /**
  2  * @fileOverview Contains the class definition for an TimeControls class.
  3  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @see Time
  5  *
  6  * TODO: Use highlight or similar effect on date and time input fields themselves when
  7  * invalid data is specified.
  8  */
  9 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
 10 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 11 /*global $, window, Class */
 12 "use strict";
 13 var TimeControls = Class.extend(
 14     /** @lends TimeControls.prototype */
 15     {
 16     /**
 17      * @constructs
 18      * @description Creates a new TimeControl component
 19      * @param {Object} controller Reference to the Helioviewer application class/controller
 20      * @param {Int} increment The amount of time to jump, in seconds, each time the back button or forward 
 21      *              button is pressed
 22      * @param {String} dateId The ID of the date form field associated with the Time.
 23      * @param {String} timeId The ID of the time form field associated with the Time.
 24      * @param {String} incrementSelect ID for the HTML element for selecting the time increment
 25      * @param {String} backBtn ID for the time "Back" button
 26      * @param {String} forwardBtn ID for the time "Forward" button
 27      */
 28     init : function (controller, increment, dateInput, timeInput, incrementSelect, backBtn, forwardBtn) {
 29         this.controller = controller;
 30         
 31         // UI components
 32         this._dateInput  = $(dateInput);
 33         this._timeInput  = $(timeInput);
 34         this._backBtn    = $(backBtn);
 35         this._forwardBtn = $(forwardBtn);
 36         this._incrementSelect = $(incrementSelect);
 37         
 38         // Initial time increment 
 39         this._timeIncrement = increment;
 40         
 41         // Update input fields
 42         this._updateInputFields();
 43         
 44         // Initialize datepicker
 45         this._initDatePicker();
 46 
 47         // Initialize event listeners
 48         this._initEvents();
 49                 
 50         // Populate select box
 51         this._addTimeIncrements();
 52     },
 53 
 54     /**
 55      * @description Updates the HTML form fields associated with the time manager.
 56      */
 57     _updateInputFields: function () {
 58         var date = this.controller.getDate();
 59 
 60         // Set text-field to the new date
 61         this._dateInput.val(date.toUTCDateString());
 62         this._timeInput.val(date.toUTCTimeString());
 63     },
 64     
 65     /**
 66      * @description returns the contents of the date input field
 67      */
 68     getDateField: function () {
 69         return this._dateInput.val();  
 70     },
 71     
 72     /**
 73      * @description returns the contents of the time input field
 74      */
 75     getTimeField: function () {
 76         return this._timeInput.val();  
 77     },
 78       
 79    /**
 80     * @description Move back one time incremement
 81     */
 82     timePrevious: function () {
 83         var newDate = this.controller.date.addSeconds(-this._timeIncrement);
 84         this.controller.date.setDate(newDate);
 85         this._updateInputFields();
 86     },
 87     
 88     /**
 89      * @function Move forward one time increment
 90      */
 91     timeNext: function () {
 92         var newDate = this.controller.date.addSeconds(this._timeIncrement);
 93         this.controller.date.setDate(newDate);
 94         this._updateInputFields();
 95     },
 96     
 97     /**
 98      * @description Populates the time increment select item
 99      */
100     _addTimeIncrements: function () {
101         var timeSteps, select, opt;
102         
103         timeSteps = [
104             {numSecs: 1,        txt: "1 Sec"},
105             {numSecs: 60,       txt: "1 Min"},
106             {numSecs: 300,      txt: "5 Mins"},
107             {numSecs: 900,      txt: "15 Mins"},
108             {numSecs: 3600,     txt: "1 Hour"},
109             {numSecs: 21600,    txt: "6 Hours"},
110             {numSecs: 43200,    txt: "12 Hours"},
111             {numSecs: 86400,    txt: "1 Day"},
112             {numSecs: 604800,   txt: "1 Week"},
113             {numSecs: 2419200,  txt: "28 Days"},
114             {numSecs: 31556926, txt: "1 Year"}
115         ];
116         
117         // Add time-steps to the select menu
118         select = this._incrementSelect;
119 
120         $(timeSteps).each(function (i, timestep) {
121             opt = $("<option value='" + timestep.numSecs + "'>" + timestep.txt + "</option>");
122             select.append(opt);
123         });
124         
125         // Select default timestep
126         $('#timestep-select').find("[value = " + this._timeIncrement + "]").attr("selected", "selected");
127 
128         // Event-handler
129         select.bind('change', this, function (e) {
130             var self = e.data;
131             self._onChange(e);
132         });
133     },
134     
135    /**
136     * @description Time-incremenet change event handler
137     * @param {Event} e Prototype Event Object
138     */
139     _onChange: function (e) {
140         this._timeIncrement = parseInt(e.target.value, 10);
141     },
142 
143     _initDatePicker: function () {
144         var btn, self = this;
145         
146         // Initialize datepicker
147         this.cal = this._dateInput.datepicker({
148             buttonImage    : 'images/blackGlass/calendar_small.png',
149             buttonImageOnly: true,
150             buttonText     : "Select a date.",
151             changeYear     : true,
152             dateFormat     : 'yy/mm/dd',
153             mandatory      : true,
154             showOn         : 'button',
155             yearRange      : '1993:2009',
156             onSelect       : function (dateStr) {
157                 window.setTimeout(function () {
158                     self.controller.date.updateDate(dateStr);
159                 }, 500);
160             }
161         });
162         
163         // Set datepicker icon
164         btn = $('.ui-datepicker-trigger');
165 
166         // Add tooltip
167         btn.attr("title", "Select an observation date.");
168         this.controller.tooltips.createTooltip($('.ui-datepicker-trigger'));
169         
170         btn.click(function () {
171             $('.ui-datepicker-trigger').qtip("hide");
172         });
173         
174         // Mouse-over effect
175         btn.hover(
176             function () {
177                 btn.attr("src", "images/blackGlass/calendar_small-hover.png");
178             },
179             function () {
180                 btn.attr("src", "images/blackGlass/calendar_small.png");                
181             }
182         );  
183     },
184 
185     /**
186      * @descriptional Initialize date and Time-related events
187      */
188     _initEvents: function () {
189         
190         // Time backward button
191         this._backBtn.bind('click', this, function (e) {
192             var self = e.data;
193             self.timePrevious();
194         });
195         
196         // Time forward button
197         this._forwardBtn.bind('click', this, function (e) {
198             var self = e.data;
199             self.timeNext();
200         });
201         
202         // Time field change
203         this._timeInput.bind('change', this, function (e) {
204             var self = e.data;
205             self.controller.date.updateTime(e.target.value);
206         });
207         
208         
209         //Date field change
210         this._dateInput.bind('change', this, function (e) {
211             var self = e.data;
212             self.controller.date.updateDate(e.target.value); 
213         });
214     },
215     
216     /**
217      * @description Returns the time increment currently displayed in the viewport.
218      * @return {int} this._timeIncrement -- time increment in secons
219      */
220     getTimeIncrement: function () {
221         return this._timeIncrement;
222     }
223 });
