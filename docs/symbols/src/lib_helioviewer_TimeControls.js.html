1 /**
  2  * @fileOverview Contains the class definition for an TimeControls class.
  3  * Syntax: Prototype
  4  * @author <a href="mailto:keith.hughitt@gmail.com">Keith Hughitt</a>
  5  */
  6 /*global TimeControls, Class, UIElement, Event, Element, $, $A */
  7 var TimeControls = Class.create(UIElement,
  8 	/** @lends TimeControls.prototype */
  9 	{
 10     /**
 11      * @constructs
 12      * @description Creates a new TimeControl component
 13      * @param {Object} controller Reference to the Helioviewer application class/controller
 14      * @param {String} incrementSelect ID for the HTML element for selecting the time increment
 15      * @param {String} backBtn ID for the time "Back" button
 16      * @param {String} forwardBtn ID for the time "Forward" button
 17      * @param {Int} timeIncrement The amount of time to jump, in seconds, each time the back button or forward button is pressed
 18      */
 19     initialize : function (controller, incrementSelect, backBtn, forwardBtn, timeIncrement) {
 20         this.controller = controller;
 21         
 22         //Private member variables
 23         this.className = "TimeControls";
 24         
 25         // Set increment 
 26         this.timeIncrement = timeIncrement;
 27 		
 28 		// Populate select box
 29 		this.addTimeIncrements(incrementSelect);
 30         
 31         // Event-handlers
 32         Event.observe(backBtn,    'click', this.timePrevious.bind(this));
 33         Event.observe(forwardBtn, 'click', this.timeNext.bind(this));
 34     },
 35     
 36     /**
 37      * @description Populates the time increment select item
 38      * @param {String} selectId The ID for the SELECT form item associated with the desired time increment
 39      */
 40     addTimeIncrements: function (selectId) {
 41 		var timeSteps, select, opt;
 42 		
 43         timeSteps = [
 44             {numSecs: 1,       txt: "1 Sec"},
 45             {numSecs: 60,      txt: "1 Min"},
 46             {numSecs: 300,     txt: "5 Mins"},
 47             {numSecs: 900,     txt: "15 Mins"},
 48             {numSecs: 3600,    txt: "1 Hour"},
 49             {numSecs: 21600,   txt: "6 Hours"},
 50             {numSecs: 43200,   txt: "12 Hours"},
 51             {numSecs: 86400,   txt: "1 Day"},
 52             {numSecs: 604800,  txt: "1 Week"},
 53             {numSecs: 2419200, txt: "28 Days"}
 54         ];
 55         
 56 		select = $(selectId);
 57 		
 58 		// Add time-steps to the select menu
 59 		$A(timeSteps).each(function (o) {
 60 			opt = new Element('option', {value: o.numSecs}).insert(o.txt);
 61 			select.insert(opt);
 62 		});
 63 		
 64 		// Select default timestep
 65 		select.select('option[value=' + this.timeIncrement + ']')[0].writeAttribute('selected', 'selected');
 66 		
 67 		// Event-handler
 68 		Event.observe(select, 'change', this.onChange.bindAsEventListener(this));
 69     },
 70     
 71    /**
 72     * @description Time-incremenet change event handler
 73     * @param {Event} e Prototype Event Object
 74     */
 75     onChange: function (e) {
 76 		this.timeIncrement = parseInt(e.target.value, 10);
 77 		this.fire('timeIncrementChange', this.timeIncrement);
 78     },
 79       
 80    /**
 81     * @description Move back one time incremement
 82     */
 83     timePrevious: function () {
 84         var newDate = this.controller.date.addSeconds(-this.timeIncrement);
 85         this.controller.setDate(newDate);
 86         this.controller.calendar.updateFields();
 87     },
 88     
 89     /**
 90      * @function Move forward one time increment
 91      */
 92     timeNext: function () {
 93         var newDate = this.controller.date.addSeconds(this.timeIncrement);
 94         this.controller.setDate(newDate);
 95         this.controller.calendar.updateFields();
 96     } 
 97 });
