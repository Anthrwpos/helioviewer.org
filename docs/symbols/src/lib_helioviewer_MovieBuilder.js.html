  9 var MovieBuilder = Class.create(UIElement, {
 10 	/*
 11 	 * @description Default options that user cannot change at the moment.
 12 	 */
 13 	defaultOptions: {
 14 		url         : "api/index.php",
 15 		numFrames   : 20,
 16 		frameRate   : 8,
 17 		sharpen     : false,
 18 		edgeEnhance : false,
 19 		format      : {win: "asf", mac: "mov", linux: "mp4"},
 20 		timeStep	: 86400
 21 	},
 22 
 23 	/**
 24 	 * @TODO Add error checking for startTime in case the user asks for a time that isn't in the database.
 25 	 * @param {Object} controller
 26 	 */	
 27 	initialize: function (controller) {
 28 		Object.extend(this, this.defaultOptions);
 29 		this.controller	= controller;
 30 		this.button		= jQuery("#movie-button");
 31 		this.viewport 	= this.controller.viewport;	
 32 		this.building	= false;
 33 		this.percent = 0;
 34 		var self = this;
 35 
 36 		this.button.click(function () {			
 37 			var helioviewer = self.controller, ok = true, visibleCoords;
 38 
 39 			if (self.building) {
 40 				ok = confirm("Warning: This will re-start the movie building process. Do you want to continue?"); //helioviewer.messageConsole.info("Your movie is already being built. The link to view the movie will appear here shortly.");
 41 			}
 42 		
 43 			if (ok) {
 44 				visibleCoords = helioviewer.viewport.getHCViewportPixelCoords(); 
 45 				self.buildMovie(visibleCoords, self);
 46 			}
 47 		});
 48 	},
 49 
 50 	buildMovie: function (visibleCoords, self) {	
 51 		var helioviewer = self.controller, zoomLevel, startTime, hqformat, layers = $A(), layerNames = $A(), layerOpacities = $A(), layerXRanges = $A(), layerYRanges = $A(), xhr, hcOffset;
 52 		self.building = true;
 53 		zoomLevel = helioviewer.viewport.zoomLevel;
 54 		// startTime is in unix timestamp format in seconds
 55 		startTime = helioviewer.date.getTime() / 1000;
 56 		hqformat = "mov";
 57 	
 58 		// Get the layer information
 59 		layers = helioviewer.getViewportInformation(visibleCoords);
 60 		layerNames 		= layers.layerNames;
 61 		layerOpacities 	= layers.layerOpacities;
 62 		layerXRanges 	= layers.layerXRanges;
 63 		layerYRanges 	= layers.layerYRanges;
 64 		hcOffset 		= layers.hcOffset;
 65 	
 66 //		var timeout = (1000*6*layerNames.length/100) * this.numFrames/10;
 67 //		this.updateProgress(timeout);	
 68 
 69 		xhr = new Ajax.Request(this.url, {
 70 			method: 'POST',
 71 			parameters: {
 72 				action: 	"buildMovie",
 73 				layers: 	layerNames.join(),
 74 				startDate: 	startTime,
 75 				timeStep: 	43200,
 76 				zoomLevel: 	zoomLevel,
 77 				numFrames: 	self.numFrames,
 78 				frameRate: 	self.frameRate,
 79 				edges: 		self.edgeEnhance,
 80 				sharpen: 	self.sharpen,
 81 				format: 	"mov",
 82 				opacity: 	layerOpacities.join(),
 83 				hcOffset: 	hcOffset.x + "," + hcOffset.y,
 84 				
 85 				// Joining these with the "/" delimiter because the x and yRanges are already separated by commas.
 86 				xRange: layerXRanges.join("/"),
 87 				yRange: layerYRanges.join("/")
 88 			},
 89 
 90 			onComplete: function (transport) {
 91 				//this.fire('video-done');
 92 				// Let user know that video is ready
 93 				var linkId, url;
 94 				linkId = helioviewer.messageConsole.link('', 'Quick-Video ready! Click here to start watching.');
 95 				url = 'api/index.php?action=playMovie&format=' + hqformat + '&url=' + transport.responseJSON;
 96 				self.building = false;
 97 		
 98 				Event.observe(linkId, 'click', function () {
 99 					Shadowbox.open({
100 						player: 'iframe',
101 						title: 'Helioviewer Movie Player',
102 						height: 650,
103 						width: 550,
104 						content: url // self.url + '?action=playMovie&format=' + hqformat + '&url=' + transport.responseJSON
105 					});
106 				}); 
107 			}
108 		}); 
109 	}, 
110

113

116 			this.percent += 1;
117 			setTimeout(function (thisObj) {
118 				thisObj.updateProgress(timeout);
119 			}, timeout, this);
120 		}	
121 	},

123 	observe: function (uielement, eventName, eventHandler) {
124 		uielement.addObserver(eventName, eventHandler.bind(this));
125 	}
126 });
