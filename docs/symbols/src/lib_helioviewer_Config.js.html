1 /**
  2  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a> 
  3  * @fileOverview This class handles the creation and validation of basic configuration parameters
  4  */
  5 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
  6 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
  7 /*global Class, $, window */
  8 "use strict";
  9 var Config = Class.extend(
 10     /** @lends Config.prototype */
 11     {
 12     /**
 13      * @description Creates a new Config. 
 14      * @constructs 
 15      */ 
 16     init: function (params) {
 17         this.params = params;
 18         
 19         this.bools  = ["distributed_tiling_enabled", "backup_enabled", "disable_cache"];
 20         this.ints   = ["build_num", "default_zoom_level", "default_timestep", "min_zoom_level",  
 21                        "max_zoom_level", "prefetch_size", "png_compression_quality", "jpeg_compression_quality",    
 22                        "bit_depth", "num_colors", "tile_pad_width", "max_movie_frames", "base_zoom_level"];
 23         this.floats = ["base_image_scale"];
 24         
 25         this.fixTypes();
 26     },
 27     
 28     /**
 29      * @description Fix types of configuration parameters
 30      */
 31     fixTypes: function () {
 32         var param, self = this;
 33 
 34         // Booleans
 35         $.each(this.bools, function () {
 36             param = self.params[this].toLowerCase();
 37              
 38             if ((param === "true") || (param === "1")) {
 39                 self.params[this] = true;
 40             }
 41             else {
 42                 self.params[this] = false;
 43             }
 44         });
 45         
 46         // Integers
 47         $.each(this.ints, function () {
 48             self.params[this] = parseInt(self.params[this], 10);
 49         });
 50         
 51         // Floats
 52         $.each(this.floats, function () {
 53             self.params[this] = parseFloat(self.params[this]);
 54         });
 55     },
 56     
 57     /**
 58      * @description Returns the configuration parameters as an associative array
 59      */
 60     toArray: function () {
 61         return {
 62             'version'           : this.params["build_num"],
 63             'defaultZoomLevel'  : this.params["default_zoom_level"],
 64             'defaultObsTime'    : this.params["default_obs_time"],
 65             'minZoomLevel'      : this.params["min_zoom_level"],
 66             'maxZoomLevel'      : this.params["max_zoom_level"],
 67             'baseZoom'          : this.params["base_zoom_level"],
 68             'baseScale'         : this.params["base_image_scale"],
 69             'prefetchSize'      : this.params["prefetch_size"],
 70             'timeIncrementSecs' : this.params["default_timestep"],
 71             'tileServers'       : this.params["tile_server"],
 72             'backupServer'      : this.params["backup_server"],
 73             'backupEnabled'     : this.params["backup_enabled"],
 74             'distributed'       : this.params["distributed_tiling_enabled"],
 75             'rootURL'           : this.params["web_root_url"]
 76         };
 77     }
 78 });
