 82     /** 
 83      * @description Pulls all necessary information from the viewport and time controls so that the user can customize
 84      *              them if they want to. These are all settings that will change whenever the viewport changes.
 85      * 
 86      * Once the information is gathered, it is stored in this object. This method is called by takeScreenshot, 
 87      * buildMovie, and by clicking the settings button.
 88 
 89      * @param {Array} visibleCoords -- heliocentric top, left, right, and bottom coordinates of either the entire 
 90      * viewport or a selected region.
 91      */
 92     getSettings: function (visibleCoords) {
 93         var helioviewer = this.controller, vpWidth, vpHeight, maxImgSize, imgWidth, imgHeight;
 94             
 95         // startTime is in unix timestamp format in seconds. date.getTime() returns milliseconds
 96         // so it needs to be divided by 1000.
 97         this.startTime  = helioviewer.date.getTime() / 1000;
 98 
 99         this.dateString = helioviewer.date.toISOString();
100         this.zoomLevel     = helioviewer.viewport.zoomLevel;
101         
102         // reset the filename. It will be built in getImageInformation()
103         this.filename = "";
104             
105         // Get the image width and height
106         vpWidth  = visibleCoords.right  - visibleCoords.left;
107         vpHeight = visibleCoords.bottom - visibleCoords.top;    
108 
109         maxImgSize = helioviewer.tileLayers.getMaxJP2Dimensions(visibleCoords.left, visibleCoords.top,
110                                                                 vpWidth, vpHeight);
111         
112         // If the image is smaller than the visibleCoords area, then just use the image's size        
113         imgWidth  = Math.round(Math.min(vpWidth,  maxImgSize.width));
114         imgHeight = Math.round(Math.min(vpHeight, maxImgSize.height));        
115         
116         // phpvideotoolkit can only use even numbers for image dimensions               
117         if (imgWidth % 2 !== 0) {
118             imgWidth += 1;
119         }
120         if (imgHeight % 2 !== 0) {
121             imgHeight += 1;
122         }
123 
124         this.width         = imgWidth;
125         this.height     = imgHeight;
126 
127         this.hcOffset = this.getHCOffset(visibleCoords);
128                                      
129         // Get the layer information
130         this.layerNames = this.getImageInformation(visibleCoords, vpWidth, vpHeight);
131         
132         // Keep track of which layers are checked (true/false). this.layerNames is modified
133         // to hold only those layers marked as "true".  
134         this.layers = this.checkLayers();    
135     },

137     /**
138      * @description Finds the jp2 image width/height, relative width/height, image coordinates, 
139      *              and hcOffset for each layer, and adds those layers that are visible to an array. Returns the array 
140      *              of layers. The filename is also built here from each layer's detector and measurement.
141      * @param {Array} visibleCoords -- heliocentric top, left, bottom, and right coordinates of either the viewport 
142      *                                 or the selected region
143      * @param {int} vpWidth -- width of the visibleCoords region (right-left)
144      * @param {int} vpHeight -- height of the visibleCoords region (bottom-top)
145      * @return {Array} layers -- array of all visible layers in string format: "obs,inst,det,meas,visible,opacity'
146      *                           x'XStart,XEnd,YStart,YEnd,offsetX,offsetY"
147      */
148     getImageInformation: function (visibleCoords, vpWidth, vpHeight) {
149         var self, layers = [], helioviewer, layerNames = [], imgCoords = [];
150         helioviewer = this.controller;
151 
152         self = this;
153         
154         // Only add visible tile layers to the array.
155         $.each(helioviewer.tileLayers, function () {
156             // "this" represents the tilelayer object
157             if (this.visible) {
158                 var left, top, right, bottom, width, height, sizeOffset, jp2Width, jp2Height,
159                     xRange, yRange, relWidth, relHeight, layer = [];
160                 
161                 // Making some aliases for clarity
162                 jp2Width     = this.width;
163                 jp2Height     = this.height;
164                 sizeOffset     = this.width / this.relWidth;
165                 relWidth     = jp2Width  / sizeOffset;
166                 relHeight     = jp2Height / sizeOffset;
167 
168                 // Convert viewport heliocentric coordinates into image coordinates
169                 left     = Math.floor(visibleCoords.left + relWidth  / 2);
170                 top     = Math.floor(visibleCoords.top  + relHeight / 2);
171                 right     = Math.floor(visibleCoords.right  + relWidth  / 2);
172                 bottom     = Math.floor(visibleCoords.bottom + relHeight / 2);
173                 
174                 /* 
175                  * Need to adjust for when 1 px on the jp2 image is not the same as 1 px on the viewport image.
176                  * Example: "left" is the pixel coordinate relative to the image in the viewport, which may 
177                  * be twice as large  as its corresponding jp2 image. The sizeOffset for this image would be 
178                  * (image width)/(2x image width) or 0.5, so to get the pixel coordinate relative to the jp2 image,
179                  * multiply "left" by .5, the sizeOffset. zoomOffset cannot be used here because images like LASCO 
180                  * C2 and C3 have different sizeOffsets than EIT images at the same zoomLevel.
181                  * If "left" or "top" is less than 0, just use 0.
182                  */
183                 left     = Math.max(left * sizeOffset, 0);
184                 top     = Math.max(top  * sizeOffset, 0);
185                 right     = right  * sizeOffset;
186                 bottom     = bottom * sizeOffset;
187         
188                 // If the calculated width is greater than the possible width, just use the possible width.
189                 width  = Math.min((right - left), jp2Width - left);
190                 height = Math.min((bottom - top), jp2Height - top);
191 
192                 // Round values off to the nearest integer, since you cannot have partial-pixels as a measurement
193                 width     = Math.round(width);
194                 height     = Math.round(height);
195                 left     = Math.round(left);
196                 top     = Math.round(top);
197                 
198                 // If the captured image just shows the black circular region of LASCO C2 
199                 // or C3, don't even use that layer.
200                 // 216 is the radius of the transparent circle in a C2 image.        
201                 if (this.detector.toString() === "C2" &&
202                     this.insideCircle(216, jp2Width / 2, left, top, width, height)) { 
203                     width = -1;
204                 }
205                 // 104 is the radius of the transparent circle in a C3 image.
206                 else if (this.detector.toString() === "C3" &&
207                          this.insideCircle(104, jp2Width / 2, left, top, width, height)) { 
208                     width = -1;
209                 }
210                             
211                 // If at least some of the image is currently visible, add that layer
212                 if (width > 0 && height > 0 && top < this.height && left < this.width) {
213                     xRange = {
214                         start:     left,
215                         size:     width
216                     };
217                     
218                     yRange = {
219                         start:     top,
220                         size:     height
221                     };
222                     
223                     // The -2 is there because 1 is added to odd-numbered dimensions for phpvideotoolkit, 
224                     // and if the dimensions are *.5, they are rounded up, making a difference of 2.
225                     // If this layer is the biggest layer, calculate the new image coords based around this layer.
226                     if (relWidth >= self.width - 2 || relHeight >= self.height - 2) {
227                         imgCoords = self.getImgCoords(xRange, yRange, jp2Width, jp2Height, relWidth, relHeight);
228                     }
229 
230                     self.filename += this.detector.toString() + "_" + this.measurement.toString() + "_";
231 
232                     // Don't make a string out of the layer yet, because more work needs to be done on it and 
233                     // making a string and breaking it apart a couple lines further down is unnecessary.
234                     layer = {
235                         name     : this.toString(),
236                         xRange     : xRange,
237                         yRange     : yRange,
238                         relWidth : relWidth,
239                         relHeight: relHeight,
240                         jp2Width : jp2Width,
241                         jp2Height: jp2Height
242                     };
243 
244                     layers.push(layer);
245                 }
246             }
247         });
248 
249         // Filename should now be: "det1_meas1_det2_meas2_zZoomlevel_widthxheight"
250         self.filename += "z" + self.zoomLevel + "_" + self.width + "x" + self.height;
251 
252         // Cannot do this function until the largest layer is found, so let the above loop finish and start a new one.
253         // This loop finds each layer's hcOffset and then pushes all the layer info into a string.
254         $.each(layers, function (i, l) {
255             var distX = 0, distY = 0;
256             /* 
257              * If the actual image is smaller than the original extracted region, need to adjust the hcOffset 
258              * to reflect this.
259              * The hcOffset = (distance from center to top left corner). The top left corner may be outside the 
260              * real jp2 image's top left corner.
261              * To calculate the distance: Assume the top left corner of the jp2 image, relative to the center, 
262              * is relWidth / 2. Next, we know that the center is a certain percent of the whole visibleCoords 
263              * width/height, so let's say it's 50% to the right and 50% to the bottom. Assume the actual image 
264              * width has the same proportions, but is just smaller. Therefore, the center is still 50% to the
265              * right and 50% to the bottom. So to find the distance between the center of the image and the 
266              * top left corner, multiply the image width/height * 50%. 
267              * 
268              * Next, adjust for the part of the jp2 image that already has the center in it. You don't want to pad
269              * the whole distance between top left corner and center, or the jp2 image's top left corner will be 
270              * where the center is. So subtract relWidth / 2 from the distance to get the final, adjusted hcOffset.
271              * 
272              * If the xRange.start is NOT 0, or the yRange.start is NOT 0, this adjustment doesn't even apply 
273              * because the upper-left corner is already inside the image, not somewhere outside it, so the 
274              * hcOffset is calculated normally.
275              */    
276             if (l.xRange.start === 0) {
277                 distX = Math.floor(imgCoords.left + l.relWidth / 2);
278             }    
279             
280             if (l.yRange.start === 0) {
281                 distY = Math.floor(imgCoords.top + l.relHeight / 2);
282             }            
283             
284             layerNames.push(l.name + "x" + l.xRange.start + "," + l.xRange.size + "," + l.yRange.start + "," + 
285                             l.yRange.size + "," + distX + "," + distY);
286         });
287 
288         return layerNames;    
289     },

291     /**
292      * How far away the heliocenter is from the top left corner of the viewport, and in which direction.
293      * Note that the convert command reverses "+" and "-" for the x-direction, 
294      * so "+" means push the image to the left, "-" means push it to the right.
295      * A +x means the heliocenter (0,0) is left of the top-left corner, and a +y means
296      * the heliocenter is above the top-left corner.
297      */    
298     getHCOffset: function (visibleCoords) {
299         var offset = {
300             x: visibleCoords.left,
301             y: visibleCoords.top
302         };
303             
304         return offset;
305     },

307     /**
308      * @description Layers have to stay in the order they appear in the viewport. 
309      * This function checks to see if any layers that exist in this.layerNames are 
310      * marked as "false" in this.layers. If they are, save them as "false". Otherwise,
311      * save the layer as "true". 
312      */
313     checkLayers: function () {
314         var matched, newLayers, baseLayers, base,
315             layers = [],
316             i = 0,
317             self = this;
318         
319         if (!this.layers) {
320             $.each(this.layerNames, function (i, layer) {
321                 
322                 // Get just the "obs,inst,det,meas" part of the string.
323                 var name = layer.split(",").slice(0, 4).join(",");
324                 
325                 // Set its checked value to true.
326                 i += 1;
327                 layers[i] = name + " true";
328             });
329 
330             return layers;
331         }    
332         
333         else {
334             matched = false;
335             newLayers = [];
336             baseLayers = [];
337             base = "";
338 
339             // Get just the "obs,inst,det,meas" part of each string. We don't want to compare
340             // the whole string because opacity, xrange, etc might differ but the layers are 
341             // still the same. 
342             $.each(this.layerNames, function (i, l) {
343                 i += 1;
344                 baseLayers[i] = l.split(",").slice(0, 4).join(",");
345             });
346 
347             // Check to see if each layer from baselayers has a pre-set checked value in layers.
348             // If does, save that value. If not, set 'checked' to true. 
349             $.each(baseLayers, function (i, bl) {
350                 var base = bl.split(",").slice(0, 4).join(",");
351                 
352                 // If it exists in layers and is set to false, add it as false to newLayers
353                 matched = $.grep(self.layers, function (item, i) {
354                     if (item === (base + " false")) {
355                         return true;
356                     }
357                 });
358                 
359                 if (matched.length > 0) {
360                     layers[i] = base + " false";
361                 }
362                 
363                 // Otherwise set it to true
364                 else {
365                     layers[i] = base + " true";
366                     newLayers.push(self.layerNames[i]);
367                 }
368             });
369             this.layerNames = newLayers;
370         }    
371         return layers;
372     },

374     /**
375      * @description Finds the image coordinates of the biggest layer in the set, since all other layers will
376      *                 be padded according to the biggest layer's coordinates. 
377      *                 This is done because in cases where the user selects a region that is larger than the actual
378      *                 image, the image willcrop by centering around the actual image. See <picture> 
379      *                 for a more detailed explanation.
380      * @param {Array} xRange -- xRange of biggest layer. [start, size]
381      * @param {Array} yRange -- yRange of biggest layer. [start, size]
382      * @param {int} jp2Width -- width of the jp2 image
383      * @param {int} jp2Height -- height of the jp2 image
384      * @param {int} relWidth -- relative width of the jp2 image
385      * @param {int} relHeight -- relative height of the jp2 image
386      */    
387     getImgCoords: function (xRange, yRange, jp2Width, jp2Height, relWidth, relHeight) {
388         var relXSize, relXStart, relYSize, relYStart,
389             imgCoords = [], 
390             xStart    = xRange.start,
391             xSize     = xRange.size,
392             yStart    = yRange.start,
393             ySize     = yRange.size;
394             
395         imgCoords = {
396             left   : 0,
397             top    : 0,
398             right  : 0,
399             bottom : 0
400         };
401         
402         // If the image starts at 0, the right side of the image is either inside the jp2 image or its border.
403         // Get the image coordinates starting with the right side and going to the left.
404         // See <picture>. 
405         if (xStart === 0) {
406             relXSize = xSize * relWidth / jp2Width;
407             imgCoords.right = relXSize - relWidth / 2;
408             imgCoords.left     = imgCoords.right - this.width;
409         }
410         // Otherwise, do the opposite, start on the left side and go toward the right.
411         else {
412             relXStart = xStart * relWidth / jp2Width;
413             imgCoords.left = relXStart - relWidth / 2;
414             imgCoords.right = imgCoords.left + this.width;
415         }
416         // Same with the top and bottom.
417         if (yStart === 0) {
418             relYSize = ySize * relHeight / jp2Height; 
419             imgCoords.bottom = relYSize - relHeight / 2;
420             imgCoords.top = imgCoords.bottom - this.height;
421         }
422         else {
423             relYStart = yStart * relHeight / jp2Height;
424             imgCoords.top = relYStart - relHeight / 2;
425             imgCoords.bottom = imgCoords.top + this.height;
426         }        
427 
428         return imgCoords;
429     },

431     /**
432      * @description Passes information to settings.php so that it can print out the html markup needed for the form.
433      * When done, it passes the markup to printContent to open Shadowbox.
434      * @param {Object} self
435      */
436     initSettingsDialog: function (self) {
437         var params, callback, startTime = self.startTime, endTime = new Date();
438 
439         // setTime() takes milliseconds, so multiply answer * 1000.
440         endTime.setTime((startTime + self.numFrames * self.timeStep) * 1000);
441         self.endDateString = endTime.toISOString();
442         
443         callback = function (contents) {
444             self.printContent(self, contents);
445         };
446         
447         params = {
448             startDate    : self.dateString,
449             endDate        : self.endDateString,
450             hqFormat    : self.hqFormat,
451             width        : self.width,
452             height        : self.height,
453             layers        : self.layers.join("/"),
454             timeStep    : self.timeStep,    
455             quality        : self.quality,
456             numFrames    : self.numFrames,
457             showImgIfGap: self.showImgIfGap,
458             emailUser    : self.emailUser,
459             emailAddress: self.emailAddress,
460             filename    : self.filename
461         };
462         
463         $.post('dialogs/settings.php', params, callback, "json");
464     },
