 10 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
 11 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 12 /*global LayerManager, TileLayer, Layer, $ */
 13 "use strict";
 14 var TileLayerManager = LayerManager.extend(
 15     /** @lends TileLayerManager.prototype */
 16     {
 17    
 18    /**
 19     * @constructs
 20     * @description Creates a new TileLayerManager instance
 21     */
 22     init: function (controller) {
 23         this._super(controller);
 24         this._layers = [];
 25         this._queue  = [
 26             "SOHO,EIT,EIT,304",
 27             "SOHO,LASCO,C2,white light",
 28             "SOHO,LASCO,C3,white light",
 29             "SOHO,LASCO,C2,white light",
 30             "SOHO,MDI,MDI,magnetogram",
 31             "SOHO,MDI,MDI,continuum",
 32             "SOHO,EIT,EIT,171",
 33             "SOHO,EIT,EIT,284",
 34             "SOHO,EIT,EIT,195"
 35         ];
 36     },
 37    
 38     /**
 39      * @description Updates the list of loaded tile layers stored in cookies
 40      */
 41     save: function () {
 42         var layers = this.toJSON();        
 43         this.controller.userSettings.set('tileLayers', layers);
 44     },
 45     
 46     /**
 47      * @description Adds a layer that is not already displayed
 48      */
 49     addNewLayer: function () {
 50         var currentLayers, next, rand, layerSettings, queue, defaultLayer = "SOHO,EIT,EIT,171";
 51         
 52         queue = this._queue;
 53         
 54         // current layers in above form
 55         currentLayers = [];
 56         $.each(this._layers, function () {
 57             currentLayers.push(this.observatory + "," + this.instrument + "," + this.detector + "," + this.measurement);
 58         });
 59         
 60         // remove existing layers from queue
 61         queue = $.grep(queue, function (item, i) {
 62             return ($.inArray(item, currentLayers) === -1);
 63         });
 64         
 65         //$.each(currentLayers, function() {
 66         //    queue = queue.without(this);
 67         //});
 68         
 69         // Pull off the next layer on the queue
 70         next = queue[0] || defaultLayer;
 71         layerSettings = this.controller.userSettings.parseLayerString(next + ",1,100");
 72         
 73         // Select tiling server if distributed tiling is enabling
 74         if ((this.controller.distributed === true) && ((this.size() % 2) === 0)) {
 75             rand = Math.floor(Math.random() * (this.controller.tileServers.length - 1)) + 1;
 76             layerSettings.server = rand;
 77         }
 78         else {
 79             layerSettings.server = 0;
 80         }
 81 
 82         // Open menu by default
 83         layerSettings.startOpened = true;
 84         
 85         // Add the layer
 86         this.addLayer(new TileLayer(this.controller, layerSettings));
 87         this.save();
 88     },
 89      
 90     /**
 91      * @description Returns the largest width and height of any layers (does not have to be from same layer)
 92      * @return {Object} The width and height of the largest layer
 93      */
 94     getMaxDimensions: function () {
 95         var maxLeft   = 0,
 96             maxTop    = 0,
 97             maxBottom = 0,
 98             maxRight  = 0;
 99         
100         $.each(this._layers, function () {
101             // Ignore if the relative dimensions haven't been retrieved yet
102             if ($.isNumber(this.relWidth)) {
103                 var d = this.getDimensions();
104                 
105                 maxLeft   = Math.max(maxLeft, d.left);
106                 maxTop    = Math.max(maxTop, d.top);
107                 maxBottom = Math.max(maxBottom, d.bottom);
108                 maxRight  = Math.max(maxRight, d.right);
109             }
110         });
111         
112         return {width: maxLeft + maxRight, height: maxTop + maxBottom};
113     },
114 
115     /**
116      * @description Gets the maximum relative width and height of all visible layers, according to jp2 image sizes,
117      *              not tilelayer sizes. Used when generating movies and screenshots, because tilelayer size is 
118      *              slightly smaller than jp2 image size and images will not align properly with tilelayer sizes.
119      * @return {Array} dimensions -- maximum width and height found.
120      * 
121      * NOTE 02/01/2010: This may longer necessary since the actual JP2 dimensions are now
122      * returned during each getClosestImage request?
123      */
124     getMaxJP2Dimensions: function (left, top, width, height) {
125         var dimensions, insideCircleC2, insideCircleC3, sizeOffset, relWidth, relHeight,
126             maxWidth = 0, maxHeight = 0;
127         
128         $.each(this._layers, function () {
129             if (this.visible) {
130                 insideCircleC2 = this.insideCircle(216, this.width / 2, left, top, width, height);
131                 insideCircleC3 = this.insideCircle(104, this.width / 2, left, top, width, height);
132                 
133                 if (this.detector.toString === "C2" && insideCircleC2) {
134                     // Do nothing
135                 }
136                 if (this.detector.toString === "C3" && insideCircleC3) {
137                     // Do nothing
138                 }
139                 
140                 else {
141                     sizeOffset = this.width / this.relWidth;
142                     
143                     relWidth  = this.width  / sizeOffset;
144                     relHeight = this.height / sizeOffset;
145                     
146                     maxWidth = Math.max(maxWidth, relWidth);
147                     maxHeight = Math.max(maxHeight, relHeight);
148                 }
149             }
150         });
151 
152         dimensions = {
153             width  : maxWidth,
154             height : maxHeight
155         };
156 
157         return dimensions;
158     },
159     
160     /**
161      * @description Generate a string of URIs for use by JHelioviewer
162      */
163     toURIString: function () {
164         var str = "";
165         
166         $.each(this._layers, function () {
167             str += this.uri + ",";
168         });
169         
170         // Remove trailing comma
171         str = str.slice(0, -1);
172         
173         return str;
174     }    
175 });
