1 /**
  2  * @fileOverview Contains the class definition for an TileLayerManager class.
  3  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @see LayerManager, TileLayer
  5  * @requires LayerManager
  6  * Syntax: Prototype
  7  * 
  8  */
  9 /*global TileLayerManager, LayerManager, Class, Layer, Ajax, Event, $, Element, Image */
 10 var TileLayerManager = Class.create(LayerManager,
 11     /** @lends TileLayerManager.prototype */
 12    {
 13    
 14    /**
 15     * @constructs
 16     * @description Creates a new TileLayerManager instance
 17     */
 18    initialize: function($super, controller) {
 19         $super(controller);
 20         this._layers = $A([]);
 21         this._queue  = $A([
 22 			"SOH,EIT,EIT,304",
 23             "SOH,LAS,0C2,0WL",
 24             "SOH,LAS,0C3,0WL",
 25             "SOH,LAS,0C2,0WL",
 26             "SOH,MDI,MDI,mag",
 27             "SOH,MDI,MDI,int",
 28             "SOH,EIT,EIT,171",
 29             "SOH,EIT,EIT,284",
 30             "SOH,EIT,EIT,195"
 31 		]);
 32    },
 33    
 34     /**
 35      * @description Updates the list of loaded tile layers stored in cookies
 36      */
 37     save: function () {
 38         var layers = this.toJSON();    	
 39     	this.controller.userSettings.set('tileLayers', layers);
 40     },
 41     
 42 	/**
 43 	 * @description Adds a layer that is not already displayed
 44 	 */
 45 	addNewLayer: function () {
 46 		var currentLayers, next, layerSettings, defaultLayer = "SOH_EIT_EIT_171";
 47 		
 48 		// current layers in above form
 49 		currentLayers = $A([]);
 50 		this._layers.each(function (l) {
 51 			currentLayers.push(l.observatory + "," + l.instrument + "," + l.detector + "," + l.measurement);
 52 		});
 53 		
 54 		// remove existing layers from queue
 55 		currentLayers.each(function(id) {
 56 			this._queue = this._queue.without(id);
 57 		});
 58 		
 59         // Pull off the next layer on the queue
 60 		next = this._queue.first() || defaultLayer;
 61         layerSettings = this.controller.userSettings.parseLayerString(next + ",1,100");
 62         
 63         // Selected tiling server if distributed tiling is enabling
 64         if ((this.controller.distributed === true) && (this.size() === 0))
 65             layerSettings.server = this.controller.tileServer2;
 66         else
 67             layerSettings.server = this.controller.tileServer1;
 68 
 69         // Open menu by default
 70         layerSettings.startOpened = true;
 71 		
 72         // Add the layer
 73 		this.addLayer(new TileLayer(this.controller.viewport, layerSettings));
 74 		this.save();
 75 	},
 76     
 77     
 78 	/**
 79 	 * @description Returns the largest width and height of any layers (does not have to be from same layer)
 80 	 * @returns {Object} The width and height of the largest layer
 81 	 */
 82 	getMaxDimensions: function () {
 83 		var maxWidth  = 0,
 84 			maxHeight = 0;
 85 		
 86 		this._layers.each(function (l) {
 87 			if (l.type === "TileLayer") {
 88 				// Ignore if the relative dimensions haven't been retrieved yet
 89 				if (Object.isNumber(l.relWidth)) {
 90 					maxWidth  = Math.max(maxWidth,  l.relWidth);
 91 					maxHeight = Math.max(maxHeight, l.relHeight); 
 92 				}
 93 			}
 94 		});
 95 		
 96 		return {width: maxWidth, height: maxHeight};
 97 	}
 98 });
