1 /**
  2  * @author <a href="mailto:vincent.k.hughitt@nasa.gov">Keith Hughitt</a> 
  3  * @fileOverview This class extends the base jQuery UI datepicker to provide some basic data-binding and enable the class to work more efficiently with Helioviewer.
  4  * @see ui.datepicker.js
  5  * 
  6  * Syntax: jQuery, Prototype
  7  */
  8 /*global Class, Calendar, $, UIElement, jQuery, window */
  9 var Calendar = Class.create(UIElement,
 10 	/** @lends Calendar.prototype */
 11 	{
 12 	/**
 13 	 * @description Creates a new Calendar. 
 14 	 * @param {Object} controller Reference to the controller class (Helioviewer).
 15 	 * @param {String} dateFieldId The ID of the date form field associated with the Calendar.
 16 	 * @param {String} timeFieldId The ID of the time form field associated with the Calendar.
 17 	 * @constructs 
 18 	 */ 
 19     initialize: function (controller, dateFieldId, timeFieldId) {
 20         this.controller = controller;
 21         this.dateField = $(dateFieldId);
 22         this.timeField = $(timeFieldId);
 23         
 24         var self = this;
 25         this.cal = jQuery("#" + dateFieldId).datepicker({
 26             buttonImage: 'images/blackGlass/calendar_small.png',
 27             buttonImageOnly: true,
 28 			buttonText: "Select a date.",
 29             changeYear: true,
 30             dateFormat: 'yy/mm/dd',
 31             mandatory: true,
 32             showOn: 'both',
 33             yearRange:  '1998:2009',
 34             onSelect: function (dateStr) {
 35                 window.setTimeout(function () {
 36                     var time = self.timeField.value,
 37                     	date = Date.parse(dateStr),
 38                     	
 39 						//Factor in time portion of timestamp
 40 						hours =   parseInt(time.substring(0, 2), 10),
 41                     	minutes = parseInt(time.substring(3, 5), 10),
 42                     	seconds = parseInt(time.substring(6, 8), 10),
 43                     
 44                     	//Convert to UTC
 45                     	utcDate = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate(), hours, minutes, seconds));
 46                     
 47                     self.fire('observationDateChange', utcDate);
 48                 }, 500);
 49                 
 50             }
 51         });
 52         
 53         // Mouse-over effect
 54         jQuery('.ui-datepicker-trigger').hover(
 55             function() {
 56                 jQuery(this).attr("src", "images/blackGlass/calendar_small-hover.png");
 57             },
 58             function () {
 59                 jQuery(this).attr("src", "images/blackGlass/calendar_small.png");                
 60             }
 61         );
 62     },
 63     
 64 	/**
 65 	 * @description Updates the HTML form fields associated with the calendar.
 66 	 */
 67     updateFields: function () {
 68         // Set text-field to the new date
 69         this.dateField.value = this.controller.date.toYmdUTCString();
 70         this.timeField.value = this.controller.date.toHmUTCString();
 71     }
 72 });
