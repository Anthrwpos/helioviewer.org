1 /**
  2  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a> 
  3  * @fileOverview This class handles the management of all date and time related
  4  * components in Helioviewer including the input fields, datepicker, and time-forward and
  5  * backward buttons.
  6  * 
  7  * @see ui.datepicker.js, TimeControls
  8  */
  9 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
 10 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 11 /*global Class, $, window */
 12 "use strict";
 13 var Time = Class.extend(
 14     /** @lends Time.prototype */
 15     {
 16     /**
 17      * @description Creates a new Time. 
 18      * @param {Object} controller Reference to the controller class (Helioviewer).
 19      * @constructs 
 20      */ 
 21     init: function (controller) {
 22         this.controller = controller;
 23         this._date = new Date(this.controller.userSettings.get('date'));
 24     },
 25     
 26     /**
 27      * @description Returns the current observation date as a JavaScript Date object
 28      */    
 29     getDate: function () {
 30         return this._date; 
 31     },
 32     
 33     /**
 34      * @description Sets the desired viewing date and time.
 35      * @param {Date} date A JavaScript Date object with the new time to use
 36      */
 37     setDate: function (date) {
 38         this._date = date;
 39         var ts = Date.UTC(date.getUTCFullYear(), date.getUTCMonth(), date.getUTCDate(),
 40                           date.getUTCHours(), date.getUTCMinutes(), date.getUTCSeconds());
 41         this.controller.userSettings.set('date', parseInt(ts, 10));
 42         this.controller.tileLayers.reloadLayers();
 43         this.controller.eventLayers.reloadLayers();
 44     },
 45     
 46     /**
 47      * @description Increments the current time by the specified number of seconds
 48      * @param {Integer} seconds
 49      */
 50     addSeconds: function (seconds) {
 51         return this._date.addSeconds(seconds);
 52     },
 53     
 54     /**
 55      * @description Returns a unix timestamp for the current observation time
 56      */
 57     getTime: function () {
 58         return this._date.getTime();  
 59     },
 60     
 61     /**
 62      * @description Gets an ISO 8601 string representation of the current observation time
 63      */
 64     toISOString: function () {
 65         // Work-around: In Firefox 3.1+ (and Webkit), Date.toISOString() Returns single-quoted strings
 66         // http://code.google.com/p/datejs/issues/detail?id=54
 67         var isoString = this._date.toISOString().replace(/"/g, '');
 68         return isoString;
 69     },
 70     
 71 
 72     
 73     /**
 74      * @description Updates the observation date
 75      * @param {Object} dateStr
 76      */
 77     updateDate: function (dateStr) {
 78         var date, time, hours, minutes, seconds, utcDate;
 79         
 80         //Check to see if the input is a valid time
 81         if (dateStr.match(/^\d{4}\/\d{2}\/\d{2}?/) && (dateStr.length === 10)) {
 82             date = Date.parse(dateStr);
 83             time = this.controller.timeControls.getTimeField();
 84             
 85             //Factor in time portion of timestamp
 86             hours = parseInt(time.substring(0, 2), 10);
 87             minutes = parseInt(time.substring(3, 5), 10);
 88             seconds = parseInt(time.substring(6, 8), 10);
 89             
 90             //Convert to UTC
 91             utcDate = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate(), hours, minutes, seconds));
 92             
 93             this.setDate(utcDate);
 94         }
 95         else {
 96             this.controller.messageConsole.warn('Invalid date. Please enter a date of the form YYYY/MM/DD.');
 97         }
 98     },
 99 
100     
101     /**
102      * @description takes a string of the form "hh:mm:ss" and after validating it, sets the
103      * new application time to it.
104      * @param {String} time - The new time to use
105      */
106     updateTime: function (time) {
107         var newTime, hours, mins, secs;
108         
109         //Check to see if the input is a valid time
110         if (time.match(/^\d{2}:\d{2}:\d{2}?/) && (time.length === 8)) {
111         
112             //Get the difference in times and add to this.date
113             newTime = time.split(':');
114             hours = parseInt(newTime[0], 10) - this._date.getUTCHours();
115             mins = parseInt(newTime[1], 10) - this._date.getUTCMinutes();
116             secs = parseInt(newTime[2], 10) - this._date.getUTCSeconds();
117             
118             this._date.addHours(hours);
119             this._date.addMinutes(mins);
120             this._date.addSeconds(secs);
121             
122             this.setDate(this._date);
123         }
124         else {
125             this.controller.messageConsole.warn('Invalid time. Please enter a time of the form HH:MM:SS.');
126         }
127     }
128 });
