1 /**
  2  * @author Jaclyn Beck
  3  * @fileoverview Contains the code for the Screenshot Builder class. Handles event listeners for the screenshot button and
  4  * 				 screenshot creation.
  5  * Syntax: Prototype, jQuery ()
  6  */
  7 var ScreenshotBuilder = Class.create(
  8 	/** @lends ScreenshotBuilder.prototype */
  9     {
 10 
 11 	/**
 12 	 * @constructs
 13 	 * @description Loads default options, grabs mediaSettings, sets up event listener for the screenshot button
 14 	 * @param {Object} controller -- the helioviewer class 
 15 	 */	
 16 	initialize: function (controller) {
 17 		Object.extend(this);
 18         this.button    	= jQuery("#screenshot-button");
 19 		this.building 	= false;
 20 		this.viewport	= controller.viewport;
 21 		this.controller = controller;
 22 		
 23 		this.mediaSettings = this.controller.mediaSettings;
 24 		
 25 		var self = this, visibleCoords;
 26 		this.button.click(function () {
 27 			if (self.building) {
 28 				self.messageConsole.info("A link to your screenshot will be available shortly.");
 29 			}
 30 			else {
 31 				visibleCoords = self.viewport.getHCViewportPixelCoords();
 32 				self.takeScreenshot(visibleCoords);
 33 			}
 34 		});
 35 	},
 36 	
 37 	/**
 38 	 * @description Gathers all necessary information to generate a screenshot, and then displays the image when it is ready.
 39 	 * @param {Object} visibleCoords -- array containing the heliocentric top, left, bottom, and right coordinates of the visible region 
 40 	 */
 41 	takeScreenshot: function (visibleCoords) {
 42 		var helioviewer = this.controller, self, xhr, imgWidth, imgHeight, url, mediaSettings, download, options, filename;
 43 		
 44 		// Refresh the settings in case something has changed.
 45 		mediaSettings 	= this.mediaSettings;
 46 		mediaSettings.getSettings(visibleCoords);
 47 
 48 		filename  = mediaSettings.filename;
 49 		this.building = true;
 50 
 51 		imgWidth  = mediaSettings.width; 
 52 		imgHeight = mediaSettings.height; 
 53 
 54 		self = this;
 55 		xhr = new Ajax.Request('api/index.php', {
 56 			method: 'POST',
 57 			parameters: {
 58 				action: 	"takeScreenshot",
 59 				layers: 	(mediaSettings.layers).join("/"),
 60 				obsDate: 	mediaSettings.startTime,
 61 				zoomLevel: 	mediaSettings.zoomLevel,
 62 				edges: 		mediaSettings.edgeEnhance,
 63 				sharpen: 	mediaSettings.sharpen,
 64 				imageSize:	imgWidth + "," + imgHeight,
 65 				filename:	filename
 66 			},
 67 			
 68 			onComplete: function (transport) {
 69 				self.building = false;
 70 				url = transport.responseJSON;
 71 
 72 				// If the response is an error message instead of a url, show the message
 73 				if (url === null) {
 74 					mediaSettings.shadowboxWarn(transport.responseText);
 75 				}
 76 				
 77 				else {		
 78 					// Options for the jGrowl notification. Clicking on the notification will let the user download the file.						
 79 					options = {
 80 						sticky: true,
 81 						open:	function (e, m) {
 82 							download = jQuery('#screenshot-' + filename);
 83 							
 84 							download.click(function () {
 85 								window.open('api/index.php?action=downloadFile&url=' + url, '_parent');
 86 							});
 87 						}
 88 					};
 89 
 90 					// Create the jGrowl notification.
 91 					helioviewer.messageConsole.info("<div id='screenshot-" + filename + "' style='cursor: pointer'>Click here to download '" + filename + "' </div>", options);
 92 
 93 					/*
 94 					7-28-2009 -- This section is disabled because shadowbox no longer pops up when the screenshot is done. A notification now handles linking to the download.
 95 					vpDimensions = helioviewer.viewport.getDimensions();
 96 	
 97 					vpWidth  = vpDimensions.width;
 98 					vpHeight = vpDimensions.height;
 99 	
100 					// The largest the image can be and not display with scrollbars is slightly smaller than the size of the viewport.
101 					// If the image is larger than that or equal in size to the viewport, scale it to 80%.			
102 					if (imgWidth >= vpWidth || imgHeight >= (vpHeight - 100)) {
103 						imgWidth  = imgWidth * 0.8;
104 						imgHeight = imgHeight * 0.8;
105 					}
106 	
107 					sbWidth  = Math.max(imgWidth + 40, 300);
108 					sbHeight = imgHeight + 45;      
109 
110 			        Shadowbox.open({
111 			            content: 	'<div id="image-container" class="ui-corner-all" style="background:black; margin: 10px; padding: 10px; width: ' + sbWidth + '; ">' +
112 									'<img src=\"' + url + '\" width=' + imgWidth + ' height=' + imgHeight + ' ><br />' +
113 									'<div id="helioviewer-url-box" style="background:black">' +
114 									'Here is the url for your screenshot:' +
115 	            					'<form style="margin-top: 5px;"><input type="text" id="helioviewer-url-input-box" style="width:98%;" value="' + url + '"></form>' +
116 	            					'</div></div>',
117 						player:		"html",
118 			            title:      "Helioviewer Screenshot Viewer",
119 						height:     sbHeight + 65,
120 						width:      sbWidth + 20
121 			        });
122 					*/	
123 				}
124 			}
125 		});
126 	}
127 });
