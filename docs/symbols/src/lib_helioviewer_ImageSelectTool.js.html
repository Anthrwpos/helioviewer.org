1 /**
  2  * @author Jaclyn Beck
  3  * @description A class that deals with using the imgAreaSelect plugin, which allows the user to click and drag 
  4  * 				to select a subregion of the image in the viewport.
  5  */
  6 
  7 var ImageSelectTool = Class.create(UIElement, {
  8 	initialize: function (controller) {
  9 		Object.extend(this);
 10 		this.button 	= jQuery("#select-region-button");
 11 		this.bodyNode 	= $("outsideBox").parentNode;
 12 		this.controller = controller;
 13 			
 14 		self = this;
 15 	
 16 		this.button.click(function() {
 17 			var width, height, imgContainer, transImg, helioviewer;
 18 			helioviewer = self.controller;		
 19 				
 20 			// Get viewport dimensions to make the transparent image with. 
 21 			width = helioviewer.viewport.dimensions.width;
 22 			height = helioviewer.viewport.dimensions.height;
 23 				
 24 			/* Create a temporary transparent image that spans the height and width of the viewport. Necessary because the viewport image
 25 			 * is done in tiles and imgAreaSelect cannot cross over tile boundaries. 
 26 			 * Add the transparent image to the viewport, on top of the other tiles. 
 27 			 */
 28 			transImg = helioviewer.viewport.domNode.appendChild(Builder.node('img', { 
 29 				id: 	'transparent-image', 
 30 				src: 	'images/transparent_512.png', 
 31 				alt: 	"", 
 32 				width: 	width, 
 33 				height: height,
 34 				style: 	"position: relative; cursor: crosshair; z-index: 5;"
 35 			}));
 36 
 37 			/* Make a temporary container for imgAreaSelect to put all of its divs into.
 38 			 * Note that this container must be OUTSIDE of the viewport because otherwise the plugin's boundaries 
 39 			 * do not span the whole image for some reason.
 40 			 */
 41 			imgContainer = self.bodyNode.appendChild(Builder.node('div', {
 42 				id:		'imgContainer'
 43 			}));
 44 					
 45 			// Add an empty div for the dialog box that will pop up later.
 46 			dialog = imgContainer.appendChild(Builder.node('div', { 
 47 				id: 	'select-image-dialog',
 48 				style:	'cursor: pointer'
 49 			}));
 50 			self.selectArea(self);
 51 		}); 
 52 	},
 53 
 54 	/**
 55 	 * Loads the imgAreaSelect plugin and uses it on the transparent image that covers the viewport.
 56 	 * @param self
 57 	 * The function imgAreaSelect returns two variables, "img", which is the original transparent image, and "selection", which
 58 	 * is an array describing the selected area. Available data for "selection" is x1, y1, x2, y2, width, and height. 
 59 	 */
 60 	selectArea: function (self) {
 61 		var helioviewer = self.controller, coords = $A(), visibleCoords = $A();
 62 		
 63 		// Use imgAreaSelect on the transparent region to get the top, left, bottom, and right coordinates of the selected region. 
 64 		jQuery("#transparent-image").imgAreaSelect({ 
 65 			handles: 	true,
 66 			parent:		"#imgContainer",
 67 			onSelectEnd: function (img, selection) {	
 68 				// Get the coordinates of the selected image, and adjust them to be heliocentric like the viewport coords. 
 69 				visibleCoords = helioviewer.viewport.getHCViewportPixelCoords();
 70 				
 71 				coords = {
 72 					top: 	selection.y1 + visibleCoords.top,
 73 					left: 	selection.x1 + visibleCoords.left,
 74 					bottom: selection.y2 + visibleCoords.bottom,
 75 					right: 	selection.x2 + visibleCoords.right
 76 				};
 77 				
 78 				//Pop up a dialog box asking the user what they want to do with it.				 
 79 				self.createDialog(coords, self);
 80 			}
 81 		});
 82 	},
 83 
 84 	/**
 85 	 * Loads a dialog pop-up that asks the user what they would like to do with the region they selected.
 86 	 * Current options are: "Take Screenshot", "Build Movie", and "Cancel".
 87 	 * @param coords -- an array of the heliocentric top, left, right, and bottom coordinates of the selected region, relative to the 
 88 	 * 					viewport coordinates.
 89 	 * @param self
 90 	 */	
 91 	createDialog: function (coords, self) {
 92 		var helioviewer = self.controller;
 93         if (jQuery(self).hasClass("dialog-loaded")) {
 94             var d = jQuery('#select-image-dialog');
 95             if (d.dialog('isOpen')) {
 96 				d.dialog('close');
 97 			}
 98 			else {
 99 				d.dialog('open');
100 			}
101         }  
102         else { 
103     		jQuery('#select-image-dialog').load("dialogs/select-region.php").dialog({
104 	    		autoOpen: true,
105     			title: 	"Helioviewer - Select Region",
106     			width: 	480,
107     			height: 150,
108     			draggable: 	true,
109 				resizable: 	false,
110 				buttons: 	{
111 					// Cancel button closes dialog, disables imgAreaSelect, and deletes the transparent image and dialog div.
112 					"Cancel": function() { 
113 						self.cleanup();
114 					}, 
115 					// Build Movie and Take Screenshot send the information on to their respective functions.
116 					"Build Movie": function() { 
117 						self.cleanup();
118 						helioviewer.movieBuilder.buildMovie(coords, helioviewer.movieBuilder);
119 					},
120 					"Take Screenshot": function() {
121 						self.cleanup();
122 						helioviewer.takeScreenshot(coords, helioviewer);
123 					}
124 				}
125     		});
126             jQuery(self).addClass("dialog-loaded");
127         } 		
128 	},
129 
130 	/**
131 	 * Removes all divs created by imgAreaSelect and the dialog loading
132 	 * @param imgContainer -- has all imgAreaSelect divs inside
133 	 * @param transImg -- temporary transparent image that imgAreaSelect is used on.
134 	 * @TODO: add error checking if the divs are already gone for whatever reason.
135 	 */	
136 	cleanup: function() {
137 		var dialogDiv = jQuery("#select-image-dialog");
138 		dialogDiv.dialog("close"); 
139 		dialogDiv.parent().remove();
140 		dialogDiv.remove();
141 		jQuery("#imgContainer").remove();
142 		jQuery("#transparent-image").remove();
143 	}
144 });
