1 /**
  2  * @fileOverview Contains the class definition for an ZoomControls class.
  3  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @author <a href="mailto:patrick.schmiedel@gmx.net">Patrick Schmiedel</a>
  5  * @see  The <a href="http://helioviewer.org/mediawiki-1.11.1/index.php?title=Zoom_Levels_and_Observations">HelioViewer Wiki</a>
  6  *       for more information on zoom levels.
  7  */
  8 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
  9 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 10 /*global Class, $ */
 11 "use strict";
 12 var ZoomControls = Class.extend(
 13     /** @lends ZoomControls.prototype */
 14     {
 15     /**
 16      * @constructs
 17      * @description Creates a new ZoomControl
 18      * @param {Object} controller A Reference to the Helioviewer application class
 19      * @param {Object} options Custom ZoomControl settings
 20      */
 21     init: function (controller, options) {
 22         $.extend(this, options);
 23         this.controller = controller;
 24         this.domNode = $(this.id);
 25         this.offset  = this.minZoomLevel + this.maxZoomLevel;
 26        
 27         var self = this;
 28         
 29         this._buildUI();
 30        
 31         // Initialize slider
 32         this.zoomSlider.slider({
 33             slide: function (event, slider) {
 34                 self._onSlide(slider.value);
 35             },
 36             min: this.minZoomLevel,
 37             max: this.maxZoomLevel,
 38             orientation: 'vertical',
 39             value: this.offset - this.zoomLevel
 40         });
 41 
 42         this._setupTooltips();
 43         this._initEvents();
 44     },
 45 
 46     /**
 47      * @description Increases or decreases zoom level in response to pressing the plus/minus buttons.
 48      * @param {Integer} dir The amount to adjust the zoom level by (+1 or -1).              
 49      */
 50     zoomButtonClicked: function (dir) {
 51         var v = this.zoomSlider.slider("value") + dir;
 52         this.zoomSlider.slider("value", v);
 53         this._setZoomLevel(v);
 54     },
 55     
 56     /**
 57      * @description Sets up tooltips for zoom controls
 58      */
 59     _setupTooltips: function () {
 60         var description, targets;
 61         
 62         description = "Drag this handle up and down to zoom in and out of the displayed image.";
 63         $("#zoomControlSlider > .ui-slider-handle").attr("title", description);
 64         
 65         targets = "#zoomControlZoomOut, #zoomControlZoomIn, #zoomControlHandle, #zoomControlSlider > .ui-slider-handle";
 66         this.controller.tooltips.createTooltip($(targets));
 67     },
 68   
 69     /**
 70      * @description Adjusts the zoom-control slider
 71      * @param {Integer} v The new zoom value.
 72      */
 73     _onSlide: function (v) {
 74         this._setZoomLevel(v);
 75     },
 76     
 77     /**
 78      * @description Translates from jQuery slider values to zoom-levels, and updates the zoom-level.
 79      * @param {Object} v jQuery slider value
 80      */
 81     _setZoomLevel: function (v) {
 82         this.controller.viewport.zoomTo(this.offset - v);      
 83     },
 84     
 85     /**
 86      * @description sets up zoom control UI element
 87      */
 88     _buildUI: function () {
 89         this.zoomInBtn  = $('<div id="zoomControlZoomIn" title="Zoom in.">+</div>');
 90         this.zoomSlider = $('<div id="zoomControlSlider"></div>');
 91         this.zoomOutBtn = $('<div id="zoomControlZoomOut" title="Zoom out.">-</div>');
 92 
 93         var sliderContainer = $('<div id="zoomSliderContainer"></div>').append(this.zoomSlider);
 94 
 95         this.domNode.append(this.zoomInBtn).append(sliderContainer).append(this.zoomOutBtn);
 96     },
 97     
 98     /**
 99      * @description Initializes zoom control-related event-handlers
100      */
101     _initEvents: function () {
102         // Zoom-in button
103         this.zoomInBtn.bind("click", {zoomControl: this}, function (e) {
104             var val = e.data.zoomControl.zoomSlider.slider("value");
105             if (val < e.data.zoomControl.maxZoomLevel) {
106                 e.data.zoomControl.zoomButtonClicked(1);
107             }
108         });
109         
110         // Zoom-out button
111         this.zoomOutBtn.bind("click", {zoomControl: this}, function (e) {
112             var val = e.data.zoomControl.zoomSlider.slider("value");
113             if (val > e.data.zoomControl.minZoomLevel) {
114                 e.data.zoomControl.zoomButtonClicked(-1);
115             }
116         });
117     }
118 });
