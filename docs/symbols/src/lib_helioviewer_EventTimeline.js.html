1 /**
  2  * @author <a href="mailto:keith.hughitt@gmail.com">Keith Hughitt</a> 
  3  * @fileOverview This class extends the base Simile Timeline
  4  * @see http://code.google.com/p/simile-widgets/
  5  * 
  6  * Syntax: Prototype
  7  */
  8 /*global Class, $, EventTimeline, Timeline, UIElement, window */
  9 var EventTimeline = Class.create(UIElement,
 10 	/** @lends EventTimeline.prototype */
 11 	{
 12 	/**
 13 	 * @description Creates a new EventTimeline. 
 14 	 * @param {Object} controller Reference to the controller class (Helioviewer).
 15 	 * @param {String} container The ID for the timeline's container.
 16 	 * @constructs 
 17 	 */ 
 18     initialize: function (controller, container) {
 19 		this.controller = controller;
 20 		this.container  = container;
 21 		this.resizeTimerID = null;
 22 		
 23 		this.eventSource = new Timeline.DefaultEventSource();
 24 		var bandInfos = [
 25 			Timeline.createBandInfo({
 26 				eventSource:    this.eventSource,
 27 				width:          "70%", 
 28 				intervalUnit:   Timeline.DateTime.MONTH, 
 29 				intervalPixels: 100
 30 			}),
 31 			Timeline.createBandInfo({
 32 				eventSource:    this.eventSource,
 33 				width:          "30%", 
 34 				intervalUnit:   Timeline.DateTime.YEAR, 
 35 				intervalPixels: 200
 36 			})
 37 		],
 38 		self = this;
 39 		
 40 		bandInfos[1].syncWith = 0;
 41 		bandInfos[1].highlight = true;
 42 		this.timeline = Timeline.create($(this.container), bandInfos);
 43 		this.timeline.loadJSON("http://localhost/dev/test.json", function (json, url) {
 44 			self.eventSource.loadJSON(json, url);
 45 		});
 46 	},
 47 
 48 	/**
 49 	 * @description Event-hanlder for window resize
 50 	 */
 51 	resize: function () {
 52 		if (this.resizeTimerID === null) {
 53 			this.resizeTimerID = window.setTimeout(function () {
 54 				this.resizeTimerID = null;
 55 				this.timeline.layout();
 56 			}, 500);
 57 		}
 58 	}	
 59 });
