164     	this.userSettings = new UserSettings(this);
165

167 		if (this.load["date"]) {
168             var timestamp = getUTCTimestamp(this.load["date"]);
169             this.userSettings.set('obs-date', timestamp);
170 		}

173 			this.userSettings.set('zoom-level', this.scaleToZoomLevel(parseInt(this.load["img-scale"], 10)));
174 		}
175

177 		if (this.load.layers) {
178 			var serverNum, api, layer, self = this, layers =  [];
179             
180 			$A(this.load.layers).each(function (layerString) {
181                 // Select tiling server if distributed tiling is enabling
182                 if ((self.distributed === true) && ((layers.length % 2) == 0)) {
183                     serverNum = 2;
184                     api = self.tileServer2;
185                 }
186                 else {
187                     serverNum = 1;
188                     api = self.tileServer1;
189                 }
190                 
191                 // Parse layer string
192                 layer = self.parseLayerString(layerString);
193                 
194                 // Load layer
195 				layers.push({
196                     tileAPI: api,
197                     server: serverNum,
198                     observatory: layer.observatory,
199                     instrument : layer.instrument,
200                     detector   : layer.detector,
201                     measurement: layer.measurement
202                });
203 			});
204 			this.userSettings.set('tile-layers', layers);
205 		}

351 	 * @description Initialize keyboard-related event handlers.
352 	 * @see Based off method by <a href="http://www.quirksmode.org/js/events_properties.html#key">PPK</a>
353 	 */
354 	_initKeyBoardListeners: function () {
355 		var self = this;
356 		Event.observe(document, 'keypress', function (e) {
357 
358 			//Ignore event if user is type in an input form field
359 			if (e.target.tagName !== "INPUT") {
360 				var code, character;
361 				if (!e) {
362 					e = window.event;
363 				}
364 				if (e.keyCode) {
365 					code = e.keyCode;
366 				}
367 				else if (e.which) {
368 					code = e.which;
369 				}
370 
371 				character = String.fromCharCode(code);
372 
373 				//TODO: use events or public method instead of zoomControl's (private) method.
374 				if (character === "-" || character === "_") {
375 					self.zoomControl.zoomButtonClicked(+1);
376 				}
377 				else if (character === "=" || character === "+") {
378 					self.zoomControl.zoomButtonClicked(-1);
379 				}
380 				else if (character === "c") {
381                     self.viewport.center();
382 				}
383 				// event label visibility toggle
384 				else if (character === "d") {
385                     self.layerManager.toggleLabels();
386 				}
387                 
388 				// toggle mouse-coords display
389 				else if (character === "m") {
390     				self.viewport.viewportHandlers.toggleMouseCoords();
391 				}
392                 
393                 // toggle full-screen display
394                 else if (character === "f") {
395                     jQuery("#fullscreen-btn").click();
396                 }
397                 
398                 // step back in time
399                 else if (character === ",") {
400                     self.timeControls.timePrevious();
401                 }
402                 
403                 // step forward in time
404                 else if (character === ".") {
405                     self.timeControls.timeNext();
406                 }                          
407 			}
408 		});
409 	},

469      * @param {Object} dateStr
470      */
471     updateDate: function (dateStr) {
472         var date, time, hours, minutes, seconds, utcDate;
473         
474         date = Date.parse(dateStr);
475         time = this.calendar.timeField.value;
476     	
477 		//Factor in time portion of timestamp
478 		hours =   parseInt(time.substring(0, 2), 10);
479     	minutes = parseInt(time.substring(3, 5), 10);
480     	seconds = parseInt(time.substring(6, 8), 10);
481     
482     	//Convert to UTC
483     	utcDate = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate(), hours, minutes, seconds));    
484         
485     	this.setDate(utcDate);
486 		this.calendar.updateFields();
487     },

503 	 * @param {Float} imgScale The image scale in arcseconds per pixel

536     /**
537      * @description Builds a URL for the current view
538      * @param {Object} level
539      * @TODO: Add support for viewport offset, event layers, opacity
540      */
541     generateURL: function () {
542         var url, date, zoom, layers, self = this;
543         
544         // Base URL
545         url = "http://www.helioviewer.org/?";
546         
547         // Add timestamp
548   		// Work-around: In Firefox 3.1+, Date.toISOString() Returns single-quoted strings
549 		// http://code.google.com/p/datejs/issues/detail?id=54
550 		if (navigator.userAgent.search(/3\.[1-9]/) !== -1) {
551 			date = this.date.toISOString();
552 		}
553 		else {
554 			date = this.date.toISOString().slice(1, -1);
555 		}
556 
557         // Add zoomlevel
558         zoom = this.zoomLevelToScale(this.viewport.zoomLevel);
559         
560         // Add layers
561         layers="";
562         $A(this.layerManager.tileLayers()).each(function(l) {
563             layers += self.layerManager.generateLayerString(l) + ",";
564         });
565         
566         // Remove trailing comma in layers list
567         layers = layers.slice(0,-1);
568         
569         // Build URL
570         url += "date=" + date + "&img-scale=" + zoom + "&layers=" + layers;
571 
572         return url;
573     },

629 	 * @namespace
630 	 * @description Helioviewer application-level event handlers

632 	handlers: {
633 		/**
634 		 * @description Changes the zoom-level to a new value
635 		 * @param {Int} level The new zoom-level to use
636 		 */
637 		zoom: function (level) {
638 			this.viewport.zoomTo(level);

640 
641 		observationDateChange: function (e) {
642             this.updateDate(e.target.value);
643 		},
644 
645 		observationTimeChange: function (e) {
646 			var time = e.target.value,
647 				regex, newTime, hours, mins, secs;
648 
649 			//make sure time entered in correct format
650 			regex = /^\d{2}:\d{2}:\d{2}?/;
651 
652 			//Check to see if the input is a valid time
653 			if (time.match(regex)) {
654 				//Get the difference in times and add to this.date
655 				newTime = time.split(':');
656 				hours = parseInt(newTime[0], 10) - this.date.getUTCHours();
657 				mins  = parseInt(newTime[1], 10) - this.date.getUTCMinutes();
658 				secs  = parseInt(newTime[2], 10) - this.date.getUTCSeconds();
659 
660 				this.date.addHours(hours);
661 				this.date.addMinutes(mins);
662 				this.date.setSeconds(secs);
663 
664 				this.setDate(this.date);
665 
666 			} else {
667 				this.messageConsole.warn('Invalid time. Please enter a time in of form HH:MM:SS');
668 			}
