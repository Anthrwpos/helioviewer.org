207 	/**
208 	 * @description Initialize keyboard-related event handlers.
209 	 * @see Based off method by <a href="http://www.quirksmode.org/js/events_properties.html#key">PPK</a>
210 	 */
211 	initKeyBoardListeners: function () {
212 		var self = this;
213 		Event.observe(document, 'keypress', function (e) {
214 
215 			//Ignore event if user is type in an input form field
216 			if (e.target.tagName !== "INPUT") {
217 				var code, character;
218 				if (!e) {
219 					e = window.event;
220 				}
221 				if (e.keyCode) {
222 					code = e.keyCode;
223 				}
224 				else if (e.which) {
225 					code = e.which;
226 				}
227 				
228 				character = String.fromCharCode(code);
229 
230 				//TODO: use events or public method instead of zoomControl's (private) method.
231 				if (character === "-" || character === "_") {
232 					self.zoomControl.zoomButtonClicked(+1);
233 				}
234 				else if (character === "=" || character === "+") {
235 					self.zoomControl.zoomButtonClicked(-1);
236 				}
237 				else if (character === "c") {
238 					self.viewports.each(function (viewport) {
239 						viewport.center();
240 					});
241 				}
242 				//event label visibility toggle
243 				else if (character === "d") {
244 					self.layerManager.layers.each(function (layer) {
245 						if (layer.type === "EventLayer") {
246 							layer.toggleLabelVisibility();
247 						}
248 					});
249 				}
250 				// toggle mouse-coords display
251 				else if (character === "m") {
252 					self.viewports.each(function (viewport) {
253 						viewport.ViewportHandlers.toggleMouseCoords();
254 					});
255 				}
256 				
257 			}
258 		});
259 	},

336 	/**
337 	 * @description Finds the closest support zoom-level to a given pixel scale (arcseconds per pixel)
338 	 * @param {Float} imgScale The image scale in arcseconds per pixel
339 	 */
340 	scaleToZoomLevel: function (imgScale) {
341 		var zoomOffset = Math.round(Math.lg((imgScale / this.baseScale)));
342 		return this.baseZoom + zoomOffset;
343 	},

365 		observationTimeChange: function (e) {
366 			var time = e.target.value,
367 				regex, newTime, hours, mins, secs;
368 
369 			//make sure time entered in correct format
370 			regex = /^\d{2}:\d{2}:\d{2}?/;
371 
372 			//Check to see if the input is a valid time
373 			if (time.match(regex)) {
374 				//Get the difference in times and add to this.date
375 				newTime = time.split(':');
376 				hours = parseInt(newTime[0], 10) - this.date.getUTCHours();
377 				mins  = parseInt(newTime[1], 10) - this.date.getUTCMinutes();
378 				secs  = parseInt(newTime[2], 10) - this.date.getUTCSeconds();
379 
380 				this.date.addHours(hours);
381 				this.date.addMinutes(mins);
382 				this.date.setSeconds(secs);
383 
384 				this.setDate(this.date);
385 				
386 			} else {
387 				this.messageConsole.warn('Invalid time. Please enter a time in of form HH:MM:SS');
388 			}
389 		},
