224     /**
225      * @description Loads user settings from URL, cookies, or defaults if no settings have been stored.
226      */
227     loadUserSettings: function () {
228         var timestamp, layerSettings, layers, rand, self = this;
229         
230         // Optional debugging information
231         // TODO 01/20/2010: Provide finer control over what should be logged, e.g. "debug=[tiles,keyboard]"
232         if (this.load.debug && (this.load.debug.toLowerCase() === "true")) {
233             this.debug = true;
234         }
235         
236         this.userSettings = new UserSettings(this);
237         
238         // Load any view parameters specified via API
239         if (this.load.date) {
240             timestamp = getUTCTimestamp(this.load.date);
241             this.userSettings.set('date', timestamp);
242         }
243 
244         if (this.load.imageScale) {
245             this.userSettings.set('zoomLevel', this.scaleToZoomLevel(this.load.imageScale));
246         }
247 
248         // Process and load and layer strings specified
249         if (this.load.imageLayers) {
250             layers = [];
251             
252             $.each(this.load.imageLayers, function () {
253                 layerSettings = self.userSettings.parseLayerString(this);
254                 
255                 // Choose server to use
256                 if ((self.distributed === true) && ((layers.length % 2) === 0)) {
257                     rand = Math.floor(Math.random() * (self.load.imageLayers.length - 1)) + 1;
258                     layerSettings.server = rand;
259                 }
260                 else {
261                     layerSettings.server = 0;
262                 }
263                 
264                 // Load layer
265                 layers.push(layerSettings);
266             });
267             this.userSettings.set('tileLayers', layers);
268         }

272     /**
273      * @description Creates an HTML button for toggling between regular and fullscreen display
274      */
275     _createFullscreenBtn: function () {
276         var btn, footer, header, vp, sb, speed, marginSize, meta, panels, outsideBox, origOutsideMarginLeft, 
277             origOutsideMarginRight, origHeaderHeight, origViewportHeight, $_fx_step_default, self, body;
278         
279         // get dom-node
280         btn = $("#fullscreen-btn");
281         
282         // CSS Selectors
283         outsideBox = $('#outsideBox');
284         body       = $('body');
285         vp         = $('#helioviewer-viewport-container-outer');
286         sb         = $('#sandbox');
287         footer     = $('#footer-links-container-outer');
288         meta       = $('#footer-container-outer');
289         header     = $('#middle-col-header');
290         panels     = $("#left-col, #right-col, #footer-links-container-outer, #social-buttons-container-outer");
291        
292         // animation speed
293         speed = 500;
294         
295         // margin-size
296         marginSize = 5;
297 
298         // Overide jQuery's animation method
299         // http://acko.net/blog/abusing-jquery-animate-for-fun-and-profit-and-bacon
300         self  = this;
301         $_fx_step_default = $.fx.step._default;
302         $.fx.step._default = function (fx) {
303             if (fx.elem.id !== "sandbox") {
304                 return $_fx_step_default(fx);
305             }
306             self.viewport.updateSandbox();
307             fx.elem.updated = true;
308         };
309         
310         // setup event-handler
311         btn.click(function () {
312             if (!btn.hasClass('requests-disabled')) {
313                             
314                 // toggle fullscreen class
315                 outsideBox.toggleClass('fullscreen-mode');
316                 
317                 // make sure action finishes before starting a new one
318                 btn.addClass('requests-disabled');
319                 
320                 // fullscreen mode
321                 if (outsideBox.hasClass('fullscreen-mode')) {
322                     
323                     // hide overflow
324                     body.css('overflow', 'hidden');
325                     
326                     meta.hide();
327     
328                     // keep track of original dimensions
329                     origOutsideMarginLeft  = outsideBox.css("margin-left");
330                     origOutsideMarginRight = outsideBox.css("margin-right");
331                     origHeaderHeight       = header.height();
332                     origViewportHeight     = vp.height();
333                     
334                     outsideBox.animate({ 
335                         marginLeft:  marginSize,
336                         marginRight: marginSize
337                     }, speed,
338                     function () {
339                         self.viewport.checkTiles();
340                         self.tileLayers.resetLayers();
341                         self.eventLayers.resetLayers();
342                         panels.hide();
343                         btn.removeClass('requests-disabled');
344                     });
345                        
346                     header.animate({
347                         height: marginSize
348                     }, speed);
349     
350                     vp.animate({
351                         height: $(window).height() - (marginSize * 3)
352                     }, speed);
353      
354                     sb.animate({
355                         right: 1 // Trash
356                     }, speed);                
357                
358                 // regular mode      
359                 } else {
360                     panels.show();
361                         
362                     outsideBox.animate({ 
363                         marginLeft:  origOutsideMarginLeft,
364                         marginRight: origOutsideMarginRight
365                     }, speed,
366                     function () {
367                         btn.removeClass('requests-disabled');    
368                         meta.show();
369                         
370                         // show overflow
371                         body.css('overflow', 'visible');
372                     });
373     
374                     vp.animate({
375                         height: origViewportHeight
376                     }, speed);
377                     header.animate({
378                         height: origHeaderHeight
379                     }, speed);
380                     sb.animate({
381                         right: 0
382                     }, speed);
383                 }
384             }
385         });
386     },

472     /**
473      * @description Finds the closest support zoom-level to a given pixel scale (arcseconds per pixel)
474      * @param {Float} imgScale The image scale in arcseconds per pixel
475      */
476     scaleToZoomLevel: function (imgScale) {
477         var zoomOffset = Math.round(Math.lg((imgScale / this.baseScale)));
478         return this.baseZoom + zoomOffset;
479     },
480     
481     /**
482      * @description Translates a given zoom-level into an image plate scale.
483      */
484     getImageScale: function () {
485         var zoomOffset = this.viewport.zoomLevel - this.baseZoom;
486         return this.baseScale * (Math.pow(2, zoomOffset));
487     },
