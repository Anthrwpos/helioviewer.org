471 	/**
472 	 * @description Initialize keyboard-related event handlers.
473 	 * @see Based off method by <a href="http://www.quirksmode.org/js/events_properties.html#key">PPK</a>
474 	 */
475 	_initKeyBoardListeners: function () {
476 		var self = this;
477 		Event.observe(document, 'keypress', function (e) {
478 
479 			//Ignore event if user is type in an input form field
480 			if (e.target.tagName !== "INPUT") {
481 				var code, character;
482 				if (!e) {
483 					e = window.event;
484 				}
485 				if (e.keyCode) {
486 					code = e.keyCode;
487 				}
488 				else if (e.which) {
489 					code = e.which;
490 				}
491 
492 				character = String.fromCharCode(code);
493 
494 				//TODO: use events or public method instead of zoomControl's (private) method.
495 				if (character === "-" || character === "_") {
496 					self.zoomControl.zoomButtonClicked(+1);
497 				}
498 				else if (character === "=" || character === "+") {
499 					self.zoomControl.zoomButtonClicked(-1);
500 				}
501 				else if (character === "c") {
502                     self.viewport.center();
503 				}
504 				// event label visibility toggle
505 				else if (character === "d") {
506                     self.eventLayers.toggleLabels();
507 				}
508                 
509 				// toggle mouse-coords display
510 				else if (character === "m") {
511     				self.viewport.viewportHandlers.toggleMouseCoords();
512 				}
513                 
514                 // toggle full-screen display
515                 else if (character === "f") {
516                     jQuery("#fullscreen-btn").click();
517                 }
518                 
519                 // step back in time
520                 else if (character === ",") {
521                     self.timeControls.timePrevious();
522                 }
523                 
524                 // step forward in time
525                 else if (character === ".") {
526                     self.timeControls.timeNext();
527                 }                          
528 			}
529 		});

589      * @description Updates the observation date
590      * @param {Object} dateStr
591      */
592     updateDate: function (dateStr) {
593         var date, time, hours, minutes, seconds, utcDate;
594         
595         date = Date.parse(dateStr);
596         time = this.calendar.timeField.value;
597     	
598 		//Factor in time portion of timestamp
599 		hours =   parseInt(time.substring(0, 2), 10);
600     	minutes = parseInt(time.substring(3, 5), 10);
601     	seconds = parseInt(time.substring(6, 8), 10);
602     
603     	//Convert to UTC
604     	utcDate = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate(), hours, minutes, seconds));    
605         
606     	this.setDate(utcDate);
607 		this.calendar.updateFields();
608     },

624 	 * @param {Float} imgScale The image scale in arcseconds per pixel

626 	scaleToZoomLevel: function (imgScale) {
627 		var zoomOffset = Math.round(Math.lg((imgScale / this.baseScale)));

740 	/*
741 	 * Figures out where the heliocenter is in terms of viewport coordinates. The viewport can be divided into 
742 	 * 9 regions, 3 along the x-axis (West, Center, Right), and 3 along the y-axis (North, Center, South).
743 	 * Returns the region that the heliocenter falls into (or the nearest region, if the heliocenter is outside the viewport). 
744 	 */	
745 	getHCOffset: function(visibleCoords) {
746 		var left, right, top, bottom, offset;
747 
748 		left 	= visibleCoords.left;
749 		right 	= visibleCoords.right;
750 		top 	= visibleCoords.top;
751 		bottom 	= visibleCoords.bottom;
752 		
753 		// How far away from the top left corner, and in which direction.
754 		// Note that the convert command reverses + and - for the x-direction, 
755 		// so + means push the image to the left, - means push it to the right.
756 		// The center coordinates do not need to be modified because of this,
757 		// since a +x means the heliocenter (0,0) is on the left, and a +y means
758 		// the heliocenter is above.
759 		offset = {
760 			x: left,
761 			y: top
762 		}
763 			
764 		return offset;
765 	},
766 	
767 	getViewportInformation: function(visibleCoords) {
768 		var layers, hcOffset;
769 		
770 		// Get the top, left, bottom, and right coordinates of what is currently visible in the viewport			
771 		viewportCoords = this.viewport.getHCViewportPixelCoords(); 
772 		centerX = viewportCoords.left + (viewportCoords.right  - viewportCoords.left) / 2;
773 		centerY = viewportCoords.top  + (viewportCoords.bottom - viewportCoords.top) / 2;
774 
775 		// Adjust so that visibleCoords points to the 512x512 region in the center of the viewport (until aspect ratios are put in)		
776 		visibleCoords.left 		= centerX - 256;
777 		visibleCoords.top 		= centerY - 256;
778 		visibleCoords.right 	= centerX + 256;
779 		visibleCoords.bottom 	= centerY + 256;
780 		
781 		hcOffset = this.getHCOffset(visibleCoords);
782 
783 		// Arrays to hold layer data to send to API.php
784 		var layerNames = $A(), layerOpacities = $A(), layerXRanges = $A(), layerYRanges = $A();
785 		
786 		// Only add visible tile layers to the array.
787 		this.tileLayers.each(function(t){
788 			if (t.visible) {
789 				var left, top, right, bottom, width, height, sizeOffset;
790 				
791                 // TODO: find a way to not hard-code these that doesn't take much time.
792                  var jp2Width = t.width;
793                  if(t.detector == "0C2")
794                          jp2Width = 1174;
795                  else if(t.detector == "0C3")
796                          jp2Width = 1418;
797 						 
798 				sizeOffset = jp2Width / t.relWidth;
799 
800 				// Convert viewport heliocentric coordinates into image coordinates
801 				left 	= Math.floor(visibleCoords.left + t.relWidth / 2);
802 				top 	= Math.floor(visibleCoords.top  + t.relHeight / 2);
803 				right 	= Math.floor(visibleCoords.right  + t.relWidth / 2);
804 				bottom 	= Math.floor(visibleCoords.bottom + t.relHeight / 2);
805 				
806 				/* 
807 				 * Need to adjust for when 1 px on the jp2 image is not the same as 1 px on the viewport image.
808 				 * Example: "left" is the pixel coordinate relative to the image in the viewport, which may be twice as large
809 				 * as its corresponding jp2 image. The sizeOffset for this image would be (image width)/(2x image width) or 0.5,
810 				 * so to get the pixel coordinate relative to the jp2 image, multiply "left" by 0.5, the sizeOffset.
811 				 * zoomOffset cannot be used here because images like LASCO C2 and C3 have different sizeOffsets than EIT images
812 				 * at the same zoomLevel.
813 				 * If "left" or "top" is less than 0, just use 0.
814 				 */
815 				left 	= Math.max(left * sizeOffset, 0);
816 				top 	= Math.max(top * sizeOffset, 0);
817 				right 	= right * sizeOffset;
818 				bottom 	= bottom * sizeOffset;
819 
820 				// If the calculated width is greater than the possible width, just use the possible width. 							
821 				width  = Math.min((right - left), t.width - left);
822 				height = Math.min((bottom - top), t.height - top);
823 
824 				// Make it a square region (until aspect ratios are figured out).				
825 				imagesize = Math.max(width, height);
826 				width 	  = imagesize;
827 				height 	  = imagesize;
828 
829 				// If the captured image just shows the black circular region of LASCO C2 or C3, don't even use that layer.		
830 				if(t.detector == "0C2" && left >=437 && right <= 737 && top >=437 && bottom < 737) 
831 					width = -1;
832 				
833 				if(t.detector == "0C3" && left >= 637 && right <= 781 && top >= 637 && bottom <= 781)
834 					width = -1;
835 							
836 				// If at least some of the image is currently visible, add that layer
837 				if (width > 0 && height > 0 && top < t.height && left < t.width) {
838 					var xRange = {
839 						start: left,
840 						size: width
841 					};
842 					
843 					var yRange = {
844 						start: top,
845 						size: height
846 					};
847 					
848 					// Add layer name, opacity, and ranges to the appropriate arrays.
849 					// Layer names are in the format OBS_INST_DET_MEAS to allow for variable-length names 
850 					var name = $A([t.observatory, t.instrument, t.detector, t.measurement])
851 
852 					layerNames.push(name.join("_"));
853 					layerOpacities.push(t.opacity);
854 					layerXRanges.push(xRange.start + "," + xRange.size);
855 					layerYRanges.push(yRange.start + "," + yRange.size);
856 				}
857 			}
858 		});
859 		
860 		var layers = {
861 			layerNames: 	layerNames,
862 			layerOpacities: layerOpacities,
863 			layerXRanges: 	layerXRanges,
864 			layerYRanges: 	layerYRanges,
865 			hcOffset: 		hcOffset
866 		};
867 	
868 		return layers;
869 	}
