  6 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
  7 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
  8 /*global Class */
  9 "use strict";
 10 var Layer = Class.extend(
 11     /** @lends Layer.prototype */
 12     {
 13     maxZoomLevel: 20, // ZoomLevel where FullSize = 1px
 14     minZoomLevel: 10,
 15     visible: true,
 16 
 17     /**
 18      * @constructs
 19      * @description Creates a new Layer
 20      * @param {Object} viewport Viewport to place the layer in
 21      * <br>
 22      * <br><div style='font-size:16px'>Options:</div><br>
 23      * <div style='margin-left:15px'>
 24      *         <b>maxZoomLevel</b>    -  Maximum zoom level supported by the layer<br>
 25      *      <b>minZoomLevel</b>    - Minimum zoom level supported by the layer<br>
 26      *        <b>visible</b> - The default layer visibility<br>
 27      * </div>
 28      */
 29     init: function (controller) {
 30         this.controller = controller;
 31         this.viewport   = controller.viewport;
 32         this.id = 'layer' + Math.floor(Math.random() * 100000 + 1);
 33     },
 34 
 35     /**
 36      * @description Adjust the Layer's z-index 
 37      * @param {Object} val Z-index to use
 38      */
 39     setZIndex: function (val) {
 40         this.domNode.css("z-index", val);
 41     },
 42 
 43     /**
 44      * @description Sets the Layer's visibility
 45      * @param {Boolean} visible Hide/Show layer 
 46      * @returns {Boolean} Returns new setting
 47      */
 48     setVisibility: function (visible) {
 49         this.visible = visible;
 50         if (visible) {
 51             this.domNode.show();
 52         }
 53         else {
 54             this.domNode.hide();
 55         }
 56         
 57         return this.visible;
 58     },
 59 
 60     /**
 61      * @description Toggle layer visibility
 62      */
 63     toggleVisibility: function () {
 64         return this.setVisibility(!this.visible);
 65     }
 66 });
