1 /**
  2  * @fileOverview "Abstract" Layer class.
  3  * @author <a href="mailto:vincent.k.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @author <a href="mailto:patrick.schmiedel@gmx.net">Patrick Schmiedel</a>
  5  */
  6 /*global Layer, Class, UIElement, Element, $ */
  7 var Layer = Class.create(UIElement, 
  8 	/** @lends Layer.prototype */
  9 	{
 10 	maxZoomLevel: 20, // ZoomLevel where FullSize = 1px
 11 	minZoomLevel: 10,
 12 	visible: true,
 13 
 14 	/**
 15 	 * @constructs
 16 	 * @description Creates a new Layer
 17 	 * @param {Object} viewport Viewport to place the layer in
 18 	 * <br>
 19 	 * <br><div style='font-size:16px'>Options:</div><br>
 20 	 * <div style='margin-left:15px'>
 21 	 * 		<b>maxZoomLevel</b>	-  Maximum zoom level supported by the layer<br>
 22 	 *      <b>minZoomLevel</b>	- Minimum zoom level supported by the layer<br>
 23 	 *		<b>visible</b> - The default layer visibility<br>
 24 	 * </div>
 25 	 */
 26 	initialize: function (viewport) {
 27 		this.viewport = viewport;
 28 		this.domNode = $(viewport.movingContainer.appendChild(new Element('div')));
 29 		this.viewport.addObserver('move', this.viewportMove.bind(this));
 30 		this.id = 'layer' + Math.floor(Math.random() * 100000 + 1);
 31 	},
 32 
 33 	/**
 34 	 * @description Adjust the Layer's z-index 
 35 	 * @param {Object} val Z-index to use
 36 	 */
 37 	setZIndex: function (val) {
 38 		this.domNode.setStyle({ zIndex: val });
 39 	},
 40 
 41 	/**
 42 	 * @description Sets the Layer's visibility
 43 	 * @param {Boolean} visible Hide/Show layer 
 44 	 * @returns {Boolean} Returns new setting
 45 	 */
 46 	setVisible: function (visible) {
 47 		this.visible = visible;
 48 		this.domNode.setStyle({ visibility: (visible ? 'visible' : 'hidden') });
 49 		return this.visible;
 50 	},
 51 
 52 	/**
 53 	 * @description Toggle layer visibility
 54 	 */
 55 	toggleVisible: function () {
 56 		return this.setVisible(!this.visible);
 57 	}
 58 });
