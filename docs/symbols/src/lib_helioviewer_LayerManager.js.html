1 /**
  2  * @fileOverview Contains class definition for a simple layer manager
  3  * @author <a href="mailto:keith.hughitt@gmail.com">Keith Hughitt</a>
  4  */
  5 /*global LayerManager, Class, UIElement, $A */
  6 var LayerManager = Class.create(UIElement,
  7 	/** @lends LayerManager.prototype */
  8 	{
  9     /**
 10      * @constructs
 11      * @description Creates a new LayerManager
 12      * @param {Object} A Rseference to the main application controller
 13      */
 14     initialize: function (controller) {
 15         this.controller = controller;
 16         this.layers = $A([]);
 17     },
 18     
 19     //hasId: function (id) {
 20 		//return (this.layers.grep(id).length > 0 ? true : false);
 21 	//},
 22 
 23 	/**
 24 	 * @description Add a new layer
 25 	 */
 26 	addLayer: function (layer) {
 27 		this.layers.push(layer);
 28 	},
 29 	
 30 	/**
 31 	 * @description Gets the number of TileLayers currently loaded
 32 	 * @return {Integer} Number of tile layers present.
 33 	 */
 34 	numTileLayers: function () {
 35 		var n = 0;
 36 		this.layers.each(function (l) {
 37 			if (l.type === "TileLayer") {
 38 				n += 1;
 39 			}
 40 		});
 41 		
 42 		return n;
 43 	},
 44 	
 45 	/**
 46 	 * @description Returns the largest width and height of any layers (does not have to be from same layer)
 47 	 * @returns {Object} The width and height of the largest layer
 48 	 */
 49 	getMaxDimensions: function () {
 50 		var maxWidth  = 0,
 51 			maxHeight = 0;
 52 		
 53 		this.layers.each(function (l) {
 54 			if (l.type === "TileLayer") {
 55 				// Ignore if the relative dimensions haven't been retrieved yet
 56 				if (Object.isNumber(l.relWidth)) {
 57 					maxWidth  = Math.max(maxWidth,  l.relWidth);
 58 					maxHeight = Math.max(maxHeight, l.relHeight); 
 59 				}
 60 			}
 61 		});
 62 		
 63 		//console.log("Max dimensions: " + maxWidth + ", " + maxHeight);
 64 		
 65 		return {width: maxWidth, height: maxHeight};
 66 	},
 67 	
 68 	/**
 69 	 * @description Checks for presence of a specific event catalog
 70 	 * @param {String} catalog Catalog ID
 71 	 */
 72 	hasEventCatalog: function (catalog) {
 73 		return (this.eventLayers().find(function (l) {
 74 			return l.catalog === catalog;
 75 		}) ? true : false);
 76 	},
 77 	
 78 	/**
 79 	 * @description Returns only event-layers.
 80 	 * @returns {Array} An array containing each of the currently displayed EVENT layers
 81 	 */
 82 	eventLayers: function () {
 83 		return this.layers.findAll(function (l) { 
 84 			return l.type === "EventLayer";
 85 		});
 86 	},
 87 	
 88 	/**
 89 	 * @description Returns only tile-layers.
 90 	 * @returns {Array} An array containing each of the currently displayed TILE layers
 91 	 */
 92 	tileLayers: function () {
 93 		return this.layers.findAll(function (l) {
 94 			return l.type === "TileLayer";
 95 		});
 96 	},
 97 	
 98 	/**
 99 	 * @description Gets the number of event layers currently loaded
100 	 * @return {Integer} Number of event layers present.
101 	 */
102 	numEventLayers: function () {
103 		var n = 0;
104 		this.layers.each(function (l) {
105 			if (l.type === "EventLayer") {
106 				n += 1;
107 			}
108 		});
109 				
110 		return n;
111 	},
112 	
113 	/**
114 	 * @description Removes a layer
115 	 * @param {Object} The layer to remove
116 	 */
117 	removeLayer: function (layer) {
118 		layer.domNode.remove();
119 		this.layers = this.layers.without(layer);
120 	},
121 	
122 	/**
123 	 * @description Reload layers
124 	 */
125 	reloadLayers: function () {
126 		this.layers.each(function (layer) {
127 			layer.reload();
128 		});
129 	},
130 
131 	/**
132 	 * @description Reloads each of the tile layers
133 	 */
134 	resetLayers: function (visible) {
135 		this.layers.each(function (layer) {
136 			layer.reset(visible);
137 		});
138 	},
139 	
140 	/**
141 	 * @description Updates the list of loaded tile layers stored in cookies
142 	 */
143 	refreshSavedTileLayers: function () {
144 		//console.log("refreshSavedTileLayers");
145 		var tilelayers = [];
146 		
147 		this.tileLayers().each(function (layer) {
148 			var settings = {
149 				tileAPI     : layer.tileAPI,
150 				observatory : layer.observatory,
151 				instrument  : layer.instrument, 
152 				detector    : layer.detector,
153 				measurement : layer.measurement
154 			};
155 			
156 			tilelayers.push(settings);
157 		});
158 		
159 		this.controller.userSettings.set('tile-layers', tilelayers);
160 	}
161 });
