1 /**
  2  * @fileOverview Contains class definition for a simple layer manager
  3  * @author <a href="mailto:vincent.k.hughitt@nasa.gov">Keith Hughitt</a>
  4  */
  5 /*global LayerManager, Class, UIElement, $A */
  6 var LayerManager = Class.create(UIElement,
  7 	/** @lends LayerManager.prototype */
  8 	{
  9     /**
 10      * @constructs
 11      * @description Creates a new LayerManager
 12      * @param {Object} A Rseference to the main application controller
 13      */
 14     initialize: function (controller) {
 15         this.controller  = controller;
 16         this.layers      = $A([]);
 17         this.labels      = false;
 18         this.distributed = true;
 19     },
 20     
 21     //hasId: function (id) {
 22 		//return (this.layers.grep(id).length > 0 ? true : false);
 23 	//},
 24 
 25 	/**
 26 	 * @description Add a new layer
 27 	 */
 28 	addLayer: function (layer) {
 29 		this.layers.push(layer);
 30 	},
 31 	
 32 	/**
 33 	 * @description Adds a layer that is not already displayed
 34 	 */
 35 	addNewLayer: function () {
 36 		var priorityQueue, currentLayers, p, api, serverNum, defaultChoice = "SOHEITEIT171";
 37 		priorityQueue = $A([
 38 			"SOHEITEIT304", "SOHLAS0C20WL", "SOHLAS0C30WL", "SOHLAS0C20WL", "SOHMDIMDImag", "SOHMDIMDIint", "SOHEITEIT171", "SOHEITEIT284", "SOHEITEIT195"
 39 		]);
 40 		
 41 		// current layers in above form
 42 		currentLayers = $A([]);
 43 		this.tileLayers().each(function (l) {
 44 			currentLayers.push(l.observatory + l.instrument + l.detector + l.measurement);
 45 		});
 46 		
 47 		// remove existing layers from queue
 48 		currentLayers.each(function(id) {
 49 			priorityQueue = priorityQueue.without(id);
 50 		});
 51 		
 52 		p = priorityQueue.first() || defaultLayer;
 53         
 54         // Selected tiling server if distributed tiling is enabling
 55         if ((this.distributed === true) && ((this.numTileLayers() % 2) == 0)) {
 56             serverNum = 2;
 57             api = this.controller.tileServer2;
 58         } else {
 59             serverNum = 1;
 60             api = this.controller.tileServer1;
 61         }
 62 		
 63 		this.addLayer(new TileLayer(this.controller.viewport, {
 64             tileAPI: api,
 65             server:  serverNum,
 66             observatory: p.substr(0,3),
 67             instrument:  p.substr(3,3),
 68             detector:    p.substr(6,3),
 69             measurement: p.substr(9,3),
 70             startOpened: true
 71         }));
 72 		this.refreshSavedTileLayers();
 73 	},
 74 	
 75 	/**
 76 	 * @description Gets the number of TileLayers currently loaded
 77 	 * @return {Integer} Number of tile layers present.
 78 	 */
 79 	numTileLayers: function () {
 80 		var n = 0;
 81 		this.layers.each(function (l) {
 82 			if (l.type === "TileLayer") {
 83 				n += 1;
 84 			}
 85 		});
 86 		
 87 		return n;
 88 	},
 89 	
 90 	/**
 91 	 * @description Returns the largest width and height of any layers (does not have to be from same layer)
 92 	 * @returns {Object} The width and height of the largest layer
 93 	 */
 94 	getMaxDimensions: function () {
 95 		var maxWidth  = 0,
 96 			maxHeight = 0;
 97 		
 98 		this.layers.each(function (l) {
 99 			if (l.type === "TileLayer") {
100 				// Ignore if the relative dimensions haven't been retrieved yet
101 				if (Object.isNumber(l.relWidth)) {
102 					maxWidth  = Math.max(maxWidth,  l.relWidth);
103 					maxHeight = Math.max(maxHeight, l.relHeight); 
104 				}
105 			}
106 		});
107 		
108 		//console.log("Max dimensions: " + maxWidth + ", " + maxHeight);
109 		
110 		return {width: maxWidth, height: maxHeight};
111 	},
112 	
113 	/**
114 	 * @description Checks for presence of a specific event catalog
115 	 * @param {String} catalog Catalog ID
116 	 */
117 	hasEventCatalog: function (catalog) {
118 		return (this.eventLayers().find(function (l) {
119 			return l.catalog === catalog;
120 		}) ? true : false);
121 	},
122 	
123 	/**
124 	 * @description Returns only event-layers.
125 	 * @returns {Array} An array containing each of the currently displayed EVENT layers
126 	 */
127 	eventLayers: function () {
128 		return this.layers.findAll(function (l) { 
129 			return l.type === "EventLayer";
130 		});
131 	},
132 	
133 	/**
134 	 * @description Returns only tile-layers.
135 	 * @returns {Array} An array containing each of the currently displayed TILE layers
136 	 */
137 	tileLayers: function () {
138 		return this.layers.findAll(function (l) {
139 			return l.type === "TileLayer";
140 		});
141 	},
142 	
143 	/**
144 	 * @description Gets the number of event layers currently loaded
145 	 * @return {Integer} Number of event layers present.
146 	 */
147 	numEventLayers: function () {
148 		var n = 0;
149 		this.layers.each(function (l) {
150 			if (l.type === "EventLayer") {
151 				n += 1;
152 			}
153 		});
154 				
155 		return n;
156 	},
157 	
158 	/**
159 	 * @description Removes a layer
160 	 * @param {Object} The layer to remove
161 	 */
162 	removeLayer: function (layer) {
163 		layer.domNode.remove();
164         //this.controller.viewport.sandbox.getDimensions();
165         //this.controller.viewport.updateSandbox();
166         //this.controller.viewport.sandbox.getDimensions();
167 		this.layers = this.layers.without(layer);
168 	},
169 	
170 	/**
171 	 * @description Reload layers (For tile layers, finds closest image)
172 	 */
173 	reloadLayers: function () {
174 		this.layers.each(function (layer) {
175 			layer.reload();
176 		});
177 	},
178 
179 	/**
180 	 * @description Resets each of the layers
181 	 */
182 	resetLayers: function () {
183 		this.layers.each(function (layer) {
184 			layer.reset(true);
185 		});
186 	},
187     
188     /**
189      * @description Returns the current label visibility
190      */
191     getLabelVisibility: function () {
192         if (this.labels === true) {
193             return "inline";
194         }
195         else {
196             return "none";
197         }
198     },
199     
200     /**
201      * @description Toggle event label visibility
202      */
203     toggleLabels: function () {
204         this.labels = !this.labels;
205         jQuery('.event-label').toggle();
206     },
207 
208 	/**
209 	 * @description Updates the list of loaded tile layers stored in cookies
210 	 */
211 	refreshSavedTileLayers: function () {
212 		//console.log("refreshSavedTileLayers");
213 		var tilelayers = [];
214 		
215 		this.tileLayers().each(function (layer) {
216 			var settings = {
217 				tileAPI     : layer.tileAPI,
218                 server      : layer.server,
219 				observatory : layer.observatory,
220 				instrument  : layer.instrument, 
221 				detector    : layer.detector,
222 				measurement : layer.measurement
223 			};
224 			
225 			tilelayers.push(settings);
226 		});
227 		
228 		this.controller.userSettings.set('tile-layers', tilelayers);
229 	}
230 });
