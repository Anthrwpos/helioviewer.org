 43     /**
 44      * @description Adds a new entry to the tile layer accordion
 45      * @param {Object} layer The new layer to add
 46      */
 47     addLayer: function (layer) {
 48         // Create accordion entry header
 49         var visibilityBtn, removeBtn, hidden, head, body, slider, ids, selected, obs, inst, det, meas, self = this;
 50         
 51         // initial visibility
 52         hidden = (layer.visible ? "" : " hidden");
 53         
 54         visibilityBtn = "<span class='layerManagerBtn visible" + hidden + "' id='visibilityBtn-" + layer.htmlId +
 55                         "' title='Toggle layer visibility'></span>";
 56         removeBtn = "<span class='ui-icon ui-icon-closethick removeBtn' id='removeBtn-" + layer.htmlId +
 57                     "' title='Remove layer'></span>";
 58         head = "<div class='layer-Head ui-accordion-header ui-helper-reset ui-state-default ui-corner-all'>" + 
 59                "<span class=tile-accordion-header-left>" + layer.name +
 60                "</span><span class=tile-accordion-header-right><span class=timestamp></span>" + 
 61                "<span class=accordion-header-divider>|</span>" + visibilityBtn + removeBtn + "</span></div>";
 62         
 63         // Create accordion entry body
 64         body = this._buildEntryBody(layer);
 65         
 66         //Add to accordion
 67         this.domNode.dynaccordion("addSection", {
 68             id:     layer.htmlId,
 69             header: head,
 70             cell:   body,
 71             open:   layer.startOpened
 72         });
 73 
 74         // SELECT id's
 75         obs  = "#observatory-select-" + layer.htmlId;
 76         inst = "#instrument-select-" + layer.htmlId;
 77         det  = "#detector-select-" + layer.htmlId;
 78         meas = "#measurement-select-" + layer.htmlId;
 79         ids = [obs, inst, det, meas];
 80 
 81         // Initialize TreeSelect
 82         selected  = [layer.observatory, layer.instrument, layer.detector, layer.measurement];
 83         this.selectMenus = new TreeSelect(ids, this.controller.dataSources, selected, function (leaf) {
 84             layer.observatory = $(obs).attr("value");
 85             layer.instrument  = $(inst).attr("value");
 86             layer.detector    = $(det).attr("value");
 87             layer.measurement = $(meas).attr("value");
 88             
 89             layer.sourceId = leaf.sourceId;
 90             layer.name     = leaf.name;
 91             layer.layeringOrder = leaf.layeringOrder;
 92             
 93             layer.reload();
 94             self.tileLayers.save();
 95         });
 96 
 97         slider = $("#opacity-slider-track-" + layer.htmlId).slider({
 98             value: layer.opacity,
 99             min  : 0,
100             max  : 100,
101             slide: function (e, ui) {
102                 if ((ui.value % 2) === 0) {
103                     layer.setOpacity(ui.value);
104                 }
105             },
106             change: function (e, ui) {
107                 layer.setOpacity(ui.value);
108                 self.tileLayers.save();
109             }
110         });
111         
112         // Keep a reference to the dom-node
113         //this.menuEntries.push({id: layer.id, header: head, cell: body});
114         this.layerSettings.push({
115             id    : layer.id,
116             header: head,
117             body  : body,
118             opacitySlider: slider
119         });
120         
121         // Event-handlers
122         this._setupEventHandlers(layer);
123         
124         // Update timestamp
125         this.updateTimeStamp(layer);
126             
127         // Setup tooltips
128         // Note: disabling until event-handler issues can be worked out
129         // See http://dev.jquery.com/ticket/4591
130         this._setupTooltips(layer.id);
131     },

133     /**
134      * @description Returns the layer settings associated with the given layer id
135      * @param {Object} id
136      */
137     getLayerSettings: function (id) {
138         var matched = $.grep(this.layerSettings, function (layer) {
139             return layer.id === id;
140         });
141         if (matched.length > 0) {
142             return matched.pop();
143         }
144         else {
145             return false;
146         }
147     },

278     /**
279      * @description Sets up event-handlers for a TileLayerAccordion entry
280      * @param {Object} layer The layer being added
281      */
282     _setupEventHandlers: function (layer) {
283         var toggleVisibility, removeLayer, visible, accordion, ids, self = this,
284             visibilityBtn = $("#visibilityBtn-" + layer.htmlId),
285             removeBtn     = $("#removeBtn-" + layer.htmlId),
286             fitsBtn       = $("#showFITSBtn-" + layer.htmlId);
287 
288         // Function for toggling layer visibility
289         toggleVisibility = function (e) {
290             visible = layer.toggleVisibility();
291             $("#visibilityBtn-" + layer.htmlId).toggleClass('hidden');
292             self.tileLayers.save();
293             e.stopPropagation();
294         };
295 
296         // Function for handling layer remove button
297         removeLayer = function (e) {
298             accordion = e.data;
299             self.tileLayers.removeLayer(layer);
300             
301             accordion._removeTooltips(layer.htmlId);
302             
303             accordion.domNode.dynaccordion('removeSection', {id: layer.htmlId});
304             accordion.removeLayerSettings(layer.id);
305             self.tileLayers.save();
306 
307             //accordion.layers = accordion.layers.without(layer.id);
308             e.stopPropagation();
309         };
310         
311         ids = ["#observatory-select-" + layer.htmlId, "#instrument-select-" + layer.htmlId, "#detector-select-" +
312               layer.htmlId, "#measurement-select-" + layer.htmlId];
313 
314         // Display FITS header
315         fitsBtn.bind('click', function () {
316             self._showFITS(layer);
317         });
318 
319         visibilityBtn.bind('click', this, toggleVisibility);
320         removeBtn.bind('click', this, removeLayer);
321     },

323     /**
324      * @description Displays the FITS header information associated with a given image
325      * @param {Object} layer
326      */
327     _showFITS: function (layer) {
328         var dialogId, sortBtn, formatted, params, callback;
329         
330         dialogId = "fits-header-" + layer.htmlId;
331         
332         // Check to see if a dialog already exists
333         if ($("#" + dialogId).length !== 0) {
334             if (!$("#" + dialogId).dialog("isOpen")) {
335                 $("#" + dialogId).dialog("open");
336             }
337             else {
338                 $("#" + dialogId).dialog("close");
339             }
340                 
341             return;
342         }
343         
344         // Ajax Responder
345         callback = function (response) {
346 
347             // Format results
348             formatted =  "<div id='" + dialogId + "' style='overflow: auto; position: relative; padding:0px'>";
349             formatted += "<div class='fits-regular'>";
350             $.each(response, function () {
351                 formatted += this + "<br>";
352             });
353             formatted += "</div>";
354             
355             // Store a sort version as well
356             formatted += "<div class='fits-sorted' style='display: none;'>";
357             $.each(response.sort(), function () {
358                 formatted += this + "<br>";
359             });
360             
361             formatted += "</div></div>";
362             
363             $("body").append(formatted);
364 
365             // Button to toggle sorting
366             sortBtn = "<span class='fits-sort-btn'>Abc</span>";
367             $("#" + dialogId).append(sortBtn);    
368             $("#" + dialogId + " > span").click(function () {
369                 $(this).toggleClass("italic");
370                 $("#" + dialogId + " .fits-sorted").toggle();
371                 $("#" + dialogId + " .fits-regular").toggle();
372             });
373                             
374             $("#" + dialogId).dialog({
375                 autoOpen: true,
376                 title: "FITS Header: " + layer.name,
377                 width: 400,
378                 height: 350,
379                 draggable: true
380             });
381         };
382         
383         // Request parameters
384         params = {
385             action: "getJP2Header",
386             file: layer.filepath + "/" + layer.filename
387         };
388         
389         $.post("api/index.php", params, callback, "json");
390     },

408     /**
409      * @description Updates the displayed timestamp for a given tile layer
410      * @param {Object} layer The layer being updated
411      */
412     updateTimeStamp: function (layer) {
413         var domNode, date, timeDiff, timestep;
414         
415         date     = new Date(getUTCTimestamp(layer.date));
416         timeDiff = (date.getTime() - this.controller.date.getTime()) / 1000;
417         timestep = this.controller.timeIncrementSecs;
418         
419         domNode = $("#" + layer.htmlId).find('.timestamp').html(layer.date.replace(/-/g, "/"));
420         domNode.removeClass("timeAhead timeBehind timeSignificantlyOff");
421         
422         if (Math.abs(timeDiff) > (4 * timestep)) {
423             domNode.addClass("timeSignificantlyOff");
424         }
425         else if (timeDiff < 0) {
426             domNode.addClass("timeBehind");
427         }
428         else if (timeDiff > 0) {
429             domNode.addClass("timeAhead");
430         }
431     },
