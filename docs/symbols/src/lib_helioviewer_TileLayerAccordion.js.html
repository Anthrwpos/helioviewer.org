369 	/**
370  	 * @description Checks to make sure the new layer settings are valid. If the new combination of
371 	 * choices are not compatable, change values to right of most-recently changed parameter to valid
372 	 * settings. Once the combination is acceptable, reload the tile layer.
373 	 * @param {TileLayer} layer The layer to which the changes have been made
374 	 * @param {String} changed The field altered
375 	 * @param {String} The new value chosen
376 	 */
377 	_onLayerSelectChange: function (layer, changed, value) {
378 		var obs, inst, det, meas, xhr, processResponse;
379 		
380 		// Ajax callback function
381 		processResponse = function (transport) {
382 			// Update options
383 			this.options = transport.responseJSON;
384 
385 			// Case 1: Observatory changed
386 			if (changed === "observatory") {
387 				this._updateOptions(layer.id, "instrument", this.options.instruments);
388 				
389 				//Make sure the instrument choice is still valid.
390 				if ($A(this.options.instruments).grep(layer.instrument).length === 0) {
391 					layer.instrument = this.options.instruments[0];
392 				}
393 			}
394 			
395 			// Case 2: Instrument changed
396 			if ((changed === "observatory") || (changed === "instrument")) {
397 				this._updateOptions(layer.id, "detector", this.options.detectors);
398 				
399 				//Make sure the detector choice is still valid.
400 				if (!$A(this.options.detectors).find(function (det) {
401     				return det.abbreviation === layer.detector;
402 				})) {
403 					layer.detector = this.options.detectors[0].abbreviation;
404 				}
405 			}
406 			
407 			// Case 3: Detector changed
408 			if ((changed === "observatory") || (changed === "instrument") || (changed === "detector")) {
409 				this._updateOptions(layer.id, "measurement", this.options.measurements);	
410 				
411 				//Make sure the measurement choice is still valid.
412 				if (!$A(this.options.measurements).find(function (meas) {
413     				return meas.abbreviation === layer.measurement;
414 				})) {
415 					layer.measurement = this.options.measurements[0].abbreviation;
416 				}
417 				
418 				
419 				//if ($A(this.options.measurements).grep(layer.measurement).length == 0) {
420 				//	layer.measurement = this.options.measurements[0];
421 				//}
422 				/*
423 				var instVal = $F('instrument-select-' + layer.id);
424 				if ($A(this.options.instruments).grep(instVal).length == 0) {
425 					layer.instrument = this.options.instruments[0];
426 					
427 					//update selectedIndex
428 					var self = this;
429 					$$('#instrument-select-' + layer.id + ' > option').each(function(opt, i) {
430 						if (opt.value === self.options.instruments[0]) {
431 							$('instrument-select-' + layer.id).selectedIndex = i;
432 						}
433 					});
434 				}*/
435 			}
436 			
437 			//reload layer settings
438 			layer.reload();
439 			
440 			// Update stored user settings
441 			this.layerManager.refreshSavedTileLayers();
442 		};
443 		
444 		// Do not need to update options if the measurement is changed
445 		if (changed !== "measurement") {
446 			// Update SELECT options
447 			obs  = (changed === "observatory" ? value : layer.observatory);
448 			inst = (changed === "instrument"  ? value : layer.instrument);
449 			det  = (changed === "detector"    ? value : layer.detector);
450 			meas = (changed === "measurement" ? value : layer.measurement);
451 			
452 			// Ajax Request
453 			xhr = new Ajax.Request(this.queryURL, {
454 				method: 'POST',
455 				onSuccess: processResponse.bind(this),
456 				parameters: {
457 					action: "getLayerAvailability",
458 					observatory: obs,
459 					instrument: inst,
460 					detector: det,
461 					measurement: meas,
462 					changed: changed,
463 					value: value
464 				}
465 			});
466 		}
467 		else {
468 			//reload layer settings
469 			layer.reload();
470 			
471 			// Update stored user settings
472 			this.layerManager.refreshSavedTileLayers();
473 		}
474 	},

531     /**
532      * @description Updates the displayed timestamp for a given tile layer
533      * @param {Object} layer The layer being updated
534      */
535     updateTimeStamp: function (layer) {
536 		var domNode, date, dateString, timeDiff, ts;
537 		
538     	//Grab timestamp dom-node
539     	domNode = $(layer.id).select('.timestamp').first();
540     	
541         //remove any pre-existing styling
542         domNode.removeClassName("timeBehind");
543         domNode.removeClassName("timeAhead");
544         domNode.removeClassName("timeSignificantlyOff");
545                 
546         // Update the timestamp
547         date = new Date(layer.timestamp * 1000);
548         dateString = date.toYmdUTCString() + ' ' + date.toHmUTCString();
549 
550         // Calc the time difference
551         timeDiff = layer.timestamp - this.layerManager.controller.date.getTime() / 1000;
552 
553         //this.domNode.select(".timestamp").first().update(dateString + ' ' + timeDiffStr);
554         domNode.update(dateString);
555         
556         //get timestep (TODO: create a better accessor)
557         //var ts = this.layerManager.controller.timeStepSlider.timestep.numSecs;
558 		ts = this.layerManager.controller.timeIncrementSecs;
559         
560         // Check to see if observation times match the actual time
561         if (timeDiff < 0) {
562         	if (Math.abs(timeDiff) > (4 * ts)) {
563         		domNode.addClassName("timeSignificantlyOff");
564         	}
565         	else {
566         		domNode.addClassName("timeBehind");
567         	}
568         }
569         else if (timeDiff > 0) {
570         	if (timeDiff > (4 * ts)) {
571         		domNode.addClassName("timeSignificantlyOff");
572         	}
573         	else {
574         		domNode.addClassName("timeAhead");
575         	}
576         }
577     },
