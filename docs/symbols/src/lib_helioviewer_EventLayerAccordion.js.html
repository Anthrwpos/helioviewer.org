1 /**
  2  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  3  * @fileOverview Contains the class definition for an EventLayerAccordion class.
  4  * @see EventLayer, LayerManager, TileLayerAccordion
  5  * @requires ui.dynaccordion.js

  7  * Syntax: jQuery, Prototype
  8  */
  9 /*global Class, EventLayerAccordion, $, jQuery, Ajax, Event, EventLayer, Hash, IconPicker, Layer */
 10 var EventLayerAccordion = Class.create(Layer,
 11 	/** @lends EventLayerAccordion.prototype */
 12 	{
 13 	/**
 14 	 * @constructs
 15 	 * @description Creates a new EventLayerAccordion
 16 	 * @param {Object} controller A reference to the application controller class
 17 	 * @param {String} containerId The ID of the container where the EventLayerAccordion should be placed.
 18 	 */
 19 	initialize: function (controller, containerId) {
 20 		this.controller = controller;
 21         this.eventLayers = controller.eventLayers;
 22 		this.container = jQuery('#' + containerId);
 23 		
 24 		// Default icons
 25 		this.eventIcons = controller.userSettings.get('eventIcons');
 26 
 27 		// Setup menu UI components
 28 		this._setupUI();
 29 		
 30 		// Setup icon-picker
 31 		this.iconPicker = new IconPicker('event-icon-menu');
 32 
 33 		// Initialize accordion
 34 		this.domNode = jQuery('#EventLayerAccordion-Container');
 35 		this.domNode.dynaccordion();
 36 
 37 		// Get Event Catalogs
 38 		this.getEventCatalogs();
 39 	},
 40 
 41 	/**
 42 	 * @description Adds a new entry to the event layer accordion
 43 	 * @param {Object} layer EventLayer to add to the accordion.
 44 	 */
 45 	addLayer: function (layer) {
 46 		// Create accordion entry header
 47 		var catalog = this.eventCatalogs.get(layer.catalog),		
 48 			etype = catalog.eventType.gsub(' ', '_'),
 49 			icon, visibilityBtn, removeBtn, head, body;
 50 		
 51 		layer.icon = this.eventIcons[catalog.id] || 'small-blue-circle';
 52 		
 53 		icon = "<span class=accordion-header-divider>|</span><button class='event-accordion-icon' id='event-icon-" + layer.id + "' title=' - Select an icon to use for this event' style='background: url(images/events/" + layer.icon + "-" + etype + ".png);'></button>";
 54 		visibilityBtn = "<span class='layerManagerBtn visible' id='visibilityBtn-" + layer.id + "' title=' - Toggle layer visibility'></span>";
 55 		removeBtn = "<span class='ui-icon ui-icon-closethick removeBtn' id='removeBtn-" + layer.id + "' type=button title=' - Remove layer'></span>";
 56 		head = "<div class='layer-Head ui-accordion-header ui-helper-reset ui-state-default ui-corner-all'><span class=event-accordion-header-left>" + catalog.name + "</span><span class=event-accordion-header-right>" + icon + visibilityBtn + removeBtn + "</span></div>";
 57 
 58 		// Create accordion entry body
 59 		body = '<div style="color: white; position: relative;">' + catalog.description + '</div>';
 60 
 61 		// Add to accordion
 62 		this.domNode.dynaccordion("addSection", {id: layer.id, header: head, cell: body});
 63 
 64         // Add tooltip
 65         this._setupTooltips(layer.id);
 66 
 67 		// Event-handlers
 68 		this._setupEventHandlers(layer);
 69 	},
 70 
 71 	/**
 72 	 * @description Get a list of the available event catalogs.
 73 	 */
 74 	getEventCatalogs: function () {
 75 		//@TODO: Move this to the new layer manager.
 76 		
 77 		// Ajax responder
 78 		var processResponse = function (transport) {
 79 			var lm = this.eventLayers,
 80 				catalogs = transport.responseJSON,
 81 				self = this
 82 			
 83 			if (typeof(catalogs) !== "undefined") {
 84 				this.eventCatalogs = new Hash();
 85 				
 86 				catalogs.each(function (catalog) {
 87 					//Ignore EIT Activity reports for the time being
 88 					if (catalog.id !== "EITPlanningService::EITActivity") {
 89 						self.eventCatalogs.set(catalog.id, catalog);
 90 					}
 91 				});
 92 				
 93 				//Initial catalogs to load
 94 				lm.addLayer(new EventLayer(this.controller.viewport, {
 95 					catalog: "VSOService::cmelist",
 96 					eventAccordion: this
 97 				}));
 98 				lm.addLayer(new EventLayer(this.controller.viewport, {
 99 					catalog: "GOESXRayService::GOESXRay",
100 					eventAccordion: this
101 				}));
102 				lm.addLayer(new EventLayer(this.controller.viewport, {
103 					catalog: "VSOService::noaa",
104 					eventAccordion: this,
105 					windowSize: 86400
106 				}));
107 			} else {
108 				this._catalogsUnavailable();
109 			}
110 		},
111 
112 		xhr = new Ajax.Request(this.controller.api, {
113 			method: "POST",
114 			parameters: { action: "getEventCatalogs" }, 
115 			onSuccess: processResponse.bind(this)
116 		});
117 	},
118 
119 	/**
120 	 * @description Setup empty event layer accordion.
121 	 */
122 	_setupUI: function () {
123 		// Create a top-level header and an "add layer" button
124 		var title = jQuery('<span class="section-header">Features/Events</span>').css({'float': 'left'}),
125 			addLayerBtn = jQuery('<a href=# class=dark>[Add]</a>').css({'margin-right': '14px'}),
126 			self = this;
127 		this.container.append(jQuery('<div></div>').css('text-align', 'right').append(title).append(addLayerBtn));
128 		this.container.append(jQuery('<div id="EventLayerAccordion-Container"></div>'));
129 		
130 		// Event-handlers
131 		addLayerBtn.click(function () {
132 			if (typeof(self.eventCatalogs) !== "undefined") {
133 				//Populate select-box
134 				var select = self._buildCatalogSelect(),
135 				
136 					okayBtn = "<button>Ok</button>",
137 				
138 					//Add to accordion
139 					tmpId = 'tmp' + new Date().getTime();
140 				self.domNode.dynaccordion("addSection", {
141 					id: tmpId,
142 					header: '<div class="layer-Head">Select catalog to use:</div>',
143 					cell: select + okayBtn,
144 					open: true
145 				});
146 				
147 				// Swap out for feature/event catalog when user changes
148 				Event.observe($(tmpId).select('button').first(), 'click', function (e) {
149 					var select = $(tmpId).select('select').first();
150 					self.domNode.dynaccordion('removeSection', {
151 						id: tmpId
152 					});
153 					self.eventLayers.addLayer(new EventLayer(self.controller.viewport, {
154 						catalog: select.value,
155 						eventAccordion: self
156 					}));
157 				});
158 			} else {
159 				//Do nothing
160 			}
161 		});
162 	},
163 	
164 	/**
165 	 * @description Builds a SELECT menu with all the catalogs not already displayed.
166 	 */
167 	_buildCatalogSelect: function () {
168 		var self = this,
169 			select = "<select class='event-layer-select' style='margin:5px;'>";
170 
171 		this.eventCatalogs.each(function (catalog) {
172 			if (!self.eventLayers.contains(catalog.key)) {
173 				select += "<option value=" + catalog.key + ">" + catalog.value.name + "</option>";
174 			}
175 		});
176 		select += "</select>";
177 		
178 		return select;
179 	},
180 	
181 	/**
182 	 * @description Display error message to let user know service is down.
183 	 */
184 	_catalogsUnavailable: function () {
185 		// Handle button
186 		$('eventAccordion').select('a.dark')[0].update("");
187 		
188 		// Error message
189 		var head = "<div class='layer-Head ui-accordion-header ui-helper-reset ui-state-default ui-corner-all'>Currently Unavailable</div>",
190 			body = "<span style='color: #FFF;'>Helioviewer's Feature/Event catalog system is currently unavailable. Please try again later.</span>";
191 		this.domNode.dynaccordion("addSection", {
192             id: 404,
193             header: head,
194             cell: body,
195             open: true
196         });
197 	},
198     
199     /**
200      * @description Initialize custom tooltips for each icon in the accordion
201      */
202     _setupTooltips: function (id) {
203         var items = $A([
204 			"#event-icon-"    + id,
205             "#visibilityBtn-" + id,
206             "#removeBtn-"     + id
207 		]),
208 
209 		hv = this.controller;
210 		items.each(function (item) {
211 			hv.addToolTip(item, {yOffset: -125});
212 		});
213     },
214     
215     /**
216      * @description Unbinds event-hanlders relating to accordion header tooltips
217      * @param {String} id
218      */
219     _removeTooltips: function (id) {
220         var ids = $A([
221 			"#event-icon-"    + id,
222             "#visibilityBtn-" + id,
223             "#removeBtn-"     + id
224 		]);
225         
226         ids.each(function (id) {
227 			jQuery(id).unbind();
228 		});
229     },
230 
231 	/**
232 	 * @description Sets up UI-related event handlers
233 	 * @param {Object} layer EventLayer being added to the accordion.
234 	 */
235 	_setupEventHandlers: function (layer) {
236 		var visibilityBtn = jQuery("#visibilityBtn-" + layer.id),
237 			removeBtn = jQuery("#removeBtn-" + layer.id),
238 			eventIcon = jQuery("#event-icon-" + layer.id),
239 			self = this,
240 
241 		/**
242 		 * @inner
243 		 * @description Toggles layer visibility
244 		 * @param {Object} e jQuery Event Object.
245 		 */
246 		toggleVisibility = function (e) {
247 			var visible = layer.toggleVisibility(),
248 				icon = (visible ? 'LayerManagerButton_Visibility_Visible.png' : 'LayerManagerButton_Visibility_Hidden.png');
249 			jQuery("#visibilityBtn-" + layer.id).css('background', 'url(images/blackGlass/' + icon + ')');
250 			e.stopPropagation();
251 		},
252 
253 		/**
254 		 * @inner
255 		 * @description Layer remove button event-handler
256 		 * @param {Object} e jQuery Event Object.
257 		 */
258 		removeLayer = function (e) {
259 			var self = e.data;
260 			self.eventLayers.removeLayer(layer);
261             self._removeTooltips(layer.id);
262 			self.domNode.dynaccordion('removeSection', {id: layer.id});
263 			e.stopPropagation();
264 		},
265 		
266 		/**
267 		 * @inner
268 		 * @description Icon selection menu event-handler
269 		 * @param {Object} e jQuery Event Object.
270 		 */
271 		showIconMenu = function (e) {
272 			layer = e.data;
273 			self.iconPicker.toggle(layer, jQuery(this).position());
274 			e.stopPropagation();
275 		};
276 
277 		visibilityBtn.bind('click', this, toggleVisibility);
278 		removeBtn.bind('click', this, removeLayer);
279 		eventIcon.bind('click', layer, showIconMenu);
280 	}
281 });
