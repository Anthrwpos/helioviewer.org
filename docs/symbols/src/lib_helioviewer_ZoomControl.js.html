1 /**
  2  * @fileOverview Contains the class definition for an ZoomControl class.
  3  * @author <a href="mailto:vincent.k.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @author <a href="mailto:patrick.schmiedel@gmx.net">Patrick Schmiedel</a>
  5  * @see  The <a href="http://helioviewer.org/mediawiki-1.11.1/index.php?title=Zoom_Levels_and_Observations">HelioViewer Wiki</a>
  6  *       for more information on zoom levels.
  7  */
  8 /*global UIElement, Class, Control, Event, $R, $ */
  9 var ZoomControl = Class.create(UIElement,
 10 	/** @lends ZoomControl.prototype */
 11 	{
 12 	/**
 13 	 * @constructs
 14 	 * @description Creates a new ZoomControl
 15 	 * @param {Object} controller A Reference to the Helioviewer application class
 16 	 * @param {Object} options Custom ZoomControl settings
 17 	 */
 18     initialize: function (controller, options) {
 19         Object.extend(this, options);
 20         this.controller = controller;
 21         this.domNode = $(this.id);
 22         this.handle =  $(this.id + 'Handle');
 23 
 24         var range = $R(this.minZoomLevel, this.maxZoomLevel);
 25         this.slider = new Control.Slider(this.handle, this.id + 'Track', {
 26             axis: 'vertical',
 27             values: range,
 28             sliderValue: this.zoomLevel,
 29             range: range,
 30             onChange: this.changed.bind(this)
 31         });
 32         //this.handle.innerHTML = this.zoomLevel;
 33         Event.observe($(this.id + 'ZoomIn'), 'click', this.zoomButtonClicked.bind(this, -1));
 34 		Event.observe($(this.id + 'ZoomIn'), 'mousedown', function (e) {
 35 			Event.stop(e);
 36 		});
 37         Event.observe($(this.id + 'ZoomOut'), 'mouseup', this.zoomButtonClicked.bind(this, 1));
 38         Event.observe($(this.id + 'ZoomOut'), 'mousedown', function (e) {
 39         	Event.stop(e);
 40         });
 41     },
 42 
 43 	/**
 44 	 * @description Increases or decreases zoom level in response to pressing the plus/minus buttons.
 45 	 * @param {Integer} dir The amount to adjust the zoom level by (+1 or -1).              
 46 	 */
 47     zoomButtonClicked: function (dir) {
 48         this.slider.setValue(this.slider.value + dir);
 49     },
 50   
 51 	/**
 52 	 * @description Adjusts the zoom-control slider
 53 	 * @param {Integer} v The new zoom value.
 54 	 */
 55     changed: function (v) {
 56     	this.fire('change', v);
 57     }
 58 });
