  7 /**
  8  * @description Pads a string to the left.
  9  * @param {String} padding Character to use for padding, e.g. " "
 10  * @param {Int} minLength Length to pad up to
 11  * @returns {String} The resulting string.
 12  */
 13 String.prototype.padLeft = function (padding, minLength) {
 14 	var str = this,
 15 		pad = '' + padding;
 16 	while (str.length < minLength) {
 17 		str = pad + str;
 18 	}
 19 	return str;
 20 };
 21 /**
 22  * @description Trims a string from the left.
 23  * @param {String} padding Character to trim.
 24  * @returns {String} The resulting string.
 25  */
 26 String.prototype.trimLeft = function (padding) {
 27 	var str = this,
 28 		pad = '' + padding;
 29 	while (str[0] === pad) {
 30 	    str = str.substr(1);
 31 	}
 32 	return str;
 33 };

 35 /**
 36  * @description Outputs a UTC Date string of the format "YYYY/MM/dd"
 37  * @returns {String} Datestring.
 38  */
 39 Date.prototype.toYmdUTCString = function () {
 40 	var year = this.getUTCFullYear() + '',
 41 		month = (this.getUTCMonth() + 1) + '',
 42 		day = this.getUTCDate() + '';
 43 	return year + '/' + month.padLeft(0, 2) + '/' + day.padLeft(0, 2);
 44 };

 46 /**
 47  * @description Outputs a UTC Date string of the format "HH:mm:ss"
 48  * @returns {String} Datestring.
 49  */
 50 Date.prototype.toHmUTCString = function () {
 51 	var hour = this.getUTCHours() + '',
 52 		min = this.getUTCMinutes() + '',
 53 		sec = this.getUTCSeconds() + '';
 54 	return hour.padLeft(0, 2) + ':' + min.padLeft(0, 2) + ':' + sec.padLeft(0, 2);
 55 };

 57 /**
 58  * @description Takes a localized javascript date and returns a date set to the UTC time.
 59  */
 60 Date.prototype.toUTCDate = function () {
 61 	var utcOffset = this.getUTCOffset(),
 62 		sign = utcOffset[0],
 63 		hours = parseInt(utcOffset.substr(1, 2), 10),
 64 		mins = parseInt(utcOffset.substr(3, 4), 10),
 65 	
 66 		numSecs = (3600 * hours) + (60 * mins);
 67 	
 68 	if (sign === "+") {
 69 		numSecs = - numSecs;
 70 	}
 71 	
 72 	this.addSeconds(numSecs);
 73 };

 88 /**
 89  * @description Converts a ISO 8601 UTC formatted date string into a (UTC) Unix timestamp
 90  *  e.g. "2003-10-05T00:00:00Z" => 1065312000000
 91  */
 92 var getUTCTimestamp = function(date) {
 93     var year, month, day, hours, minutes, seconds, ms;
 94     
 95     year    = parseInt(date.substr(0,4), 10);
 96     month   = parseInt(date.substr(5,2), 10) - 1;
 97     day     = parseInt(date.substr(8,2), 10);
 98     hours   = parseInt(date.substr(11,2), 10);
 99     minutes = parseInt(date.substr(14,2), 10);
100     seconds = parseInt(date.substr(17,2), 10);
101     ms = 0;
102 
103     return Date.UTC(year,month,day,hours,minutes,seconds,ms);
104 };

106 /**
107  * @description Determine what operating system the user is likely to be on: For use when chosing movie codecs, etc.
108  * @returns {String} Abbreviation of the user's OS
109  */
110 var getOS = function () {
111 	var os = "other";
112 	
113 	if (navigator.appVersion.indexOf("Win") !== -1) {
114 		os = "win";
115 	}
116 	if (navigator.appVersion.indexOf("Mac") !== -1) {
117 		os = "mac";
118 	}
119 	if (navigator.appVersion.indexOf("X11") !== -1) {
120 		os = "linux";
121 	}
122 	if (navigator.appVersion.indexOf("Linux") !== -1) {
123 		os = "linux";
124 	}
125 	
126 	return os;
127 };

129 /**
130  * @description Convert from cartesian to polar coordinates
131  * @param {Int} x X coordinate
132  * @param {Int} y Y coordinate
133  * @returns {Object} Polar coordinates (r, theta) resulting from conversion 
134  */
135 Math.toPolarCoords = function (x, y) {
136 	var radians = Math.atan(y / x);
137 	
138 	if  ((x > 0) && (y < 0)) {
139 		radians += (2 * Math.PI);
140 	}
141 	else if (x < 0) {
142 		radians += Math.PI;
143 	}
144 	else if ((x === 0) && (y > 0)) {
145 		radians = Math.PI / 2;
146 	}
147 	else if ((x === 0) && (y < 0)) {
148 		radians = (3 * Math.PI) / 2;
149 	}
150 		
151 	return {
152 		r     : Math.sqrt(Math.pow(x, 2) + Math.pow(y, 2)),
153 		theta : (180 / Math.PI) * radians
154 	};
155 };

157 /**
158  * @description Return base-2 logarithm of the given number (Note: log_b(x) = log_c(x) / log_c(b))
159  * @param {Number} x Number
160  * @returns {Number} The base-2 logarithm of the input value
161  */
162 Math.lg = function (x) {
163 	return (Math.log(x) / Math.log(2));
