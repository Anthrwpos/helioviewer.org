 11 /**
 12  * @description Outputs a UTC Date string of the format "YYYY/MM/dd"
 13  * @returns {String} Datestring.
 14  */
 15 Date.prototype.toUTCDateString = function () {
 16     var year  = this.getUTCFullYear()    + '',
 17         month = (this.getUTCMonth() + 1) + '',
 18         day   = this.getUTCDate()        + '';
 19     return year + '/' + month.padLeft(0, 2) + '/' + day.padLeft(0, 2);
 20 };

 22 /**
 23  * @description Outputs a UTC Date string of the format "HH:mm:ss"
 24  * @returns {String} Datestring.
 25  */
 26 Date.prototype.toUTCTimeString = function () {
 27     var hour = this.getUTCHours()  + '',
 28         min = this.getUTCMinutes() + '',
 29         sec = this.getUTCSeconds() + '';
 30     return hour.padLeft(0, 2) + ':' + min.padLeft(0, 2) + ':' + sec.padLeft(0, 2);
 31 };

 33 /**
 34  * @description Takes a localized javascript date and returns a date set to the UTC time.
 35  * 
 36  * NOTE (11/03/2009) Sign for numsecs below should be reversed?
 37  */
 38 Date.prototype.toUTCDate = function () {
 39     var utcOffset = this.getUTCOffset(),
 40         sign = utcOffset[0],
 41         hours = parseInt(utcOffset.substr(1, 2), 10),
 42         mins = parseInt(utcOffset.substr(3, 4), 10),
 43     
 44         numSecs = (3600 * hours) + (60 * mins);
 45     
 46     if (sign === "+") {
 47         numSecs = - numSecs;
 48     }
 49     
 50     this.addSeconds(numSecs);
 51 };

 53 /**
 54  * @description Pads a string to the left.
 55  * @param {String} padding Character to use for padding, e.g. " "
 56  * @param {Int} minLength Length to pad up to
 57  * @returns {String} The resulting string.
 58  */
 59 String.prototype.padLeft = function (padding, minLength) {
 60     var str = this,
 61         pad = '' + padding;
 62     while (str.length < minLength) {
 63         str = pad + str;
 64     }
 65     return str;
 66 };
 67 /**
 68  * @description Trims a string from the left.
 69  * @param {String} padding Character to trim.
 70  * @returns {String} The resulting string.
 71  */
 72 String.prototype.trimLeft = function (padding) {
 73     var str = this,
 74         pad = '' + padding;
 75     while (str[0] === pad) {
 76         str = str.substr(1);
 77     }
 78     return str;
 79 };

102 /**
103  * @description Converts a ISO 8601 UTC formatted date string into a (UTC) Unix timestamp
104  *  e.g. "2003-10-05T00:00:00Z" => 1065312000000
105  */
106 var getUTCTimestamp = function (date) {
107     var year, month, day, hours, minutes, seconds, ms;
108     
109     year    = parseInt(date.substr(0, 4), 10);
110     month   = parseInt(date.substr(5, 2), 10) - 1;
111     day     = parseInt(date.substr(8, 2), 10);
112     hours   = parseInt(date.substr(11, 2), 10);
113     minutes = parseInt(date.substr(14, 2), 10);
114     seconds = parseInt(date.substr(17, 2), 10);
115     ms = 0;
116 
117     return Date.UTC(year, month, day, hours, minutes, seconds, ms);
118 };

120 /**
121  * @description Determine what operating system the user is likely to be on: For use when chosing movie codecs, etc.
122  * @returns {String} Abbreviation of the user's OS
123  */
124 var getOS = function () {
125     var os = "other";
126     
127     if (navigator.appVersion.indexOf("Win") !== -1) {
128         os = "win";
129     }
130     if (navigator.appVersion.indexOf("Mac") !== -1) {
131         os = "mac";
132     }
133     if (navigator.appVersion.indexOf("X11") !== -1) {
134         os = "linux";
135     }
136     if (navigator.appVersion.indexOf("Linux") !== -1) {
137         os = "linux";
138     }
139     
140     return os;
141 };

143 /**
144  * @description Convert from cartesian to polar coordinates
145  * @param {Int} x X coordinate
146  * @param {Int} y Y coordinate
147  * @returns {Object} Polar coordinates (r, theta) resulting from conversion 
148  */
149 Math.toPolarCoords = function (x, y) {
150     var radians = Math.atan(y / x);
151     
152     if  ((x > 0) && (y < 0)) {
153         radians += (2 * Math.PI);
154     }
155     else if (x < 0) {
156         radians += Math.PI;
157     }
158     else if ((x === 0) && (y > 0)) {
159         radians = Math.PI / 2;
160     }
161     else if ((x === 0) && (y < 0)) {
162         radians = (3 * Math.PI) / 2;
163     }
164         
165     return {
166         r     : Math.sqrt(Math.pow(x, 2) + Math.pow(y, 2)),
167         theta : (180 / Math.PI) * radians
168     };
169 };

215 /**
216  * @description Converts coordinates from solar radii to helioprojective (arc-seconds as seen from earth).
217  * 
218  * @input {Float} rx    Solar radii from the center of the sun in the x-direction.
219  * @input {Float} ry    Solar radii from the center of the sun in the y-direction.
220  * @input {Float} scale The physical scale covered by a single pixel at the current resolution.
221  * @input {Float} rsun  The radius of the sun in pixels at the current resolution
222  * 
223  * @return {Object} Returns an object literal containing the converted x and y coordinates.
224  * 
225  * @see Viewport.getRSun
226  */
227 var solarRadiiToHelioprojective = function (rx, ry, scale, rsun) {
228     var rsunInArcSeconds = rsun * scale;
229     
230     return {
231         x: rx * rsunInArcSeconds,
232         y: ry * rsunInArcSeconds 
233     };
234 };

236 /**
237  * @description Converts coordinates from helioprojective (arc-seconds as seen from earth) to solar radii.
238  * 
239  * @input {Float} hx    Helioprojective x-coordinate.
240  * @input {Float} hy    Helioprojective y-coordinate.
241  * @input {Float} scale The physical scale covered by a single pixel at the current resolution.
242  * @input {Float} rsun  The radius of the sun in pixels at the current resolution
243  * 
244  * @return {Object} Returns an object literal containing the converted x and y coordinates.
245  * 
246  * @see Viewport.getRSun
247  */
248 var helioprojectiveToSolarRadii = function (hx, hy, scale, rsun) {
249     var rsunInArcSeconds = rsun * scale;
250     
251     return {
252         x: hx / rsunInArcSeconds,
253         y: hy / rsunInArcSeconds 
254     };
255 };
