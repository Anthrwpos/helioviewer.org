1 /**
  2  * @fileOverview Contains the class definition for an ViewportHandlers class.
  3  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  4  * @author <a href="mailto:patrick.schmiedel@gmx.net">Patrick Schmiedel</a>
  5  * 
  6  * Mouse-dragging:
  7  * 
  8  * Two throttles are maintained: one to control how frequently the image is updated on the screen,
  9  * and the other to decide when to update the visible tiles. Since the later is more computationally
 10  * expensive, it should occur less frequently. Whenever the user stops dragging, both should be forced
 11  * to update.
 12  * 
 13  */
 14 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
 15 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 16 /*global Class, document, $ */
 17 "use strict";
 18 var ViewportHandlers = Class.extend(
 19     /** @lends ViewportHandlers.prototype */
 20     {
 21     mouseStartingPosition: { x: 0, y: 0 },
 22     mouseCurrentPosition : { x: 0, y: 0 },
 23     mouseCoords          : { x: 0, y: 0 },
 24     moveCounter         : 0,
 25     imageUpdateThrottle : 3,
 26     tileUpdateThrottle  : 9,
 27     rSunArcSeconds      : 975,
 28     animatedTextShadow  : true,
 29 
 30     /**
 31      * @constructs
 32      * @description Contains a collection of event-handlers for dealing with Viewport-related events
 33      * @see Viewport
 34      * @param {Object} viewport A Reference to the Helioviewer application class
 35      */
 36     init: function (viewport) {
 37         this.viewport = viewport;
 38         this._initEvents();
 39     },
 40 
 41      /**
 42       * @description Fired when a mouse is pressed
 43       * @param {Event} event Prototype Event class
 44       */
 45     mouseDown: function (event) {
 46         var vp, sb;
 47         
 48         vp = $("#helioviewer-viewport");
 49         sb = $("#sandbox");
 50         
 51         vp.css("cursor", "all-scroll");
 52 
 53         // Don't do anything if entire image is already visible
 54         if ((sb.width() === 0) && (sb.height() === 0)) {
 55             return;
 56         }
 57         
 58         this.mouseStartingPosition = {
 59             x: event.pageX, 
 60             y: event.pageY
 61         };
 62         
 63         this.viewport.startMoving();
 64     },
 65     
 66      /**
 67       * @description Handle drag events
 68       * @param {Object} an Event object
 69       */
 70     mouseMove: function (event) {
 71         if (!this.viewport.isMoving) {
 72             return;
 73         }
 74           
 75         // Threshold
 76         this.moveCounter = this.moveCounter + 1;
 77         if ((this.moveCounter % this.imageUpdateThrottle) !== 0) {
 78             return;
 79         }
 80 
 81         this.moveCounter = this.moveCounter % this.tileUpdateThrottle;
 82 
 83         this.mouseCurrentPosition = {
 84             x: event.pageX, 
 85             y: event.pageY
 86         };
 87 
 88         // Update text-shadows
 89         //this.updateShadows();
 90 
 91         this.viewport.moveBy(this.mouseStartingPosition.x - this.mouseCurrentPosition.x,
 92                              this.mouseStartingPosition.y - this.mouseCurrentPosition.y);
 93     },
 94     
 95      /**
 96       * @description Fired when a mouse button is released
 97       * @param {Event} event Event object
 98       */
 99     mouseUp: function (event) {
100         $("#helioviewer-viewport").css("cursor", "pointer");
101 
102         if (this.viewport.isMoving) {
103             this.viewport.endMoving();
104         }
105     },
106 
107 
108     /**
109      * @description Handles double-clicks
110      * @param {Event} e Event class
111      */
112     doubleClick: function (e) {
113         var pos, viewport = this.viewport;
114           
115         //check to make sure that you are not already at the minimum/maximum zoom-level
116         if ((e.shiftKey || (viewport.zoomLevel > viewport.controller.minZoomLevel)) && 
117             (viewport.zoomLevel < viewport.controller.maxZoomLevel)) {
118             pos = this.getRelativeCoords(e.pageX, e.pageY);
119                
120             viewport.center();                    
121             this.viewport.startMoving();
122 
123             //adjust for zoom
124             if (e.shiftKey) {
125                 viewport.moveBy(0.5 * pos.x, 0.5 * pos.y);
126                 viewport.controller.zoomControls.zoomButtonClicked(-1);
127             }
128             else {
129                 viewport.moveBy(2 * pos.x, 2 * pos.y);
130                 viewport.controller.zoomControls.zoomButtonClicked(1);
131             }
132             
133             this.viewport.endMoving();
134         }
135     },
136      
137     /**
138      * @description Handles mouse-wheel movements
139      * @param {Event} event Event class
140      */
141     mouseWheel: function (e, delta) {
142         var dir = delta > 0 ? $("#zoomControlZoomIn").click() : $("#zoomControlZoomOut").click();
143     },
144      
145     /**
146      * @description Get the mouse-coords relative to top-left of the viewport frame
147      * @param {Int} screenx X-dimensions of the user's screen
148      * @param {Int} screeny Y-dimensions of the user's screen
149      */
150     getRelativeCoords: function (screenx, screeny) {
151         var offset, mouseCoords;
152          
153         offset = $('#helioviewer-viewport-container-inner').position();
154 
155         mouseCoords = {
156             x: screenx - offset.left - 1,
157             y: screeny - offset.top - 1
158         };
159 
160         return mouseCoords;
161     },

163     /**
164      * @description Adds an animated text shadow based on the position and size of the Sun (Firefox 3.5+)
165      * Added 2009/06/26
166      * TODO: Apply to other text based on it's position on screen? Adjust blue based on zoom-level?
167      *       Use viewport size to determine appropriate scales for X & Y offsets (normalize)
168      *       Re-use computeCoordinates?
169      */
170     updateShadows: function () {
171         // Not supported in older versions of Firefox, or in IE
172         if (!$.support.textShadow) {
173             return;
174         }
175         
176         var viewportOffset, sunCenterOffset, coords, viewportCenter, offsetX, offsetY;
177         
178         viewportOffset  = $("#helioviewer-viewport").offset();
179         sunCenterOffset = $("#moving-container").offset();
180 
181         // Compute coordinates of heliocenter relative to top-left corner of the viewport
182         coords = {
183             x: sunCenterOffset.left - viewportOffset.left,
184             y: sunCenterOffset.top - viewportOffset.top
185         };
186         
187         // Coordinates of heliocenter relative to the viewport center
188         viewportCenter = this.viewport.getCenter();
189         coords.x = coords.x - viewportCenter.x;
190         coords.y = coords.y - viewportCenter.y;
191         
192         // Shadow offset
193         offsetX = ((500 - coords.x) / 100) + "px";
194         offsetY = ((500 - coords.y) / 150) + "px";
195 
196         //console.log("x: " + coords.x + ", y: " + coords.y);
197         $("#footer-links > .light").css("text-shadow", offsetX + " " + offsetY + " 3px #000");
198     },

200     /**
201      * @description Toggles mouse-coords visibility
202      * 
203      * TODO (2009/07/27) Disable mouse-coords display during drag & drop
204      */
205     toggleMouseCoords: function () {
206         var vp, mouseCoordsX, mouseCoordsY, updateMouseCoords, warning, self = this;
207         
208         // Case 1: Disabled -> Arcseconds 
209         if (this.viewport.mouseCoords === "disabled") {
210             this.viewport.mouseCoords = "arcseconds";
211             $('#mouse-coords').toggle();
212         }
213 
214         // Case 2: Arcseconds -> Polar Coords
215         else if (this.viewport.mouseCoords === "arcseconds") {
216             this.viewport.mouseCoords = "polar";
217         }
218 
219         // Case 3: Polar Coords -> Disabled
220         else {
221             $('#mouse-coords').toggle();
222             this.viewport.mouseCoords = "disabled";
223         }
224           
225         // Warn once
226         if (this.viewport.controller.userSettings.get('warnMouseCoords') === false) {
227             warning = "<b>Note:</b> Mouse-coordinates should not be used for science operations!";
228             this.viewport.controller.messageConsole.log(warning);
229             this.viewport.controller.userSettings.set('warnMouseCoords', true);
230         }
231           
232         // Cartesian & Polar coords
233         if (this.viewport.mouseCoords !== "disabled") {
234             mouseCoordsX = $('#mouse-coords-x');
235             mouseCoordsY = $('#mouse-coords-y');
236                
237             // Clear old values
238             mouseCoordsX.empty();
239             mouseCoordsY.empty();
240 
241             // Remove existing event handler if switching from cartesian -> polar
242             if (this.viewport.mouseCoords === "polar") {
243                 $('#moving-container').unbind('mousemove', updateMouseCoords);
244             }
245                
246             /**
247              * @description Event-handler for displaying mouse-coords
248              * @param {Event} e
249              */
250             updateMouseCoords = function (e) {
251                 var cartesian, polar;
252                                         
253                 // Store current mouse-coordinates
254                 self.mouseCoords = {x: e.pageX, y: e.pageY};
255                 
256                 // Threshold
257                 self.moveCounter = self.moveCounter + 1;
258                 if ((self.moveCounter % self.imageUpdateThrottle) !== 0) {
259                     return;
260                 }
261 
262                 self.moveCounter = self.moveCounter % self.tileUpdateThrottle;
263                     
264                 // Compute coordinates relative to top-left corner of the viewport
265                 cartesian = self.computeMouseCoords(e.pageX, e.pageY);
266                                     
267                 // Arc-seconds
268                 if (self.viewport.mouseCoords === "arcseconds") {
269                     mouseCoordsX.html("x: " + cartesian.x + " ′′");
270                     mouseCoordsY.html("y: " + cartesian.y + " ′′");
271                          
272                 // Polar coords
273                 } else {
274                     polar = Math.toPolarCoords(cartesian.x, -cartesian.y);     
275                     
276                     mouseCoordsX.html(((polar.r / self.rSunArcSeconds) + "").substring(0, 5) +
277                         " R<span style='vertical-align: sub; font-size:10px;'>☉</span>");
278                     mouseCoordsY.html(Math.round(polar.theta) + " °");
279                 }
280             };
281             $('#moving-container').bind('mousemove', updateMouseCoords);     
282                
283             // Execute handler once immediately to show new coords
284             updateMouseCoords({pageX: this.mouseCoords.x, pageY: this.mouseCoords.y});
285                
286         } else {
287             $('#moving-container').unbind('mousemove', updateMouseCoords);
288         }
289     },

291     /**
292      * @description Computes the scaled mouse coordinates relative to the size and center of the Sun.
293      * 
294      *  Explanation:
295      * 
296      *    X = location of mouse-pointer
297      *    V = viewport top-left corner
298      *    S = sandbox top-left corner
299      *    M = moving container top-let corner
300      *    
301      *  Each of the two-letter abbreviations represents the vector <x,y> going from one
302      *  location to the other. See wiki documentation below for more details.
303      * 
304      * @see http://helioviewer.org/wiki/index.php?title=Co-ordinate_System_I
305      */
306     computeMouseCoords: function (screenX, screenY) {
307         var VX, negSV, SV, SM, MX, scale, x, y;
308         
309         // Coordinates realtive to viewport top-left corner
310         VX = this.getRelativeCoords(screenX, screenY);
311         negSV = $("#sandbox").position();
312         SV = {
313             x: -negSV.left,
314             y: -negSV.top
315         };
316         SM = $('#moving-container').position();                    
317         MX = {
318             x: VX.x + (SV.x - SM.left),
319             y: VX.y + (SV.y - SM.top)
320         };
321           
322         //scale
323         scale = this.viewport.controller.baseScale * Math.pow(
324                 2, this.viewport.zoomLevel - this.viewport.controller.baseZoom
325         );
326         x = Math.round((scale * MX.x));
327         y = - Math.round((scale * MX.y));
328         
329         // Return scaled coords
330         return {
331             x: x,
332             y: y
333         };
334     },

336     /**
337      * @description
338      */
339     _initEvents: function () {
340         var doc, vp, self = this;
341 
342         doc = $(document);
343         vp  = $("#helioviewer-viewport");
344         
345         //Mouse-related event-handlers
346         doc.mousemove(function (e) {
347             self.mouseMove(e);
348         });
349         doc.mouseup(function (e) {
350             self.mouseUp(e);
351         });
352         vp.mousedown(function (e) {
353             self.mouseDown(e);
354         });
355 
356           // Double-clicks
357         vp.dblclick(function (e) {
358             self.doubleClick(e);
359         });
360 
361           // Mouse-wheel
362         vp.mousewheel(function (e, delta) {
363             self.mouseWheel(e, delta);
364         });
365     }
