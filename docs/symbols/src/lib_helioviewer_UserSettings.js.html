1 /**
  2  * @fileOverview Contains the class definition for an UserSettings class.
  3  * Syntax: Prototype, jQuery
  4  * @author <a href="mailto:keith.hughitt@gmail.com">Keith Hughitt</a>
  5  */
  6 var UserSettings = Class.create(
  7 	/** @lends UserSettings.prototype */
  8 	{
  9 	/**
 10 	 * @constructs
 11 	 * @description Creates a new UserSettings instance
 12 	 * @param {Object} controller A reference to the Helioviewer application class
 13 	 */
 14 	initialize: function (controller) {
 15 		this.controller = controller;
 16 		
 17 		/**
 18 		 * @description Default user settings
 19 		 */
 20 		this._DEFAULTS = $H({
 21 			'obs-date'			: 1065312000000,
 22 			'zoom-level'		: this.controller.defaultZoomLevel,
 23 			'tile-layers'		: [{ tileAPI: "api/index.php", observatory: 'SOH', instrument: 'EIT', detector: 'EIT', measurement: '195' }],
 24 			'warn-zoom-level'	: false,
 25 			'warn-mouse-coords'	: false,
 26 			'event-icons'		: {
 27 				'VSOService::noaa':				'small-blue-circle',
 28 				'GOESXRayService::GOESXRay':	'small-green-diamond',
 29 				'VSOService::cmelist':			'small-yellow-square'
 30 			}
 31 		});
 32 
 33 		this.cookies = new CookieJar({
 34 			expires: 31536000, //1 year
 35 			path: '/'
 36 		});
 37 		
 38 		if (!this._exists()) {
 39 			this._loadDefaults();
 40 		}
 41 	},
 42 	
 43 	/**
 44 	 * @description Saves a specified setting
 45 	 * @param {String} key The setting to update
 46 	 * @param {JSON} value The new value for the setting
 47 	 */
 48 	set: function (key, value) {
 49 		if (this._validate(key, value)) {
 50 			this.cookies.put(key, value);
 51 		} else {
 52 			console.log("Ignoring invalid user-setting...");
 53 		}
 54 	},
 55 	
 56 	/**
 57 	 * @description Gets a specified setting
 58 	 * @param {String} key The setting to retrieve
 59 	 * @returns {JSON} The value of the desired setting
 60 	 */
 61 	get: function (key) {
 62 		return this.cookies.get(key);
 63 	},
 64 	
 65 	/**
 66 	 * @description Checks to see if user-settings cookies have been set.
 67 	 */
 68 	_exists: function () {
 69 		return (this.cookies.getKeys().length > 0);
 70 	},
 71 	
 72 	/**
 73 	 * @description Validates a setting (Currently checks observation date and zoom-level)
 74 	 * @param {String} setting The setting to be validated
 75 	 * @param {String} value The value of the setting to check
 76 	 */
 77 	_validate: function (setting, value) {
 78 		switch(setting) {
 79 			case "obs-date":
 80 				if (isNaN(value))
 81 					return false;
 82 				break;
 83 			case "zoom-level":
 84 				if ((isNaN(value)) || (value < this.controller.minZoomLevel) || (value > this.controller.maxZoomLevel))
 85 					return false;
 86 				break;
 87 			default:
 88 				break;		
 89 		}
 90 		return true;
 91 	},
 92 	
 93 	/**
 94 	 * @description Resets a single setting to it's default value
 95 	 * @param {String} setting The setting for which the default value should be loaded
 96 	 */
 97 	_resetSetting: function (setting) {
 98 		this.set(setting, this._getDefault(setting));		
 99 	},
100 	
101 	/**
102 	 * @description Gets the default value for a given setting
103 	 * @param {Object} setting
104 	 */
105 	_getDefault: function (setting) {
106 		return this._DEFAULTS.get(setting);
107 	},
108 	
109 	/**
110 	 * @description Loads defaults if cookies have not been set prior.
111 	 */
112 	_loadDefaults: function () {
113 		var self = this;
114 		this._DEFAULTS.each(function(setting) {
115 			self.set(setting.key, setting.value);
116 		});
117 	}
118 });
