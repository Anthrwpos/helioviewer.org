1 /**
  2  * @fileOverview Contains the class definition for an UserSettings class.
  3  * Syntax: Prototype
  4  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  5  * 
  6  * Keith 2009/07/01:
  7  *     There is currently a bug in the Prototype 1.6 series which prevents
  8  *     Firefox 3.5's native JSON functionality from working properly.
  9  *     The issue should be fixed in the upcoming 1.7 series.
 10  *     See: https://prototype.lighthouseapp.com/projects/8886/tickets/730
 11  */
 12 /*global Class, CookieJar, $H */
 13 var UserSettings = Class.create(
 14 	/** @lends UserSettings.prototype */
 15 	{
 16 	/**
 17 	 * @constructs
 18 	 * @description Creates a new UserSettings instance.
 19 	 * @param {Object} controller A reference to the Helioviewer application class
 20 	 */
 21 	initialize: function (controller) {
 22 		this.controller = controller;
 23 				
 24         // Initialize storage
 25         this._initStorage();
 26 	},
 27 
 28 	/**
 29 	 * @description Saves a specified setting
 30 	 * @param {String} key The setting to update
 31 	 * @param {JSON} value The new value for the setting
 32 	 */
 33 	set: function (key, value) {
 34         if (this._validate(key, value)) {
 35             // Update settings
 36             this.settings[key] = value;
 37             
 38             // localStorage + native JSON
 39             if (this.controller.support.localStorage && this.controller.support.nativeJSON)
 40                 localStorage.setObject("settings", this.settings);
 41             
 42             // localStorage only
 43             else if (this.controller.support.localStorage)
 44                 localStorage.setItem("settings", Object.toJSON(this.settings));
 45             
 46             // cookies
 47             else            
 48                 this.cookies.put("settings", this.settings);
 49         }
 50 	},
 51 	
 52 	/**
 53 	 * @description Gets a specified setting
 54 	 * @param {String} key The setting to retrieve
 55 	 * @returns {JSON} The value of the desired setting
 56 	 */
 57 	get: function (key) {
 58         return this.settings[key];
 59 	},
 60 	
 61     /**
 62      * @description Builds a URL for the current view
 63      * @param {Object} level
 64      * @TODO: Add support for viewport offset, event layers, opacity
 65      * @TODO: Make into a static method for use by Jetpack, etc? http://www.ruby-forum.com/topic/154386
 66      */
 67     toURL: function () {
 68         var url, date, zoom, imageLayers;
 69         
 70         // Base URL
 71         url = "http://www.helioviewer.org/?";
 72         
 73         // Add timestamp
 74         date = this.controller.date.toISOString();
 75         
 76   		// Work-around: In Firefox 3.1+, Date.toISOString() Returns single-quoted strings
 77 		// http://code.google.com/p/datejs/issues/detail?id=54
 78 		if (navigator.userAgent.search(/3\.[1-9]/) == -1)
 79     		date = date.slice(1, -1);
 80 	
 81         // Add zoomlevel
 82         zoom = this.controller.getImageScale();
 83         
 84         // Image layers
 85         imageLayers = this.controller.tileLayers.toString();
 86         
 87         // Build URL
 88         url += "date=" + date + "&imageScale=" + zoom + "&imageLayers=" + imageLayers;
 89 
 90         return url;
 91     },
 92 
 93     /**
 94      * @description Breaks up a given layer identifier (e.g. SOH_LAS_0C2_0WL) into
 95      *              it's component parts and returns a javascript representation.
 96 	 * @param {String} The layer identifier as an underscore-concatenated string
 97 	 * @returns {Object} a simple javascript object representing the layer params
 98 	 * @see TileLayer.toString
 99      */
100     parseLayerString: function (str) {
101         var params = str.split(",");
102         return {
103             observatory: params[0],
104             instrument : params[1],
105             detector   : params[2],
106             measurement: params[3],
107             visible    : Boolean(parseInt(params[4])),
108             opacity    : parseInt(params[5]),
109             server     : params[6] || null
110         };
111     },
112 
113 	/**
114 	 * @description Checks to see if there are any existing stored user settings
115 	 */
116 	_exists: function () {
117         return (this.controller.support.localStorage ? (localStorage.length > 0) : (this.cookies.getKeys().length > 0));
118 	},
119     	
120     /**
121      * @description Decides on best storage format to use, and initializes it
122      */
123     _initStorage: function() {
124         // Initialize CookieJar if localStorage isn't supported
125         if (!this.controller.support.localStorage) {
126             this.cookies = new CookieJar({
127                 expires: 31536000, //1 year
128                 path: '/'
129             });
130         }
131         
132         // If no stored user settings exist, load defaults
133 		if (!this._exists())
134 			this._loadDefaults();
135         else
136             this._loadSavedSettings();
137             
138         // If version is out of date, reset settings
139         if (this.get('version') != this.controller.version)
140             this._loadDefaults();
141     },
142     
143 	/**
144 	 * @description Validates a setting (Currently checks observation date and zoomLevel)
145 	 * @param {String} setting The setting to be validated
146 	 * @param {String} value The value of the setting to check
147 	 */
148 	_validate: function (setting, value) {
149 		switch (setting) {
150 		case "date":
151 			if (isNaN(value)) {
152 				return false;
153 			}
154 			break;
155 		case "zoomLevel":
156 			if ((isNaN(value)) || (value < this.controller.minZoomLevel) || (value > this.controller.maxZoomLevel)) {
157 				return false;
158 			}
159 			break;
160 		default:
161 			break;		
162 		}
163 		return true;
164 	},
165 	
166 	/**
167 	 * @description Loads defaults user settings
168 	 */
169 	_loadDefaults: function () {
170         var defaults = this._getDefaults();
171         
172         if (this.controller.support.localStorage) {
173             localStorage.clear()
174             
175             if (this.controller.support.nativeJSON)
176                 localStorage.setObject("settings", defaults);
177             else 
178                 localStorage.setItem("settings", Object.toJSON(defaults));
179         }
180         else 
181             this.cookies.put("settings", defaults);
182             
183         this.settings = defaults;
184     },
185     
186 	/**
187      * Retrieves the saved user settings and saved them locally
188      */
189     _loadSavedSettings: function () {
190         // If native JSON is supported, return value directly
191         if (this.controller.support.localStorage && this.controller.support.nativeJSON)
192             this.settings = localStorage.getObject("settings");
193 
194         else if (this.controller.support.localStorage)
195             this.settings = localStorage.getItem("settings").evalJSON();
196             
197         // Otherwise, check type and return
198         else
199             this.settings = this.cookies.get("settings");
200     },
201     
202     /**
203      * @description Creates a hash containing the default settings to use
204      */
205     _getDefaults: function () {
206 		return {
207 			date			: getUTCTimestamp(this.controller.defaultObsTime),
208 			zoomLevel		: this.controller.defaultZoomLevel,
209             version         : this.controller.version,
210 			warnZoomLevel	: false,
211 			warnMouseCoords : false,
212            	tileLayers		: [{
213                 server     : "api/index.php",
214                 observatory: 'SOH',
215                 instrument : 'EIT',
216                 detector   : 'EIT',
217                 measurement: '304',
218                 visible    : true,
219                 opacity    : 100
220              }],
221 			eventIcons      : {
222 				'VSOService::noaa'         : 'small-blue-circle',
223 				'GOESXRayService::GOESXRay': 'small-green-diamond',
224 				'VSOService::cmelist'      : 'small-yellow-square'
225 			}
226 		};
227     } 
228 });
