1 /**
  2  * @fileOverview Contains the class definition for an UserSettings class.
  3  * Syntax: Prototype, jQuery
  4  * @author <a href="mailto:vincent.k.hughitt@nasa.gov">Keith Hughitt</a>
  5  * @TODO Store either last version of HV or Cookie version to use enable forced updates
  6  */
  7 /*global Class, CookieJar, $H */
  8 var UserSettings = Class.create(
  9 	/** @lends UserSettings.prototype */
 10 	{
 11 	/**
 12 	 * @constructs
 13 	 * @description Creates a new UserSettings instance. Because all cookie data is stored as strings, numeric types
 14 	 *              must be specified (e.g. in _INTGER_PARAMS) so that the application knows to parse them as such.
 15 	 * @param {Object} controller A reference to the Helioviewer application class
 16 	 */
 17 	initialize: function (controller) {
 18 		this.controller = controller;
 19 		
 20 		/**
 21 		 * @description Default user settings
 22 		 */
 23 		this._DEFAULTS = $H({
 24 			'obs-date'			: 1065312000000,
 25 			'zoom-level'		: this.controller.defaultZoomLevel,
 26 			'tile-layers'		: [{ tileAPI: "api/index.php", server: 1, observatory: 'SOH', instrument: 'EIT', detector: 'EIT', measurement: '304' }],
 27             'version'           : this.controller.version,
 28 			'warn-zoom-level'	: false,
 29 			'warn-mouse-coords'	: false,
 30 			'event-icons'		: {
 31 				'VSOService::noaa':				'small-blue-circle',
 32 				'GOESXRayService::GOESXRay':	'small-green-diamond',
 33 				'VSOService::cmelist':			'small-yellow-square'
 34 			}
 35 		});
 36 		
 37 		this._INTEGER_PARAMS = $A(['obs-date', 'zoom-level']);
 38 		this._FLOAT_PARAMS   = $A([]);
 39 		this._BOOLEAN_PARAMS = $A(['warn-zoom-level', 'warn-mouse-coords']);
 40 
 41 
 42 		this.cookies = new CookieJar({
 43 			expires: 31536000, //1 year
 44 			path: '/'
 45 		});
 46 		
 47 		if ((!this._exists()) || (this.get('version') != this.controller.version)) {
 48 			this._loadDefaults();
 49 		}
 50 	},
 51 	
 52 	/**
 53 	 * @description Saves a specified setting
 54 	 * @param {String} key The setting to update
 55 	 * @param {JSON} value The new value for the setting
 56 	 */
 57 	set: function (key, value) {
 58 		if (this._validate(key, value)) {
 59 			this.cookies.put(key, value);
 60 		} else {
 61 			//console.log("Ignoring invalid user-setting...");
 62 		}
 63 	},
 64 	
 65 	/**
 66 	 * @description Gets a specified setting
 67 	 * @param {String} key The setting to retrieve
 68 	 * @returns {JSON} The value of the desired setting
 69 	 */
 70 	get: function (key) {
 71 		// Parse numeric types
 72 		if (this._INTEGER_PARAMS.include(key)) {
 73 			return parseInt(this.cookies.get(key));
 74 		}
 75 		else if (this._FLOAT_PARAMS.include(key)) {
 76 			return parseFloat(this.cookies.get(key));
 77 		}
 78 		else if (this._BOOLEAN_PARAMS.include(key)) {
 79 			return this.cookies.get(key) == "true" ? true : false;
 80 		}
 81 		return this.cookies.get(key);
 82 	},
 83 	
 84 	/**
 85 	 * @description Checks to see if user-settings cookies have been set.
 86 	 */
 87 	_exists: function () {
 88 		return (this.cookies.getKeys().length > 0);
 89 	},
 90 	
 91 	/**
 92 	 * @description Validates a setting (Currently checks observation date and zoom-level)
 93 	 * @param {String} setting The setting to be validated
 94 	 * @param {String} value The value of the setting to check
 95 	 */
 96 	_validate: function (setting, value) {
 97 		switch (setting) {
 98 		case "obs-date":
 99 			if (isNaN(value)) {
100 				return false;
101 			}
102 			break;
103 		case "zoom-level":
104 			if ((isNaN(value)) || (value < this.controller.minZoomLevel) || (value > this.controller.maxZoomLevel)) {
105 				return false;
106 			}
107 			break;
108 		default:
109 			break;		
110 		}
111 		return true;
112 	},
113 	
114 	/**
115 	 * @description Resets a single setting to it's default value
116 	 * @param {String} setting The setting for which the default value should be loaded
117 	 */
118 	_resetSetting: function (setting) {
119 		this.set(setting, this._getDefault(setting));		
120 	},
121 	
122 	/**
123 	 * @description Gets the default value for a given setting
124 	 * @param {Object} setting
125 	 */
126 	_getDefault: function (setting) {
127 		return this._DEFAULTS.get(setting);
128 	},
129 	
130 	/**
131 	 * @description Loads defaults if cookies have not been set prior.
132 	 */
133 	_loadDefaults: function () {
134 		var self = this;
135 		this._DEFAULTS.each(function (setting) {
136 			self.set(setting.key, setting.value);
137 		});
138 	}
139 });
