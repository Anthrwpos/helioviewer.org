1 /**
  2  * @fileOverview Contains the class definition for an UserSettings class.
  3  * @author <a href="mailto:keith.hughitt@nasa.gov">Keith Hughitt</a>
  4  * 
  5  * Keith 2009/07/01:

  7  *     Firefox 3.5's native JSON functionality from working properly.

  9  *     See: https://prototype.lighthouseapp.com/projects/8886/tickets/730
 10  */
 11 /*jslint browser: true, white: true, onevar: true, undef: true, nomen: false, eqeqeq: true, plusplus: true, 
 12 bitwise: true, regexp: true, strict: true, newcap: true, immed: true, maxlen: 120, sub: true */
 13 /*global Class, CookieJar, $, localStorage, getUTCTimestamp */
 14 "use strict";
 15 var UserSettings = Class.extend(
 16     /** @lends UserSettings.prototype */
 17     {
 18     /**
 19      * @constructs
 20      * @description Creates a new UserSettings instance.
 21      * @param {Object} controller A reference to the Helioviewer application class
 22      */
 23     init: function (controller) {
 24         this.controller = controller;
 25                 
 26         // Initialize storage
 27         this._initStorage();
 28     },
 29 
 30     /**
 31      * @description Saves a specified setting
 32      * @param {String} key The setting to update
 33      * @param {JSON} value The new value for the setting
 34      */
 35     set: function (key, value) {
 36         if (this._validate(key, value)) {
 37             // Update settings
 38             this.settings[key] = value;
 39             
 40             // localStorage + native JSON
 41             if ($.support.localStorage && $.support.nativeJSON) {
 42                 localStorage.setObject("settings", this.settings);
 43             }
 44             
 45             // localStorage only
 46             else if ($.support.localStorage) {
 47                 localStorage.setItem("settings", $.toJSON(this.settings));
 48             }
 49             
 50             // cookies
 51             else {         
 52                 this.cookies.set("settings", this.settings);
 53             }
 54         }
 55     },
 56     
 57     /**
 58      * @description Gets a specified setting
 59      * @param {String} key The setting to retrieve
 60      * @returns {JSON} The value of the desired setting
 61      */
 62     get: function (key) {
 63         return this.settings[key];
 64     },
 65     
 66     /**
 67      * @description Builds a URL for the current view
 68      * @param {Object} level
 69      * @TODO: Add support for viewport offset, event layers, opacity
 70      * @TODO: Make into a static method for use by Jetpack, etc? http://www.ruby-forum.com/topic/154386
 71      */
 72     toURL: function () {
 73         var url, date, zoom, imageLayers;
 74         
 75         // Base URL
 76         //url = "http://www.helioviewer.org/?";
 77         url = this.controller.rootURL + "/?";
 78         
 79         // Add timestamp
 80         date = this.controller.date.toISOString();
 81     
 82         // Add zoomlevel
 83         zoom = this.controller.getImageScale();
 84         
 85         // Image layers
 86         imageLayers = this.controller.tileLayers.toString();
 87         
 88         // Build URL
 89         url += "date=" + date + "&imageScale=" + zoom + "&imageLayers=" + imageLayers;
 90 
 91         return url;
 92     },
 93 
 94     /**
 95      * @description Breaks up a given layer identifier (e.g. SOHO,LASCO,C2,white light) into
 96      *              its component parts and returns a javascript representation.
 97      * @param {String} The layer identifier as an underscore-concatenated string
 98      * @returns {Object} a simple javascript object representing the layer params
 99      * @see TileLayer.toString
100      */
101     parseLayerString: function (str) {
102         var params = str.split(",");
103         return {
104             observatory: params[0],
105             instrument : params[1],
106             detector   : params[2],
107             measurement: params[3],
108             visible    : Boolean(parseInt(params[4], 10)),
109             opacity    : parseInt(params[5], 10),
110             server     : parseInt(params[6], 10) || 0
111         };
112     },
113 
114     /**
115      * @description Checks to see if there are any existing stored user settings
116      */
117     _exists: function () {
118         //return ($.support.localStorage ? (localStorage.getItem("settings") !== null) 
119         // : (this.cookies.getKeys().length > 0));
120         return ($.support.localStorage ? (localStorage.getItem("settings") !== null) 
121                 : (this.cookies.toString().length > 2));
122     },
123         
124     /**
125      * @description Decides on best storage format to use, and initializes it
126      */
127     _initStorage: function () {
128         // Initialize CookieJar if localStorage isn't supported
129         if (!$.support.localStorage) {
130             this.cookies = $.cookieJar("settings");
131         }
132         
133         // If no stored user settings exist, load defaults
134         if (!this._exists()) {
135             this._loadDefaults();
136         }
137         else {
138             this._loadSavedSettings();
139         }
140             
141         // If version is out of date, reset settings
142         if (this.get('version') !== this.controller.version) {
143             this._loadDefaults();
144         }
145     },
146     
147     /**
148      * @description Validates a setting (Currently checks observation date and zoomLevel)
149      * @param {String} setting The setting to be validated
150      * @param {String} value The value of the setting to check
151      */
152     _validate: function (setting, value) {
153         switch (setting) {
154         case "date":
155             if (isNaN(value)) {
156                 return false;
157             }
158             break;
159         case "zoomLevel":
160             if ((isNaN(value)) || (value < this.controller.minZoomLevel) || (value > this.controller.maxZoomLevel)) {
161                 return false;
162             }
163             break;
164         default:
165             break;        
166         }
167         return true;
168     },
169     
170     /**
171      * @description Loads defaults user settings
172      */
173     _loadDefaults: function () {
174         var defaults = this._getDefaults();
175         
176         if ($.support.localStorage) {
177             localStorage.clear();
178             
179             if ($.support.nativeJSON) {
180                 localStorage.setObject("settings", defaults);
181             }
182             else { 
183                 localStorage.setItem("settings", $.toJSON(defaults));
184             }
185         }
186         else {
187             this.cookies.set("settings", defaults);
188         }
189             
190         this.settings = defaults;
191     },
192     
193     /**
194      * Retrieves the saved user settings and saved them locally
195      */
196     _loadSavedSettings: function () {
197         // If native JSON is supported, return value directly
198         if ($.support.localStorage && $.support.nativeJSON) {
199             this.settings = localStorage.getObject("settings");
200         }
201 
202         else if ($.support.localStorage) {
203             this.settings = $.parseJSON(localStorage.getItem("settings"));
204         }
205             
206         // Otherwise, check type and return
207         else {
208             this.settings = this.cookies.get("settings");
209         }
210     },
211     
212     /**
213      * @description Creates a hash containing the default settings to use
214      */
215     _getDefaults: function () {
216         return {
217             date            : getUTCTimestamp(this.controller.defaultObsTime),
218             zoomLevel        : this.controller.defaultZoomLevel,
219             version         : this.controller.version,
220             warnZoomLevel    : false,
221             warnMouseCoords : false,
222             tileLayers        : [{
223                 server     : 0,
224                 observatory: 'SOHO',
225                 instrument : 'EIT',
226                 detector   : 'EIT',
227                 measurement: '304',
228                 visible    : true,
229                 opacity    : 100
230             }],
231             eventIcons      : {
232                 'VSOService::noaa'         : 'small-blue-circle',
233                 'GOESXRayService::GOESXRay': 'small-green-diamond',
234                 'VSOService::cmelist'      : 'small-yellow-square'
235             }
236         };
237     } 
238 });
